50     std::string GetCertTitle(CERTCertificate* cert) {
51       std::string rv;
52       if (cert->nickname) {
53         rv = cert->nickname;
54       } else {
55         char* cn = CERT_GetCommonName(&cert->subject);
56         if (cn) {
57           rv = cn;
58           PORT_Free(cn);
59         } else if (cert->subjectName) {
60           rv = cert->subjectName;
61         } else if (cert->emailAddr) {
62           rv = cert->emailAddr;
63         }
64       }
65       // TODO(mattm): Should we return something other than an empty string when all
66       // the checks fail?
67       return rv;
68     }
69     
70     std::string GetCertTokenName(CERTCertificate* cert) {
71       std::string token;
72       if (cert->slot) {
73         token = PK11_GetTokenName(cert->slot);
74       }
75       return token;
76     }
