62     // Based on nsNSSCertificateDB::handleCACertDownload, minus the UI bits.
63     bool ImportCACerts(const net::CertificateList& certificates,
64                        net::X509Certificate* root,
65                        net::NSSCertDatabase::TrustBits trustBits,
66                        net::NSSCertDatabase::ImportCertFailureList* not_imported) {
67       if (certificates.empty() || !root)
68         return false;
69     
70       crypto::ScopedPK11Slot slot(crypto::GetPublicNSSKeySlot());
71       if (!slot.get()) {
72         LOG(ERROR) << "Couldn't get internal key slot!";
73         return false;
74       }
75     
76       // Mozilla had some code here to check if a perm version of the cert exists
77       // already and use that, but CERT_NewTempCertificate actually does that
78       // itself, so we skip it here.
79     
80       if (!CERT_IsCACert(root->os_cert_handle(), NULL)) {
81         not_imported->push_back(net::NSSCertDatabase::ImportCertFailure(
82             root, net::ERR_IMPORT_CA_CERT_NOT_CA));
83       } else if (root->os_cert_handle()->isperm) {
84         // Mozilla just returns here, but we continue in case there are other certs
85         // in the list which aren't already imported.
86         // TODO(mattm): should we set/add trust if it differs from the present
87         // settings?
88         not_imported->push_back(net::NSSCertDatabase::ImportCertFailure(
89             root, net::ERR_IMPORT_CERT_ALREADY_EXISTS));
90       } else {
91         // Mozilla uses CERT_AddTempCertToPerm, however it is privately exported,
92         // and it doesn't take the slot as an argument either.  Instead, we use
93         // PK11_ImportCert and CERT_ChangeCertTrust.
94         SECStatus srv = PK11_ImportCert(
95             slot.get(),
96             root->os_cert_handle(),
97             CK_INVALID_HANDLE,
98             root->GetDefaultNickname(net::CA_CERT).c_str(),
99             PR_FALSE /* includeTrust (unused) */);
100         if (srv != SECSuccess) {
101           LOG(ERROR) << "PK11_ImportCert failed with error " << PORT_GetError();
102           return false;
103         }
104         if (!SetCertTrust(root, net::CA_CERT, trustBits))
105           return false;
106       }
107     
108       PRTime now = PR_Now();
109       // Import additional delivered certificates that can be verified.
110       // This is sort of merged in from Mozilla's ImportValidCACertsInList.  Mozilla
111       // uses CERT_FilterCertListByUsage to filter out non-ca certs, but we want to
112       // keep using X509Certificates, so that we can use them to build the
113       // |not_imported| result.  So, we keep using our net::CertificateList and
114       // filter it ourself.
115       for (size_t i = 0; i < certificates.size(); i++) {
116         const scoped_refptr<net::X509Certificate>& cert = certificates[i];
117         if (cert == root) {
118           // we already processed that one
119           continue;
120         }
121     
122         // Mozilla uses CERT_FilterCertListByUsage(certList, certUsageAnyCA,
123         // PR_TRUE).  Afaict, checking !CERT_IsCACert on each cert is equivalent.
124         if (!CERT_IsCACert(cert->os_cert_handle(), NULL)) {
125           not_imported->push_back(net::NSSCertDatabase::ImportCertFailure(
126               cert, net::ERR_IMPORT_CA_CERT_NOT_CA));
127           VLOG(1) << "skipping cert (non-ca)";
128           continue;
129         }
130     
131         if (cert->os_cert_handle()->isperm) {
132           not_imported->push_back(net::NSSCertDatabase::ImportCertFailure(
133               cert, net::ERR_IMPORT_CERT_ALREADY_EXISTS));
134           VLOG(1) << "skipping cert (perm)";
135           continue;
136         }
137     
138         if (CERT_VerifyCert(CERT_GetDefaultCertDB(), cert->os_cert_handle(),
139             PR_TRUE, certUsageVerifyCA, now, NULL, NULL) != SECSuccess) {
140           // TODO(mattm): use better error code (map PORT_GetError to an appropriate
141           // error value).  (maybe make MapSecurityError or MapCertErrorToCertStatus
142           // public.)
143           not_imported->push_back(net::NSSCertDatabase::ImportCertFailure(
144               cert, net::ERR_FAILED));
145           VLOG(1) << "skipping cert (verify) " << PORT_GetError();
146           continue;
147         }
148     
149         // Mozilla uses CERT_ImportCerts, which doesn't take a slot arg.  We use
150         // PK11_ImportCert instead.
151         SECStatus srv = PK11_ImportCert(
152             slot.get(),
153             cert->os_cert_handle(),
154             CK_INVALID_HANDLE,
155             cert->GetDefaultNickname(net::CA_CERT).c_str(),
156             PR_FALSE /* includeTrust (unused) */);
157         if (srv != SECSuccess) {
158           LOG(ERROR) << "PK11_ImportCert failed with error " << PORT_GetError();
159           // TODO(mattm): Should we bail or continue on error here?  Mozilla doesn't
160           // check error code at all.
161           not_imported->push_back(net::NSSCertDatabase::ImportCertFailure(
162               cert, net::ERR_IMPORT_CA_CERT_FAILED));
163         }
164       }
165     
166       // Any errors importing individual certs will be in listed in |not_imported|.
167       return true;
168     }

170     // Based on nsNSSCertificateDB::ImportServerCertificate.
171     bool ImportServerCert(
172         const net::CertificateList& certificates,
173         net::NSSCertDatabase::TrustBits trustBits,
174         net::NSSCertDatabase::ImportCertFailureList* not_imported) {
175       if (certificates.empty())
176         return false;
177     
178       crypto::ScopedPK11Slot slot(crypto::GetPublicNSSKeySlot());
179       if (!slot.get()) {
180         LOG(ERROR) << "Couldn't get internal key slot!";
181         return false;
182       }
183     
184       for (size_t i = 0; i < certificates.size(); ++i) {
185         const scoped_refptr<net::X509Certificate>& cert = certificates[i];
186     
187         // Mozilla uses CERT_ImportCerts, which doesn't take a slot arg.  We use
188         // PK11_ImportCert instead.
189         SECStatus srv = PK11_ImportCert(
190             slot.get(),
191             cert->os_cert_handle(),
192             CK_INVALID_HANDLE,
193             cert->GetDefaultNickname(net::SERVER_CERT).c_str(),
194             PR_FALSE /* includeTrust (unused) */);
195         if (srv != SECSuccess) {
196           LOG(ERROR) << "PK11_ImportCert failed with error " << PORT_GetError();
197           not_imported->push_back(net::NSSCertDatabase::ImportCertFailure(
198               cert, net::ERR_IMPORT_SERVER_CERT_FAILED));
199           continue;
200         }
201       }
202     
203       SetCertTrust(certificates[0].get(), net::SERVER_CERT, trustBits);
204       // TODO(mattm): Report SetCertTrust result?  Putting in not_imported
205       // wouldn't quite match up since it was imported...
206     
207       // Any errors importing individual certs will be in listed in |not_imported|.
208       return true;
209     }

211     // Based on nsNSSCertificateDB::SetCertTrust.
212     bool
213     SetCertTrust(const net::X509Certificate* cert,
214                  net::CertType type,
215                  net::NSSCertDatabase::TrustBits trustBits)
216     {
217       const unsigned kSSLTrustBits = net::NSSCertDatabase::TRUSTED_SSL |
218           net::NSSCertDatabase::DISTRUSTED_SSL;
219       const unsigned kEmailTrustBits = net::NSSCertDatabase::TRUSTED_EMAIL |
220           net::NSSCertDatabase::DISTRUSTED_EMAIL;
221       const unsigned kObjSignTrustBits = net::NSSCertDatabase::TRUSTED_OBJ_SIGN |
222           net::NSSCertDatabase::DISTRUSTED_OBJ_SIGN;
223       if ((trustBits & kSSLTrustBits) == kSSLTrustBits ||
224           (trustBits & kEmailTrustBits) == kEmailTrustBits ||
225           (trustBits & kObjSignTrustBits) == kObjSignTrustBits) {
226         LOG(ERROR) << "SetCertTrust called with conflicting trust bits "
227                    << trustBits;
228         NOTREACHED();
229         return false;
230       }
231     
232       SECStatus srv;
233       CERTCertificate *nsscert = cert->os_cert_handle();
234       if (type == net::CA_CERT) {
235         // Note that we start with CERTDB_VALID_CA for default trust and explicit
236         // trust, but explicitly distrusted usages will be set to
237         // CERTDB_TERMINAL_RECORD only.
238         CERTCertTrust trust = {CERTDB_VALID_CA, CERTDB_VALID_CA, CERTDB_VALID_CA};
239     
240         if (trustBits & net::NSSCertDatabase::DISTRUSTED_SSL)
241           trust.sslFlags = CERTDB_TERMINAL_RECORD;
242         else if (trustBits & net::NSSCertDatabase::TRUSTED_SSL)
243           trust.sslFlags |= CERTDB_TRUSTED_CA | CERTDB_TRUSTED_CLIENT_CA;
244     
245         if (trustBits & net::NSSCertDatabase::DISTRUSTED_EMAIL)
246           trust.emailFlags = CERTDB_TERMINAL_RECORD;
247         else if (trustBits & net::NSSCertDatabase::TRUSTED_EMAIL)
248           trust.emailFlags |= CERTDB_TRUSTED_CA | CERTDB_TRUSTED_CLIENT_CA;
249     
250         if (trustBits & net::NSSCertDatabase::DISTRUSTED_OBJ_SIGN)
251           trust.objectSigningFlags = CERTDB_TERMINAL_RECORD;
252         else if (trustBits & net::NSSCertDatabase::TRUSTED_OBJ_SIGN)
253           trust.objectSigningFlags |= CERTDB_TRUSTED_CA | CERTDB_TRUSTED_CLIENT_CA;
254     
255         srv = CERT_ChangeCertTrust(CERT_GetDefaultCertDB(), nsscert, &trust);
256       } else if (type == net::SERVER_CERT) {
257         CERTCertTrust trust = {0};
258         // We only modify the sslFlags, so copy the other flags.
259         CERT_GetCertTrust(nsscert, &trust);
260         trust.sslFlags = 0;
261     
262         if (trustBits & net::NSSCertDatabase::DISTRUSTED_SSL)
263           trust.sslFlags |= CERTDB_TERMINAL_RECORD;
264         else if (trustBits & net::NSSCertDatabase::TRUSTED_SSL)
265           trust.sslFlags |= CERTDB_TRUSTED | CERTDB_TERMINAL_RECORD;
266     
267         srv = CERT_ChangeCertTrust(CERT_GetDefaultCertDB(), nsscert, &trust);
268       } else {
269         // ignore user and email/unknown certs
270         return true;
271       }
272       if (srv != SECSuccess)
273         LOG(ERROR) << "SetCertTrust failed with error " << PORT_GetError();
274       return srv == SECSuccess;
275     }
