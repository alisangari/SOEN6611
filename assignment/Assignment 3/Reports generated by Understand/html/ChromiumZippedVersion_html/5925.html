141     // Test if a template specialization is an instantiation.
142     static bool IsTemplateInstantiation(CXXRecordDecl* record) {
143       ClassTemplateSpecializationDecl* spec =
144           dyn_cast<ClassTemplateSpecializationDecl>(record);
145       if (!spec)
146         return false;
147       switch (spec->getTemplateSpecializationKind()) {
148         case TSK_ImplicitInstantiation:
149         case TSK_ExplicitInstantiationDefinition:
150           return true;
151         case TSK_Undeclared:
152         case TSK_ExplicitSpecialization:
153           return false;
154         // TODO: unsupported cases.
155         case TSK_ExplicitInstantiationDeclaration:
156           return false;
157       }
158       assert(false && "Unknown template specialization kind");
159     }

162     class CollectVisitor : public RecursiveASTVisitor<CollectVisitor> {
163      public:
164       CollectVisitor() {}
165     
166       RecordVector& record_decls() { return record_decls_; }
167       MethodVector& trace_decls() { return trace_decls_; }
168     
169       bool shouldVisitTemplateInstantiations() { return false; }
170     
171       // Collect record declarations, including nested declarations.
172       bool VisitCXXRecordDecl(CXXRecordDecl* record) {
173         if (record->hasDefinition() && record->isCompleteDefinition())
174           record_decls_.push_back(record);
175         return true;
176       }
177     
178       // Collect tracing method definitions, but don't traverse method bodies.
179       bool TraverseCXXMethodDecl(CXXMethodDecl* method) {
180         if (method->isThisDeclarationADefinition() && Config::IsTraceMethod(method))
181           trace_decls_.push_back(method);
182         return true;
183       }

185      private:
186       RecordVector record_decls_;
187       MethodVector trace_decls_;

190     // This visitor checks that a finalizer method does not have invalid access to
191     // fields that are potentially finalized. A potentially finalized field is
192     // either a Member, a heap-allocated collection or an off-heap collection that
193     // contains Members.  Invalid uses are currently identified as passing the field
194     // as the argument of a procedure call or using the -> or [] operators on it.
195     class CheckFinalizerVisitor
196         : public RecursiveASTVisitor<CheckFinalizerVisitor> {
197      private:
198       // Simple visitor to determine if the content of a field might be collected
199       // during finalization.
200       class MightBeCollectedVisitor : public EdgeVisitor {
201        public:
202         MightBeCollectedVisitor() : might_be_collected_(false) {}
203         bool might_be_collected() { return might_be_collected_; }
204         void VisitMember(Member* edge) override { might_be_collected_ = true; }
205         void VisitCollection(Collection* edge) override {
206           if (edge->on_heap()) {
207             might_be_collected_ = !edge->is_root();
208           } else {
209             edge->AcceptMembers(this);
210           }
211         }
212     
213        private:
214         bool might_be_collected_;
215       };
216     
217      public:
218       typedef std::vector<std::pair<MemberExpr*, FieldPoint*> > Errors;
219     
220       CheckFinalizerVisitor(RecordCache* cache)
221           : blacklist_context_(false), cache_(cache) {}
222     
223       Errors& finalized_fields() { return finalized_fields_; }
224     
225       bool WalkUpFromCXXOperatorCallExpr(CXXOperatorCallExpr* expr) {
226         // Only continue the walk-up if the operator is a blacklisted one.
227         switch (expr->getOperator()) {
228           case OO_Arrow:
229           case OO_Subscript:
230             this->WalkUpFromCallExpr(expr);
231           default:
232             return true;
233         }
234       }
235     
236       // We consider all non-operator calls to be blacklisted contexts.
237       bool WalkUpFromCallExpr(CallExpr* expr) {
238         bool prev_blacklist_context = blacklist_context_;
239         blacklist_context_ = true;
240         for (size_t i = 0; i < expr->getNumArgs(); ++i)
241           this->TraverseStmt(expr->getArg(i));
242         blacklist_context_ = prev_blacklist_context;
243         return true;
244       }
245     
246       bool VisitMemberExpr(MemberExpr* member) {
247         FieldDecl* field = dyn_cast<FieldDecl>(member->getMemberDecl());
248         if (!field)
249           return true;
250     
251         RecordInfo* info = cache_->Lookup(field->getParent());
252         if (!info)
253           return true;
254     
255         RecordInfo::Fields::iterator it = info->GetFields().find(field);
256         if (it == info->GetFields().end())
257           return true;
258     
259         if (blacklist_context_ && MightBeCollected(&it->second))
260           finalized_fields_.push_back(std::make_pair(member, &it->second));
261         return true;
262       }
263     
264       bool MightBeCollected(FieldPoint* point) {
265         MightBeCollectedVisitor visitor;
266         point->edge()->Accept(&visitor);
267         return visitor.might_be_collected();
268       }
269     
270      private:
271       bool blacklist_context_;
272       Errors finalized_fields_;
273       RecordCache* cache_;
274     };

299     // This visitor checks a tracing method by traversing its body.
300     // - A member field is considered traced if it is referenced in the body.
301     // - A base is traced if a base-qualified call to a trace method is found.
302     class CheckTraceVisitor : public RecursiveASTVisitor<CheckTraceVisitor> {
303      public:
304       CheckTraceVisitor(CXXMethodDecl* trace, RecordInfo* info)
305           : trace_(trace), info_(info) {}
306     
307       // Allow recursive traversal by using VisitMemberExpr.
308       bool VisitMemberExpr(MemberExpr* member) {
309         // If this member expression references a field decl, mark it as traced.
310         if (FieldDecl* field = dyn_cast<FieldDecl>(member->getMemberDecl())) {
311           if (IsTemplateInstantiation(info_->record())) {
312             // Pointer equality on fields does not work for template instantiations.
313             // The trace method refers to fields of the template definition which
314             // are different from the instantiated fields that need to be traced.
315             const string& name = field->getNameAsString();
316             for (RecordInfo::Fields::iterator it = info_->GetFields().begin();
317                  it != info_->GetFields().end();
318                  ++it) {
319               if (it->first->getNameAsString() == name) {
320                 MarkTraced(it);
321                 break;
322               }
323             }
324           } else {
325             RecordInfo::Fields::iterator it = info_->GetFields().find(field);
326             if (it != info_->GetFields().end())
327               MarkTraced(it);
328           }
329           return true;
330         }
331     
332         // If this is a weak callback function we only check field tracing.
333         if (IsWeakCallback())
334           return true;
335     
336         // For method calls, check tracing of bases and other special GC methods.
337         if (CXXMethodDecl* fn = dyn_cast<CXXMethodDecl>(member->getMemberDecl())) {
338           const string& name = fn->getNameAsString();
339           // Check weak callbacks.
340           if (name == kRegisterWeakMembersName) {
341             if (fn->isTemplateInstantiation()) {
342               const TemplateArgumentList& args =
343                   *fn->getTemplateSpecializationInfo()->TemplateArguments;
344               // The second template argument is the callback method.
345               if (args.size() > 1 &&
346                   args[1].getKind() == TemplateArgument::Declaration) {
347                 if (FunctionDecl* callback =
348                         dyn_cast<FunctionDecl>(args[1].getAsDecl())) {
349                   if (callback->hasBody()) {
350                     CheckTraceVisitor nested_visitor(info_);
351                     nested_visitor.TraverseStmt(callback->getBody());
352                   }
353                 }
354               }
355             }
356             return true;
357           }
358     
359           // Currently, a manually dispatched class cannot have mixin bases (having
360           // one would add a vtable which we explicitly check against). This means
361           // that we can only make calls to a trace method of the same name. Revisit
362           // this if our mixin/vtable assumption changes.
363           if (Config::IsTraceMethod(fn) &&
364               fn->getName() == trace_->getName() &&
365               member->hasQualifier()) {
366             if (const Type* type = member->getQualifier()->getAsType()) {
367               if (CXXRecordDecl* decl = type->getAsCXXRecordDecl()) {
368                 RecordInfo::Bases::iterator it = info_->GetBases().find(decl);
369                 if (it != info_->GetBases().end())
370                   it->second.MarkTraced();
371               }
372             }
373           }
374         }
375         return true;
376       }
377     
378      private:
379       // Nested checking for weak callbacks.
380       CheckTraceVisitor(RecordInfo* info) : trace_(0), info_(info) {}
381     
382       bool IsWeakCallback() { return !trace_; }
383     
384       void MarkTraced(RecordInfo::Fields::iterator it) {
385         // In a weak callback we can't mark strong fields as traced.
386         if (IsWeakCallback() && !it->second.edge()->IsWeakMember())
387           return;
388         it->second.MarkTraced();
389       }
390     
391       CXXMethodDecl* trace_;
392       RecordInfo* info_;
393     };

395     // This visitor checks that the fields of a class and the fields of
396     // its part objects don't define GC roots.
397     class CheckGCRootsVisitor : public RecursiveEdgeVisitor {
398      public:
399       typedef std::vector<FieldPoint*> RootPath;
400       typedef std::vector<RootPath> Errors;
401     
402       CheckGCRootsVisitor() {}
403     
404       Errors& gc_roots() { return gc_roots_; }
405     
406       bool ContainsGCRoots(RecordInfo* info) {
407         for (RecordInfo::Fields::iterator it = info->GetFields().begin();
408              it != info->GetFields().end();
409              ++it) {
410           current_.push_back(&it->second);
411           it->second.edge()->Accept(this);
412           current_.pop_back();
413         }
414         return !gc_roots_.empty();
415       }
416     
417       void VisitValue(Value* edge) override {
418         // TODO: what should we do to check unions?
419         if (edge->value()->record()->isUnion())
420           return;
421     
422         // If the value is a part object, then continue checking for roots.
423         for (Context::iterator it = context().begin();
424              it != context().end();
425              ++it) {
426           if (!(*it)->IsCollection())
427             return;
428         }
429         ContainsGCRoots(edge->value());
430       }
431     
432       void VisitPersistent(Persistent* edge) override {
433         gc_roots_.push_back(current_);
434       }
435     
436       void AtCollection(Collection* edge) override {
437         if (edge->is_root())
438           gc_roots_.push_back(current_);
439       }
440     
441      protected:
442       RootPath current_;
443       Errors gc_roots_;
444     };

446     // This visitor checks that the fields of a class are "well formed".
447     // - OwnPtr, RefPtr and RawPtr must not point to a GC derived types.
448     // - An on-heap class must never contain GC roots.
449     // - Only stack-allocated types may point to stack-allocated types.
450     class CheckFieldsVisitor : public RecursiveEdgeVisitor {
451      public:
452       typedef std::vector<std::pair<FieldPoint*, Edge*> > Errors;
453     
454       CheckFieldsVisitor(const BlinkGCPluginOptions& options)
455           : options_(options), current_(0), stack_allocated_host_(false) {}
456     
457       Errors& invalid_fields() { return invalid_fields_; }
458     
459       bool ContainsInvalidFields(RecordInfo* info) {
460         stack_allocated_host_ = info->IsStackAllocated();
461         managed_host_ = stack_allocated_host_ ||
462                         info->IsGCAllocated() ||
463                         info->IsNonNewable() ||
464                         info->IsOnlyPlacementNewable();
465         for (RecordInfo::Fields::iterator it = info->GetFields().begin();
466              it != info->GetFields().end();
467              ++it) {
468           context().clear();
469           current_ = &it->second;
470           current_->edge()->Accept(this);
471         }
472         return !invalid_fields_.empty();
473       }
474     
475       void VisitMember(Member* edge) override {
476         if (managed_host_)
477           return;
478         // A member is allowed to appear in the context of a root.
479         for (Context::iterator it = context().begin();
480              it != context().end();
481              ++it) {
482           if ((*it)->Kind() == Edge::kRoot)
483             return;
484         }
485         invalid_fields_.push_back(std::make_pair(current_, edge));
486       }
487     
488       void VisitValue(Value* edge) override {
489         // TODO: what should we do to check unions?
490         if (edge->value()->record()->isUnion())
491           return;
492     
493         if (!stack_allocated_host_ && edge->value()->IsStackAllocated()) {
494           invalid_fields_.push_back(std::make_pair(current_, edge));
495           return;
496         }
497     
498         if (!Parent() || !edge->value()->IsGCAllocated())
499           return;
500     
501         // In transition mode, disallow  OwnPtr<T>, RawPtr<T> to GC allocated T's,
502         // also disallow T* in stack-allocated types.
503         if (options_.enable_oilpan) {
504           if (Parent()->IsOwnPtr() ||
505               Parent()->IsRawPtrClass() ||
506               (stack_allocated_host_ && Parent()->IsRawPtr())) {
507             invalid_fields_.push_back(std::make_pair(current_, Parent()));
508             return;
509           }
510     
511           return;
512         }
513     
514         if (Parent()->IsRawPtr() || Parent()->IsRefPtr() || Parent()->IsOwnPtr()) {
515           invalid_fields_.push_back(std::make_pair(current_, Parent()));
516           return;
517         }
518       }
519     
520      private:
521       const BlinkGCPluginOptions& options_;
522       FieldPoint* current_;
523       bool stack_allocated_host_;
524       bool managed_host_;
525       Errors invalid_fields_;
526     };

634               0,                                       // ResultPathName
635               0));                                     // TempPathName
636           if (err.empty() && json_) {
637             json_->OpenList();
638           } else {
639             json_ = 0;
640             llvm::errs()
641                 << "[blink-gc] "
642                 << "Failed to create an output file for the object graph.\n";
643           }
644         }
645     
646         for (RecordVector::iterator it = visitor.record_decls().begin();
647              it != visitor.record_decls().end();
648              ++it) {
649           CheckRecord(cache_.Lookup(*it));
650         }
651     
652         for (MethodVector::iterator it = visitor.trace_decls().begin();
653              it != visitor.trace_decls().end();
654              ++it) {
655           CheckTracingMethod(*it);
656         }
657     
658         if (json_) {
659           json_->CloseList();
660           delete json_;
661           json_ = 0;
662         }
663       }

665       // Main entry for checking a record declaration.
666       void CheckRecord(RecordInfo* info) {
667         if (IsIgnored(info))
668           return;
669     
670         CXXRecordDecl* record = info->record();
671     
672         // TODO: what should we do to check unions?
673         if (record->isUnion())
674           return;
675     
676         // If this is the primary template declaration, check its specializations.
677         if (record->isThisDeclarationADefinition() &&
678             record->getDescribedClassTemplate()) {
679           ClassTemplateDecl* tmpl = record->getDescribedClassTemplate();
680           for (ClassTemplateDecl::spec_iterator it = tmpl->spec_begin();
681                it != tmpl->spec_end();
682                ++it) {
683             CheckClass(cache_.Lookup(*it));
684           }
685           return;
686         }
687     
688         CheckClass(info);
689       }

691       // Check a class-like object (eg, class, specialization, instantiation).
692       void CheckClass(RecordInfo* info) {
693         if (!info)
694           return;
695     
696         // Check consistency of stack-allocated hierarchies.
697         if (info->IsStackAllocated()) {
698           for (RecordInfo::Bases::iterator it = info->GetBases().begin();
699                it != info->GetBases().end();
700                ++it) {
701             if (!it->second.info()->IsStackAllocated())
702               ReportDerivesNonStackAllocated(info, &it->second);
703           }
704         }
705     
706         if (CXXMethodDecl* trace = info->GetTraceMethod()) {
707           if (trace->isPure())
708             ReportClassDeclaresPureVirtualTrace(info, trace);
709         } else if (info->RequiresTraceMethod()) {
710           ReportClassRequiresTraceMethod(info);
711         }
712     
713         {
714           CheckFieldsVisitor visitor(options_);
715           if (visitor.ContainsInvalidFields(info))
716             ReportClassContainsInvalidFields(info, &visitor.invalid_fields());
717         }
718     
719         if (info->IsGCDerived()) {
720           CheckLeftMostDerived(info);
721     
722           CheckDispatch(info);
723     
724           if (CXXMethodDecl* newop = info->DeclaresNewOperator())
725             ReportClassOverridesNew(info, newop);
726     
727           {
728             CheckGCRootsVisitor visitor;
729             if (visitor.ContainsGCRoots(info))
730               ReportClassContainsGCRoots(info, &visitor.gc_roots());
731           }
732     
733           if (info->NeedsFinalization())
734             CheckFinalization(info);
735         }
736     
737         DumpClass(info);
738       }

740       void CheckLeftMostDerived(RecordInfo* info) {
741         CXXRecordDecl* left_most = info->record();
742         CXXRecordDecl::base_class_iterator it = left_most->bases_begin();
743         while (it != left_most->bases_end()) {
744           left_most = it->getType()->getAsCXXRecordDecl();
745           it = left_most->bases_begin();
746         }
747         if (!Config::IsGCBase(left_most->getName()))
748           ReportClassMustLeftMostlyDeriveGC(info);
749       }

751       void CheckDispatch(RecordInfo* info) {
752         bool finalized = info->IsGCFinalized();
753         CXXMethodDecl* trace_dispatch = info->GetTraceDispatchMethod();
754         CXXMethodDecl* finalize_dispatch = info->GetFinalizeDispatchMethod();
755         if (!trace_dispatch && !finalize_dispatch)
756           return;
757     
758         CXXRecordDecl* base = trace_dispatch ? trace_dispatch->getParent()
759                                              : finalize_dispatch->getParent();
760     
761         // Check that dispatch methods are defined at the base.
762         if (base == info->record()) {
763           if (!trace_dispatch)
764             ReportMissingTraceDispatchMethod(info);
765           if (finalized && !finalize_dispatch)
766             ReportMissingFinalizeDispatchMethod(info);
767           if (!finalized && finalize_dispatch) {
768             ReportClassRequiresFinalization(info);
769             NoteUserDeclaredFinalizer(finalize_dispatch);
770           }
771         }
772     
773         // Check that classes implementing manual dispatch do not have vtables.
774         if (info->record()->isPolymorphic())
775           ReportVirtualAndManualDispatch(
776               info, trace_dispatch ? trace_dispatch : finalize_dispatch);
777     
778         // If this is a non-abstract class check that it is dispatched to.
779         // TODO: Create a global variant of this local check. We can only check if
780         // the dispatch body is known in this compilation unit.
781         if (info->IsConsideredAbstract())
782           return;
783     
784         const FunctionDecl* defn;
785     
786         if (trace_dispatch && trace_dispatch->isDefined(defn)) {
787           CheckDispatchVisitor visitor(info);
788           visitor.TraverseStmt(defn->getBody());
789           if (!visitor.dispatched_to_receiver())
790             ReportMissingTraceDispatch(defn, info);
791         }
792     
793         if (finalized && finalize_dispatch && finalize_dispatch->isDefined(defn)) {
794           CheckDispatchVisitor visitor(info);
795           visitor.TraverseStmt(defn->getBody());
796           if (!visitor.dispatched_to_receiver())
797             ReportMissingFinalizeDispatch(defn, info);
798         }
799       }

801       // TODO: Should we collect destructors similar to trace methods?
802       void CheckFinalization(RecordInfo* info) {
803         CXXDestructorDecl* dtor = info->record()->getDestructor();
804     
805         // For finalized classes, check the finalization method if possible.
806         if (info->IsGCFinalized()) {
807           if (dtor && dtor->hasBody()) {
808             CheckFinalizerVisitor visitor(&cache_);
809             visitor.TraverseCXXMethodDecl(dtor);
810             if (!visitor.finalized_fields().empty()) {
811               ReportFinalizerAccessesFinalizedFields(
812                   dtor, &visitor.finalized_fields());
813             }
814           }
815           return;
816         }
817     
818         // Don't require finalization of a mixin that has not yet been "mixed in".
819         if (info->IsGCMixin())
820           return;
821     
822         // Report the finalization error, and proceed to print possible causes for
823         // the finalization requirement.
824         ReportClassRequiresFinalization(info);
825     
826         if (dtor && dtor->isUserProvided())
827           NoteUserDeclaredDestructor(dtor);
828     
829         for (RecordInfo::Bases::iterator it = info->GetBases().begin();
830              it != info->GetBases().end();
831              ++it) {
832           if (it->second.info()->NeedsFinalization())
833             NoteBaseRequiresFinalization(&it->second);
834         }
835     
836         for (RecordInfo::Fields::iterator it = info->GetFields().begin();
837              it != info->GetFields().end();
838              ++it) {
839           if (it->second.edge()->NeedsFinalization())
840             NoteField(&it->second, diag_field_requires_finalization_note_);
841         }
842       }

844       // This is the main entry for tracing method definitions.
845       void CheckTracingMethod(CXXMethodDecl* method) {
846         RecordInfo* parent = cache_.Lookup(method->getParent());
847         if (IsIgnored(parent))
848           return;
849     
850         // Check templated tracing methods by checking the template instantiations.
851         // Specialized templates are handled as ordinary classes.
852         if (ClassTemplateDecl* tmpl =
853                 parent->record()->getDescribedClassTemplate()) {
854           for (ClassTemplateDecl::spec_iterator it = tmpl->spec_begin();
855                it != tmpl->spec_end();
856                ++it) {
857             // Check trace using each template instantiation as the holder.
858             if (IsTemplateInstantiation(*it))
859               CheckTraceOrDispatchMethod(cache_.Lookup(*it), method);
860           }
861           return;
862         }
863     
864         CheckTraceOrDispatchMethod(parent, method);
865       }
866     
867       // Determine what type of tracing method this is (dispatch or trace).
868       void CheckTraceOrDispatchMethod(RecordInfo* parent, CXXMethodDecl* method) {
869         bool isTraceAfterDispatch;
870         if (Config::IsTraceMethod(method, &isTraceAfterDispatch)) {
871           if (isTraceAfterDispatch || !parent->GetTraceDispatchMethod()) {
872             CheckTraceMethod(parent, method, isTraceAfterDispatch);
873           }
874           // Dispatch methods are checked when we identify subclasses.
875         }
876       }
877     
878       // Check an actual trace method.
879       void CheckTraceMethod(RecordInfo* parent,
880                             CXXMethodDecl* trace,
881                             bool isTraceAfterDispatch) {
882         // A non-virtual trace method must not override another trace.
883         if (!isTraceAfterDispatch && !trace->isVirtual()) {
884           for (RecordInfo::Bases::iterator it = parent->GetBases().begin();
885                it != parent->GetBases().end();
886                ++it) {
887             RecordInfo* base = it->second.info();
888             // We allow mixin bases to contain a non-virtual trace since it will
889             // never be used for dispatching.
890             if (base->IsGCMixin())
891               continue;
892             if (CXXMethodDecl* other = base->InheritsNonVirtualTrace())
893               ReportOverriddenNonVirtualTrace(parent, trace, other);
894           }
895         }
896     
897         CheckTraceVisitor visitor(trace, parent);
898         visitor.TraverseCXXMethodDecl(trace);
899     
900         for (RecordInfo::Bases::iterator it = parent->GetBases().begin();
901              it != parent->GetBases().end();
902              ++it) {
903           if (!it->second.IsProperlyTraced())
904             ReportBaseRequiresTracing(parent, trace, it->first);
905         }
906     
907         for (RecordInfo::Fields::iterator it = parent->GetFields().begin();
908              it != parent->GetFields().end();
909              ++it) {
910           if (!it->second.IsProperlyTraced()) {
911             // Discontinue once an untraced-field error is found.
912             ReportFieldsRequireTracing(parent, trace);
913             break;
914           }
915         }
916       }

959           void AtValue(Value* e) override {
960             // The liveness kind of a path from the point to this value
961             // is given by the innermost place that is non-strong.
962             Edge::LivenessKind kind = Edge::kStrong;
963             if (Config::IsIgnoreCycleAnnotated(point_->field())) {
964               kind = Edge::kWeak;
965             } else {
966               for (Context::iterator it = context().begin();
967                    it != context().end();
968                    ++it) {
969                 Edge::LivenessKind pointer_kind = (*it)->Kind();
970                 if (pointer_kind != Edge::kStrong) {
971                   kind = pointer_kind;
972                   break;
973                 }
974               }
975             }
976             DumpEdge(
977                 src_, e->value(), point_->field()->getNameAsString(), kind, loc_);
978           }
979     
980          private:
981           JsonWriter* json_;
982           RecordInfo* src_;
983           FieldPoint* point_;
984           string loc_;
985         };

987         DumpEdgeVisitor visitor(json_);
988     
989         RecordInfo::Bases& bases = info->GetBases();

991              it != bases.end();

994                            it->second.info(),
995                            "<super>",

998         }

1001         for (RecordInfo::Fields::iterator it = fields.begin();
1002              it != fields.end();
1003              ++it) {
1004           visitor.DumpField(info,
1005                             &it->second,
1006                             GetLocString(it->second.field()->getLocStart()));
1007         }

1016       const string GetLocString(SourceLocation loc) {
1017         const SourceManager& source_manager = instance_.getSourceManager();
1018         PresumedLoc ploc = source_manager.getPresumedLoc(loc);
1019         if (ploc.isInvalid())
1020           return "";
1021         string loc_str;
1022         llvm::raw_string_ostream OS(loc_str);
1023         OS << ploc.getFilename()
1024            << ":" << ploc.getLine()
1025            << ":" << ploc.getColumn();
1026         return OS.str();
1027       }

1029       bool IsIgnored(RecordInfo* record) {
1030         return !record ||
1031                !InCheckedNamespace(record) ||
1032                IsIgnoredClass(record) ||
1033                InIgnoredDirectory(record);

1036       bool IsIgnoredClass(RecordInfo* info) {
1037         // Ignore any class prefixed by SameSizeAs. These are used in
1038         // Blink to verify class sizes and don't need checking.
1039         const string SameSizeAs = "SameSizeAs";
1040         if (info->name().compare(0, SameSizeAs.size(), SameSizeAs) == 0)
1041           return true;
1042         return options_.ignored_classes.find(info->name()) !=
1043                options_.ignored_classes.end();
1044       }

1046       bool InIgnoredDirectory(RecordInfo* info) {
1047         string filename;
1048         if (!GetFilename(info->record()->getLocStart(), &filename))
1049           return false;  // TODO: should we ignore non-existing file locations?
1050         std::vector<string>::iterator it = options_.ignored_directories.begin();
1051         for (; it != options_.ignored_directories.end(); ++it)
1052           if (filename.find(*it) != string::npos)
1053             return true;
1054         return false;
1055       }

1057       bool InCheckedNamespace(RecordInfo* info) {
1058         if (!info)
1059           return false;
1060         DeclContext* context = info->record()->getDeclContext();
1061         if (context->isRecord())
1062           return InCheckedNamespace(cache_.Lookup(context));
1063         if (context->isNamespace()) {
1064           const NamespaceDecl* decl = dyn_cast<NamespaceDecl>(context);
1065           if (decl->isAnonymousNamespace())
1066             return false;
1067           return options_.checked_namespaces.find(decl->getNameAsString()) !=
1068               options_.checked_namespaces.end();
1069         }
1070         return false;
1071       }

1074         const SourceManager& source_manager = instance_.getSourceManager();
1075         SourceLocation spelling_location = source_manager.getSpellingLoc(loc);
1076         PresumedLoc ploc = source_manager.getPresumedLoc(spelling_location);
1077         if (ploc.isInvalid()) {

1083         return true;
