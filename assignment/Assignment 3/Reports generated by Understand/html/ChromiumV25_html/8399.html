72     void
73     _mesa_glsl_read_ir(_mesa_glsl_parse_state *state, exec_list *instructions,
74     		   const char *src, bool scan_for_protos)
75     {
76        s_expression *expr = s_expression::read_expression(state, src);
77        if (expr == NULL) {
78           ir_read_error(state, NULL, "couldn't parse S-Expression.");
79           return;
80        }
81        
82        if (scan_for_protos) {
83           scan_for_prototypes(state, instructions, expr);
84           if (state->error)
85     	 return;
86        }
87     
88        read_instructions(state, instructions, expr, NULL);
89        talloc_free(expr);
90     
91        if (debug)
92           validate_ir_tree(instructions);
93     }

95     static void
96     ir_read_error(_mesa_glsl_parse_state *state, s_expression *expr,
97     	      const char *fmt, ...)
98     {
99        va_list ap;
100     
101        state->error = true;
102     
103        if (state->current_function != NULL)
104           state->info_log = talloc_asprintf_append(state->info_log,
105     			   "In function %s:\n",
106     			   state->current_function->function_name());
107        state->info_log = talloc_strdup_append(state->info_log, "error: ");
108     
109        va_start(ap, fmt);
110        state->info_log = talloc_vasprintf_append(state->info_log, fmt, ap);
111        va_end(ap);
112        state->info_log = talloc_strdup_append(state->info_log, "\n");
113     
114        if (expr != NULL) {
115           state->info_log = talloc_strdup_append(state->info_log,
116     					     "...in this context:\n   ");
117           expr->print();
118           state->info_log = talloc_strdup_append(state->info_log, "\n\n");
119        }
120     }

122     static const glsl_type *
123     read_type(_mesa_glsl_parse_state *st, s_expression *expr)
124     {
125        s_list *list = SX_AS_LIST(expr);
126        if (list != NULL) {
127           s_symbol *type_sym = SX_AS_SYMBOL(list->subexpressions.get_head());
128           if (type_sym == NULL) {
129     	 ir_read_error(st, expr, "expected type (array ...) or (struct ...)");
130     	 return NULL;
131           }
132           if (safe_strcmp(type_sym->value(), "array") == 0) {
133     	 if (list->length() != 3) {
134     	    ir_read_error(st, expr, "expected type (array <type> <int>)");
135     	    return NULL;
136     	 }
137     
138     	 // Read base type
139     	 s_expression *base_expr = (s_expression*) type_sym->next;
140     	 const glsl_type *base_type = read_type(st, base_expr);
141     	 if (base_type == NULL) {
142     	    ir_read_error(st, NULL, "when reading base type of array");
143     	    return NULL;
144     	 }
145     
146     	 // Read array size
147     	 s_int *size = SX_AS_INT(base_expr->next);
148     	 if (size == NULL) {
149     	    ir_read_error(st, expr, "found non-integer array size");
150     	    return NULL;
151     	 }
152     
153     	 return glsl_type::get_array_instance(base_type, size->value());
154           } else if (safe_strcmp(type_sym->value(), "struct") == 0) {
155     	 assert(false); // FINISHME
156           } else {
157     	 ir_read_error(st, expr, "expected (array ...) or (struct ...); "
158     				 "found (%s ...)", type_sym->value());
159     	 return NULL;
160           }
161        }
162        
163        s_symbol *type_sym = SX_AS_SYMBOL(expr);
164        if (type_sym == NULL) {
165           ir_read_error(st, expr, "expected <type> (symbol or list)");
166           return NULL;
167        }
168     
169        const glsl_type *type = st->symbols->get_type(type_sym->value());
170        if (type == NULL)
171           ir_read_error(st, expr, "invalid type: %s", type_sym->value());
172     
173        return type;
174     }

177     static void
178     scan_for_prototypes(_mesa_glsl_parse_state *st, exec_list *instructions,
179     		    s_expression *expr)
180     {
181        s_list *list = SX_AS_LIST(expr);
182        if (list == NULL) {
183           ir_read_error(st, expr, "Expected (<instruction> ...); found an atom.");
184           return;
185        }
186     
187        foreach_iter(exec_list_iterator, it, list->subexpressions) {
188           s_list *sub = SX_AS_LIST(it.get());
189           if (sub == NULL)
190     	 continue; // not a (function ...); ignore it.
191     
192           s_symbol *tag = SX_AS_SYMBOL(sub->subexpressions.get_head());
193           if (tag == NULL || safe_strcmp(tag->value(), "function") != 0)
194     	 continue; // not a (function ...); ignore it.
195     
196           ir_function *f = read_function(st, sub, true);
197           if (f == NULL)
198     	 return;
199           instructions->push_tail(f);
200        }
201     }

203     static ir_function *
204     read_function(_mesa_glsl_parse_state *st, s_list *list, bool skip_body)
205     {
206        void *ctx = st;
207        bool added = false;
208        if (list->length() < 3) {
209           ir_read_error(st, list, "Expected (function <name> (signature ...) ...)");
210           return NULL;
211        }
212     
213        s_symbol *name = SX_AS_SYMBOL(list->subexpressions.head->next);
214        if (name == NULL) {
215           ir_read_error(st, list, "Expected (function <name> ...)");
216           return NULL;
217        }
218     
219        ir_function *f = st->symbols->get_function(name->value());
220        if (f == NULL) {
221           f = new(ctx) ir_function(name->value());
222           added = st->symbols->add_function(f->name, f);
223           assert(added);
224        }
225     
226        exec_list_iterator it = list->subexpressions.iterator();
227        it.next(); // skip "function" tag
228        it.next(); // skip function name
229        for (/* nothing */; it.has_next(); it.next()) {
230           s_list *siglist = SX_AS_LIST(it.get());
231           if (siglist == NULL) {
232     	 ir_read_error(st, list, "Expected (function (signature ...) ...)");
233     	 return NULL;
234           }
235     
236           s_symbol *tag = SX_AS_SYMBOL(siglist->subexpressions.get_head());
237           if (tag == NULL || safe_strcmp(tag->value(), "signature") != 0) {
238     	 ir_read_error(st, siglist, "Expected (signature ...)");
239     	 return NULL;
240           }
241     
242           read_function_sig(st, f, siglist, skip_body);
243        }
244        return added ? f : NULL;
245     }

247     static void
248     read_function_sig(_mesa_glsl_parse_state *st, ir_function *f, s_list *list,
249     		  bool skip_body)
250     {
251        void *ctx = st;
252        if (list->length() != 4) {
253           ir_read_error(st, list, "Expected (signature <type> (parameters ...) "
254     			      "(<instruction> ...))");
255           return;
256        }
257     
258        s_expression *type_expr = (s_expression*) list->subexpressions.head->next;
259        const glsl_type *return_type = read_type(st, type_expr);
260        if (return_type == NULL)
261           return;
262     
263        s_list *paramlist = SX_AS_LIST(type_expr->next);
264        s_list *body_list = SX_AS_LIST(paramlist->next);
265        if (paramlist == NULL || body_list == NULL) {
266           ir_read_error(st, list, "Expected (signature <type> (parameters ...) "
267     			      "(<instruction> ...))");
268           return;
269        }
270        s_symbol *paramtag = SX_AS_SYMBOL(paramlist->subexpressions.get_head());
271        if (paramtag == NULL || safe_strcmp(paramtag->value(), "parameters") != 0) {
272           ir_read_error(st, paramlist, "Expected (parameters ...)");
273           return;
274        }
275     
276        // Read the parameters list into a temporary place.
277        exec_list hir_parameters;
278        st->symbols->push_scope();
279     
280        exec_list_iterator it = paramlist->subexpressions.iterator();
281        for (it.next() /* skip "parameters" */; it.has_next(); it.next()) {
282           s_list *decl = SX_AS_LIST(it.get());
283           ir_variable *var = read_declaration(st, decl);
284           if (var == NULL)
285     	 return;
286     
287           hir_parameters.push_tail(var);
288        }
289     
290        ir_function_signature *sig = f->exact_matching_signature(&hir_parameters);
291        if (sig == NULL && skip_body) {
292           /* If scanning for prototypes, generate a new signature. */
293           sig = new(ctx) ir_function_signature(return_type);
294           sig->is_builtin = true;
295           f->add_signature(sig);
296        } else if (sig != NULL) {
297           const char *badvar = sig->qualifiers_match(&hir_parameters);
298           if (badvar != NULL) {
299     	 ir_read_error(st, list, "function `%s' parameter `%s' qualifiers "
300     		       "don't match prototype", f->name, badvar);
301     	 return;
302           }
303     
304           if (sig->return_type != return_type) {
305     	 ir_read_error(st, list, "function `%s' return type doesn't "
306     		       "match prototype", f->name);
307     	 return;
308           }
309        } else {
310           /* No prototype for this body exists - skip it. */
311           st->symbols->pop_scope();
312           return;
313        }
314        assert(sig != NULL);
315     
316        sig->replace_parameters(&hir_parameters);
317     
318        if (!skip_body && !body_list->subexpressions.is_empty()) {
319           if (sig->is_defined) {
320     	 ir_read_error(st, list, "function %s redefined", f->name);
321     	 return;
322           }
323           st->current_function = sig;
324           read_instructions(st, &sig->body, body_list, NULL);
325           st->current_function = NULL;
326           sig->is_defined = true;
327        }
328     
329        st->symbols->pop_scope();
330     }

332     static void
333     read_instructions(_mesa_glsl_parse_state *st, exec_list *instructions,
334     		  s_expression *expr, ir_loop *loop_ctx)
335     {
336        // Read in a list of instructions
337        s_list *list = SX_AS_LIST(expr);
338        if (list == NULL) {
339           ir_read_error(st, expr, "Expected (<instruction> ...); found an atom.");
340           return;
341        }
342     
343        foreach_iter(exec_list_iterator, it, list->subexpressions) {
344           s_expression *sub = (s_expression*) it.get();
345           ir_instruction *ir = read_instruction(st, sub, loop_ctx);
346           if (ir != NULL) {
347     	 /* Global variable declarations should be moved to the top, before
348     	  * any functions that might use them.  Functions are added to the
349     	  * instruction stream when scanning for prototypes, so without this
350     	  * hack, they always appear before variable declarations.
351     	  */
352     	 if (st->current_function == NULL && ir->as_variable() != NULL)
353     	    instructions->push_head(ir);
354     	 else
355     	    instructions->push_tail(ir);
356           }
357        }
358     }

361     static ir_instruction *
362     read_instruction(_mesa_glsl_parse_state *st, s_expression *expr,
363     	         ir_loop *loop_ctx)
364     {
365        void *ctx = st;
366        s_symbol *symbol = SX_AS_SYMBOL(expr);
367        if (symbol != NULL) {
368           if (safe_strcmp(symbol->value(), "break") == 0 && loop_ctx != NULL)
369     	 return new(ctx) ir_loop_jump(ir_loop_jump::jump_break);
370           if (safe_strcmp(symbol->value(), "continue") == 0 && loop_ctx != NULL)
371     	 return new(ctx) ir_loop_jump(ir_loop_jump::jump_continue);
372        }
373     
374        s_list *list = SX_AS_LIST(expr);
375        if (list == NULL || list->subexpressions.is_empty()) {
376           ir_read_error(st, expr, "Invalid instruction.\n");
377           return NULL;
378        }
379     
380        s_symbol *tag = SX_AS_SYMBOL(list->subexpressions.get_head());
381        if (tag == NULL) {
382           ir_read_error(st, expr, "expected instruction tag");
383           return NULL;
384        }
385     
386        ir_instruction *inst = NULL;
387        if (safe_strcmp(tag->value(), "declare") == 0) {
388           inst = read_declaration(st, list);
389        } else if (safe_strcmp(tag->value(), "assign") == 0) {
390           inst = read_assignment(st, list);
391        } else if (safe_strcmp(tag->value(), "if") == 0) {
392           inst = read_if(st, list, loop_ctx);
393        } else if (safe_strcmp(tag->value(), "loop") == 0) {
394           inst = read_loop(st, list);
395        } else if (safe_strcmp(tag->value(), "return") == 0) {
396           inst = read_return(st, list);
397        } else if (safe_strcmp(tag->value(), "function") == 0) {
398           inst = read_function(st, list, false);
399        } else {
400           inst = read_rvalue(st, list);
401           if (inst == NULL)
402     	 ir_read_error(st, NULL, "when reading instruction");
403        }
404        return inst;
405     }

408     static ir_variable *
409     read_declaration(_mesa_glsl_parse_state *st, s_list *list)
410     {
411        void *ctx = st;
412        if (list->length() != 4) {
413           ir_read_error(st, list, "expected (declare (<qualifiers>) <type> "
414     			      "<name>)");
415           return NULL;
416        }
417     
418        s_list *quals = SX_AS_LIST(list->subexpressions.head->next);
419        if (quals == NULL) {
420           ir_read_error(st, list, "expected a list of variable qualifiers");
421           return NULL;
422        }
423     
424        s_expression *type_expr = (s_expression*) quals->next;
425        const glsl_type *type = read_type(st, type_expr);
426        if (type == NULL)
427           return NULL;
428     
429        s_symbol *var_name = SX_AS_SYMBOL(type_expr->next);
430        if (var_name == NULL) {
431           ir_read_error(st, list, "expected variable name, found non-symbol");
432           return NULL;
433        }
434     
435        ir_variable *var = new(ctx) ir_variable(type, var_name->value(),
436     					   ir_var_auto);
437     
438        foreach_iter(exec_list_iterator, it, quals->subexpressions) {
439           s_symbol *qualifier = SX_AS_SYMBOL(it.get());
440           if (qualifier == NULL) {
441     	 ir_read_error(st, list, "qualifier list must contain only symbols");
442     	 delete var;
443     	 return NULL;
444           }
445     
446           // FINISHME: Check for duplicate/conflicting qualifiers.
447           if (safe_strcmp(qualifier->value(), "centroid") == 0) {
448     	 var->centroid = 1;
449           } else if (safe_strcmp(qualifier->value(), "invariant") == 0) {
450     	 var->invariant = 1;
451           } else if (safe_strcmp(qualifier->value(), "uniform") == 0) {
452     	 var->mode = ir_var_uniform;
453           } else if (safe_strcmp(qualifier->value(), "auto") == 0) {
454     	 var->mode = ir_var_auto;
455           } else if (safe_strcmp(qualifier->value(), "in") == 0) {
456     	 var->mode = ir_var_in;
457           } else if (safe_strcmp(qualifier->value(), "out") == 0) {
458     	 var->mode = ir_var_out;
459           } else if (safe_strcmp(qualifier->value(), "inout") == 0) {
460     	 var->mode = ir_var_inout;
461           } else if (safe_strcmp(qualifier->value(), "smooth") == 0) {
462     	 var->interpolation = ir_var_smooth;
463           } else if (safe_strcmp(qualifier->value(), "flat") == 0) {
464     	 var->interpolation = ir_var_flat;
465           } else if (safe_strcmp(qualifier->value(), "noperspective") == 0) {
466     	 var->interpolation = ir_var_noperspective;
467           } else {
468     	 ir_read_error(st, list, "unknown qualifier: %s", qualifier->value());
469     	 delete var;
470     	 return NULL;
471           }
472        }
473     
474        // Add the variable to the symbol table
475        st->symbols->add_variable(var->name, var);
476     
477        return var;
478     }

481     static ir_if *
482     read_if(_mesa_glsl_parse_state *st, s_list *list, ir_loop *loop_ctx)
483     {
484        void *ctx = st;
485        if (list->length() != 4) {
486           ir_read_error(st, list, "expected (if <condition> (<then> ...) "
487                               "(<else> ...))");
488           return NULL;
489        }
490     
491        s_expression *cond_expr = (s_expression*) list->subexpressions.head->next;
492        ir_rvalue *condition = read_rvalue(st, cond_expr);
493        if (condition == NULL) {
494           ir_read_error(st, NULL, "when reading condition of (if ...)");
495           return NULL;
496        }
497     
498        s_expression *then_expr = (s_expression*) cond_expr->next;
499        s_expression *else_expr = (s_expression*) then_expr->next;
500     
501        ir_if *iff = new(ctx) ir_if(condition);
502     
503        read_instructions(st, &iff->then_instructions, then_expr, loop_ctx);
504        read_instructions(st, &iff->else_instructions, else_expr, loop_ctx);
505        if (st->error) {
506           delete iff;
507           iff = NULL;
508        }
509        return iff;
510     }

513     static ir_loop *
514     read_loop(_mesa_glsl_parse_state *st, s_list *list)
515     {
516        void *ctx = st;
517        if (list->length() != 6) {
518           ir_read_error(st, list, "expected (loop <counter> <from> <to> "
519     			      "<increment> <body>)");
520           return NULL;
521        }
522     
523        s_expression *count_expr = (s_expression*) list->subexpressions.head->next;
524        s_expression *from_expr  = (s_expression*) count_expr->next;
525        s_expression *to_expr    = (s_expression*) from_expr->next;
526        s_expression *inc_expr   = (s_expression*) to_expr->next;
527        s_expression *body_expr  = (s_expression*) inc_expr->next;
528     
529        // FINISHME: actually read the count/from/to fields.
530     
531        ir_loop *loop = new(ctx) ir_loop;
532        read_instructions(st, &loop->body_instructions, body_expr, loop);
533        if (st->error) {
534           delete loop;
535           loop = NULL;
536        }
537        return loop;
538     }

541     static ir_return *
542     read_return(_mesa_glsl_parse_state *st, s_list *list)
543     {
544        void *ctx = st;
545        if (list->length() != 2) {
546           ir_read_error(st, list, "expected (return <rvalue>)");
547           return NULL;
548        }
549     
550        s_expression *expr = (s_expression*) list->subexpressions.head->next;
551     
552        ir_rvalue *retval = read_rvalue(st, expr);
553        if (retval == NULL) {
554           ir_read_error(st, NULL, "when reading return value");
555           return NULL;
556        }
557     
558        return new(ctx) ir_return(retval);
559     }

562     static ir_rvalue *
563     read_rvalue(_mesa_glsl_parse_state *st, s_expression *expr)
564     {
565        s_list *list = SX_AS_LIST(expr);
566        if (list == NULL || list->subexpressions.is_empty())
567           return NULL;
568     
569        s_symbol *tag = SX_AS_SYMBOL(list->subexpressions.get_head());
570        if (tag == NULL) {
571           ir_read_error(st, expr, "expected rvalue tag");
572           return NULL;
573        }
574     
575        ir_rvalue *rvalue = read_dereference(st, list);
576        if (rvalue != NULL || st->error)
577           return rvalue;
578        else if (safe_strcmp(tag->value(), "swiz") == 0) {
579           rvalue = read_swizzle(st, list);
580        } else if (safe_strcmp(tag->value(), "expression") == 0) {
581           rvalue = read_expression(st, list);
582        } else if (safe_strcmp(tag->value(), "call") == 0) {
583           rvalue = read_call(st, list);
584        } else if (safe_strcmp(tag->value(), "constant") == 0) {
585           rvalue = read_constant(st, list);
586        } else {
587           rvalue = read_texture(st, list);
588           if (rvalue == NULL && !st->error)
589     	 ir_read_error(st, expr, "unrecognized rvalue tag: %s", tag->value());
590        }
591     
592        return rvalue;
593     }

595     static ir_assignment *
596     read_assignment(_mesa_glsl_parse_state *st, s_list *list)
597     {
598        void *ctx = st;
599        if (list->length() != 5) {
600           ir_read_error(st, list, "expected (assign <condition> (<write mask>) "
601     			      "<lhs> <rhs>)");
602           return NULL;
603        }
604     
605        s_expression *cond_expr = (s_expression*) list->subexpressions.head->next;
606        s_list       *mask_list = SX_AS_LIST(cond_expr->next);
607        s_expression *lhs_expr  = (s_expression*) cond_expr->next->next;
608        s_expression *rhs_expr  = (s_expression*) lhs_expr->next;
609     
610        ir_rvalue *condition = read_rvalue(st, cond_expr);
611        if (condition == NULL) {
612           ir_read_error(st, NULL, "when reading condition of assignment");
613           return NULL;
614        }
615     
616        if (mask_list == NULL || mask_list->length() > 1) {
617           ir_read_error(st, mask_list, "expected () or (<write mask>)");
618           return NULL;
619        }
620     
621        unsigned mask = 0;
622        if (mask_list->length() == 1) {
623           s_symbol *mask_symbol = SX_AS_SYMBOL(mask_list->subexpressions.head);
624           if (mask_symbol == NULL) {
625     	 ir_read_error(st, list, "expected a write mask; found non-symbol");
626     	 return NULL;
627           }
628     
629           const char *mask_str = mask_symbol->value();
630           unsigned mask_length = strlen(mask_str);
631           if (mask_length > 4) {
632     	 ir_read_error(st, list, "invalid write mask: %s", mask_str);
633     	 return NULL;
634           }
635     
636           const unsigned idx_map[] = { 3, 0, 1, 2 }; /* w=bit 3, x=0, y=1, z=2 */
637     
638           for (unsigned i = 0; i < mask_length; i++) {
639     	 if (mask_str[i] < 'w' || mask_str[i] > 'z') {
640     	    ir_read_error(st, list, "write mask contains invalid character: %c",
641     			  mask_str[i]);
642     	    return NULL;
643     	 }
644     	 mask |= 1 << idx_map[mask_str[i] - 'w'];
645           }
646        }
647     
648        ir_dereference *lhs = read_dereference(st, lhs_expr);
649        if (lhs == NULL) {
650           ir_read_error(st, NULL, "when reading left-hand side of assignment");
651           return NULL;
652        }
653     
654        ir_rvalue *rhs = read_rvalue(st, rhs_expr);
655        if (rhs == NULL) {
656           ir_read_error(st, NULL, "when reading right-hand side of assignment");
657           return NULL;
658        }
659     
660        if (mask == 0 && (lhs->type->is_vector() || lhs->type->is_scalar())) {
661           ir_read_error(st, list, "non-zero write mask required.");
662           return NULL;
663        }
664     
665        return new(ctx) ir_assignment(lhs, rhs, condition, mask);
666     }

668     static ir_call *
669     read_call(_mesa_glsl_parse_state *st, s_list *list)
670     {
671        void *ctx = st;
672        if (list->length() != 3) {
673           ir_read_error(st, list, "expected (call <name> (<param> ...))");
674           return NULL;
675        }
676     
677        s_symbol *name = SX_AS_SYMBOL(list->subexpressions.head->next);
678        s_list *params = SX_AS_LIST(name->next);
679        if (name == NULL || params == NULL) {
680           ir_read_error(st, list, "expected (call <name> (<param> ...))");
681           return NULL;
682        }
683     
684        exec_list parameters;
685     
686        foreach_iter(exec_list_iterator, it, params->subexpressions) {
687           s_expression *expr = (s_expression*) it.get();
688           ir_rvalue *param = read_rvalue(st, expr);
689           if (param == NULL) {
690     	 ir_read_error(st, list, "when reading parameter to function call");
691     	 return NULL;
692           }
693           parameters.push_tail(param);
694        }
695     
696        ir_function *f = st->symbols->get_function(name->value());
697        if (f == NULL) {
698           ir_read_error(st, list, "found call to undefined function %s",
699     		    name->value());
700           return NULL;
701        }
702     
703        ir_function_signature *callee = f->matching_signature(&parameters);
704        if (callee == NULL) {
705           ir_read_error(st, list, "couldn't find matching signature for function "
706                         "%s", name->value());
707           return NULL;
708        }
709     
710        return new(ctx) ir_call(callee, &parameters);
711     }

713     static ir_expression *
714     read_expression(_mesa_glsl_parse_state *st, s_list *list)
715     {
716        void *ctx = st;
717        const unsigned list_length = list->length();
718        if (list_length < 4) {
719           ir_read_error(st, list, "expected (expression <type> <operator> "
720     			      "<operand> [<operand>])");
721           return NULL;
722        }
723     
724        s_expression *type_expr = (s_expression*) list->subexpressions.head->next;
725        const glsl_type *type = read_type(st, type_expr);
726        if (type == NULL)
727           return NULL;
728     
729        /* Read the operator */
730        s_symbol *op_sym = SX_AS_SYMBOL(type_expr->next);
731        if (op_sym == NULL) {
732           ir_read_error(st, list, "expected operator, found non-symbol");
733           return NULL;
734        }
735     
736        ir_expression_operation op = ir_expression::get_operator(op_sym->value());
737        if (op == (ir_expression_operation) -1) {
738           ir_read_error(st, list, "invalid operator: %s", op_sym->value());
739           return NULL;
740        }
741         
742        /* Now that we know the operator, check for the right number of operands */ 
743        if (ir_expression::get_num_operands(op) == 2) {
744           if (list_length != 5) {
745     	 ir_read_error(st, list, "expected (expression <type> %s <operand> "
746     				 " <operand>)", op_sym->value());
747     	 return NULL;
748           }
749        } else {
750           if (list_length != 4) {
751     	 ir_read_error(st, list, "expected (expression <type> %s <operand>)",
752     		       op_sym->value());
753     	 return NULL;
754           }
755        }
756     
757        s_expression *exp1 = (s_expression*) (op_sym->next);
758        ir_rvalue *arg1 = read_rvalue(st, exp1);
759        if (arg1 == NULL) {
760           ir_read_error(st, NULL, "when reading first operand of %s",
761     		    op_sym->value());
762           return NULL;
763        }
764     
765        ir_rvalue *arg2 = NULL;
766        if (ir_expression::get_num_operands(op) == 2) {
767           s_expression *exp2 = (s_expression*) (exp1->next);
768           arg2 = read_rvalue(st, exp2);
769           if (arg2 == NULL) {
770     	 ir_read_error(st, NULL, "when reading second operand of %s",
771     		       op_sym->value());
772     	 return NULL;
773           }
774        }
775     
776        return new(ctx) ir_expression(op, type, arg1, arg2);
777     }

779     static ir_swizzle *
780     read_swizzle(_mesa_glsl_parse_state *st, s_list *list)
781     {
782        if (list->length() != 3) {
783           ir_read_error(st, list, "expected (swiz <swizzle> <rvalue>)");
784           return NULL;
785        }
786     
787        s_symbol *swiz = SX_AS_SYMBOL(list->subexpressions.head->next);
788        if (swiz == NULL) {
789           ir_read_error(st, list, "expected a valid swizzle; found non-symbol");
790           return NULL;
791        }
792     
793        if (strlen(swiz->value()) > 4) {
794           ir_read_error(st, list, "expected a valid swizzle; found %s",
795     		    swiz->value());
796           return NULL;
797        }
798     
799        s_expression *sub = (s_expression*) swiz->next;
800        if (sub == NULL) {
801           ir_read_error(st, list, "expected rvalue: (swizzle %s <rvalue>)",
802     		    swiz->value());
803           return NULL;
804        }
805     
806        ir_rvalue *rvalue = read_rvalue(st, sub);
807        if (rvalue == NULL)
808           return NULL;
809     
810        ir_swizzle *ir = ir_swizzle::create(rvalue, swiz->value(),
811     				       rvalue->type->vector_elements);
812        if (ir == NULL)
813           ir_read_error(st, list, "invalid swizzle");
814     
815        return ir;
816     }

818     static ir_constant *
819     read_constant(_mesa_glsl_parse_state *st, s_list *list)
820     {
821        void *ctx = st;
822        if (list->length() != 3) {
823           ir_read_error(st, list, "expected (constant <type> (...))");
824           return NULL;
825        }
826     
827        s_expression *type_expr = (s_expression*) list->subexpressions.head->next;
828        const glsl_type *type = read_type(st, type_expr);
829        if (type == NULL)
830           return NULL;
831     
832        s_list *values = SX_AS_LIST(type_expr->next);
833        if (values == NULL) {
834           ir_read_error(st, list, "expected (constant <type> (...))");
835           return NULL;
836        }
837     
838        if (type->is_array()) {
839           const unsigned elements_supplied = values->length();
840           if (elements_supplied != type->length) {
841     	 ir_read_error(st, values, "expected exactly %u array elements, "
842     		       "given %u", type->length, elements_supplied);
843     	 return NULL;
844           }
845     
846           exec_list elements;
847           foreach_iter(exec_list_iterator, it, values->subexpressions) {
848     	 s_expression *expr = (s_expression *) it.get();
849     	 s_list *elt = SX_AS_LIST(expr);
850     	 if (elt == NULL) {
851     	    ir_read_error(st, expr, "expected (constant ...) array element");
852     	    return NULL;
853     	 }
854     
855     	 ir_constant *ir_elt = read_constant(st, elt);
856     	 if (ir_elt == NULL)
857     	    return NULL;
858     	 elements.push_tail(ir_elt);
859           }
860           return new(ctx) ir_constant(type, &elements);
861        }
862     
863        const glsl_type *const base_type = type->get_base_type();
864     
865        ir_constant_data data = { { 0 } };
866     
867        // Read in list of values (at most 16).
868        int k = 0;
869        foreach_iter(exec_list_iterator, it, values->subexpressions) {
870           if (k >= 16) {
871     	 ir_read_error(st, values, "expected at most 16 numbers");
872     	 return NULL;
873           }
874     
875           s_expression *expr = (s_expression*) it.get();
876     
877           if (base_type->base_type == GLSL_TYPE_FLOAT) {
878     	 s_number *value = SX_AS_NUMBER(expr);
879     	 if (value == NULL) {
880     	    ir_read_error(st, values, "expected numbers");
881     	    return NULL;
882     	 }
883     	 data.f[k] = value->fvalue();
884           } else {
885     	 s_int *value = SX_AS_INT(expr);
886     	 if (value == NULL) {
887     	    ir_read_error(st, values, "expected integers");
888     	    return NULL;
889     	 }
890     
891     	 switch (base_type->base_type) {
892     	 case GLSL_TYPE_UINT: {
893     	    data.u[k] = value->value();
894     	    break;
895     	 }
896     	 case GLSL_TYPE_INT: {
897     	    data.i[k] = value->value();
898     	    break;
899     	 }
900     	 case GLSL_TYPE_BOOL: {
901     	    data.b[k] = value->value();
902     	    break;
903     	 }
904     	 default:
905     	    ir_read_error(st, values, "unsupported constant type");
906     	    return NULL;
907     	 }
908           }
909           ++k;
910        }
911     
912        return new(ctx) ir_constant(type, &data);
913     }

915     static ir_dereference *
916     read_dereference(_mesa_glsl_parse_state *st, s_expression *expr)
917     {
918        s_list *list = SX_AS_LIST(expr);
919        if (list == NULL || list->subexpressions.is_empty())
920           return NULL;
921     
922        s_symbol *tag = SX_AS_SYMBOL(list->subexpressions.head);
923        assert(tag != NULL);
924     
925        if (safe_strcmp(tag->value(), "var_ref") == 0)
926           return read_var_ref(st, list);
927        if (safe_strcmp(tag->value(), "array_ref") == 0)
928           return read_array_ref(st, list);
929        if (safe_strcmp(tag->value(), "record_ref") == 0)
930           return read_record_ref(st, list);
931        return NULL;
932     }
933     
934     static ir_dereference *
935     read_var_ref(_mesa_glsl_parse_state *st, s_list *list)
936     {
937        void *ctx = st;
938        if (list->length() != 2) {
939           ir_read_error(st, list, "expected (var_ref <variable name>)");
940           return NULL;
941        }
942        s_symbol *var_name = SX_AS_SYMBOL(list->subexpressions.head->next);
943        if (var_name == NULL) {
944           ir_read_error(st, list, "expected (var_ref <variable name>)");
945           return NULL;
946        }
947     
948        ir_variable *var = st->symbols->get_variable(var_name->value());
949        if (var == NULL) {
950           ir_read_error(st, list, "undeclared variable: %s", var_name->value());
951           return NULL;
952        }
953     
954        return new(ctx) ir_dereference_variable(var);
955     }

957     static ir_dereference *
958     read_array_ref(_mesa_glsl_parse_state *st, s_list *list)
959     {
960        void *ctx = st;
961        if (list->length() != 3) {
962           ir_read_error(st, list, "expected (array_ref <rvalue> <index>)");
963           return NULL;
964        }
965     
966        s_expression *subj_expr = (s_expression*) list->subexpressions.head->next;
967        ir_rvalue *subject = read_rvalue(st, subj_expr);
968        if (subject == NULL) {
969           ir_read_error(st, NULL, "when reading the subject of an array_ref");
970           return NULL;
971        }
972     
973        s_expression *idx_expr = (s_expression*) subj_expr->next;
974        ir_rvalue *idx = read_rvalue(st, idx_expr);
975        return new(ctx) ir_dereference_array(subject, idx);
976     }

978     static ir_dereference *
979     read_record_ref(_mesa_glsl_parse_state *st, s_list *list)
980     {
981        void *ctx = st;
982        if (list->length() != 3) {
983           ir_read_error(st, list, "expected (record_ref <rvalue> <field>)");
984           return NULL;
985        }
986     
987        s_expression *subj_expr = (s_expression*) list->subexpressions.head->next;
988        ir_rvalue *subject = read_rvalue(st, subj_expr);
989        if (subject == NULL) {
990           ir_read_error(st, NULL, "when reading the subject of a record_ref");
991           return NULL;
992        }
993     
994        s_symbol *field = SX_AS_SYMBOL(subj_expr->next);
995        if (field == NULL) {
996           ir_read_error(st, list, "expected (record_ref ... <field name>)");
997           return NULL;
998        }
999        return new(ctx) ir_dereference_record(subject, field->value());
1000     }

1002     static bool
1003     valid_texture_list_length(ir_texture_opcode op, s_list *list)
1004     {
1005        unsigned required_length = 7;
1006        if (op == ir_txf)
1007           required_length = 5;
1008        else if (op == ir_tex)
1009           required_length = 6;
1010     
1011        return list->length() == required_length;
1012     }

1014     static ir_texture *
1015     read_texture(_mesa_glsl_parse_state *st, s_list *list)
1016     {
1017        void *ctx = st;
1018        s_symbol *tag = SX_AS_SYMBOL(list->subexpressions.head);
1019        assert(tag != NULL);
1020     
1021        ir_texture_opcode op = ir_texture::get_opcode(tag->value());
1022        if (op == (ir_texture_opcode) -1)
1023           return NULL;
1024     
1025        if (!valid_texture_list_length(op, list)) {
1026           ir_read_error(st, NULL, "invalid list size in (%s ...)", tag->value());
1027           return NULL;
1028        }
1029     
1030        ir_texture *tex = new(ctx) ir_texture(op);
1031     
1032        // Read sampler (must be a deref)
1033        s_expression *sampler_expr = (s_expression *) tag->next;
1034        ir_dereference *sampler = read_dereference(st, sampler_expr);
1035        if (sampler == NULL) {
1036           ir_read_error(st, NULL, "when reading sampler in (%s ...)", tag->value());
1037           return NULL;
1038        }
1039        tex->set_sampler(sampler);
1040     
1041        // Read coordinate (any rvalue)
1042        s_expression *coordinate_expr = (s_expression *) sampler_expr->next;
1043        tex->coordinate = read_rvalue(st, coordinate_expr);
1044        if (tex->coordinate == NULL) {
1045           ir_read_error(st, NULL, "when reading coordinate in (%s ...)",
1046     		    tag->value());
1047           return NULL;
1048        }
1049     
1050        // Read texel offset, i.e. (0 0 0)
1051        s_list *offset_list = SX_AS_LIST(coordinate_expr->next);
1052        if (offset_list == NULL || offset_list->length() != 3) {
1053           ir_read_error(st, offset_list, "expected (<int> <int> <int>)");
1054           return NULL;
1055        }
1056        s_int *offset_x = SX_AS_INT(offset_list->subexpressions.head);
1057        s_int *offset_y = SX_AS_INT(offset_x->next);
1058        s_int *offset_z = SX_AS_INT(offset_y->next);
1059        if (offset_x == NULL || offset_y == NULL || offset_z == NULL) {
1060           ir_read_error(st, offset_list, "expected (<int> <int> <int>)");
1061           return NULL;
1062        }
1063        tex->offsets[0] = offset_x->value();
1064        tex->offsets[1] = offset_y->value();
1065        tex->offsets[2] = offset_z->value();
1066     
1067        if (op == ir_txf) {
1068           s_expression *lod_expr = (s_expression *) offset_list->next;
1069           tex->lod_info.lod = read_rvalue(st, lod_expr);
1070           if (tex->lod_info.lod == NULL) {
1071     	 ir_read_error(st, NULL, "when reading LOD in (txf ...)");
1072     	 return NULL;
1073           }
1074        } else {
1075           s_expression *proj_expr = (s_expression *) offset_list->next;
1076           s_int *proj_as_int = SX_AS_INT(proj_expr);
1077           if (proj_as_int && proj_as_int->value() == 1) {
1078     	 tex->projector = NULL;
1079           } else {
1080     	 tex->projector = read_rvalue(st, proj_expr);
1081     	 if (tex->projector == NULL) {
1082     	    ir_read_error(st, NULL, "when reading projective divide in (%s ..)",
1083     	                  tag->value());
1084     	    return NULL;
1085     	 }
1086           }
1087     
1088           s_list *shadow_list = SX_AS_LIST(proj_expr->next);
1089           if (shadow_list == NULL) {
1090     	 ir_read_error(st, NULL, "shadow comparitor must be a list");
1091     	 return NULL;
1092           }
1093           if (shadow_list->subexpressions.is_empty()) {
1094     	 tex->shadow_comparitor= NULL;
1095           } else {
1096     	 tex->shadow_comparitor = read_rvalue(st, shadow_list);
1097     	 if (tex->shadow_comparitor == NULL) {
1098     	    ir_read_error(st, NULL, "when reading shadow comparitor in (%s ..)",
1099     			  tag->value());
1100     	    return NULL;
1101     	 }
1102           }
1103           s_expression *lod_expr = (s_expression *) shadow_list->next;
1104     
1105           switch (op) {
1106           case ir_txb:
1107     	 tex->lod_info.bias = read_rvalue(st, lod_expr);
1108     	 if (tex->lod_info.bias == NULL) {
1109     	    ir_read_error(st, NULL, "when reading LOD bias in (txb ...)");
1110     	    return NULL;
1111     	 }
1112     	 break;
1113           case ir_txl:
1114     	 tex->lod_info.lod = read_rvalue(st, lod_expr);
1115     	 if (tex->lod_info.lod == NULL) {
1116     	    ir_read_error(st, NULL, "when reading LOD in (txl ...)");
1117     	    return NULL;
1118     	 }
1119     	 break;
1120           case ir_txd: {
1121     	 s_list *lod_list = SX_AS_LIST(lod_expr);
1122     	 if (lod_list->length() != 2) {
1123     	    ir_read_error(st, lod_expr, "expected (dPdx dPdy) in (txd ...)");
1124     	    return NULL;
1125     	 }
1126     	 s_expression *dx_expr = (s_expression *) lod_list->subexpressions.head;
1127     	 s_expression *dy_expr = (s_expression *) dx_expr->next;
1128     
1129     	 tex->lod_info.grad.dPdx = read_rvalue(st, dx_expr);
1130     	 if (tex->lod_info.grad.dPdx == NULL) {
1131     	    ir_read_error(st, NULL, "when reading dPdx in (txd ...)");
1132     	    return NULL;
1133     	 }
1134     	 tex->lod_info.grad.dPdy = read_rvalue(st, dy_expr);
1135     	 if (tex->lod_info.grad.dPdy == NULL) {
1136     	    ir_read_error(st, NULL, "when reading dPdy in (txd ...)");
1137     	    return NULL;
1138     	 }
1139     	 break;
1140           }
1141           default:
1142     	 // tex doesn't have any extra parameters and txf was handled earlier.
1143     	 break;
1144           };
1145        }
1146        return tex;
1147     }
