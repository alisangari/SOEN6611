5     // This file defines a bunch of recurring problems in the Chromium C++ code.
6     //
7     // Checks that are implemented:
8     // - Constructors/Destructors should not be inlined if they are of a complex
9     //   class type.
10     // - Missing "virtual" keywords on methods that should be virtual.
11     // - Non-annotated overriding virtual methods.
12     // - Virtual methods with nonempty implementations in their headers.
13     // - Classes that derive from base::RefCounted / base::RefCountedThreadSafe
14     //   should have protected or private destructors.
15

32     bool TypeHasNonTrivialDtor(const Type* type) {
33       if (const CXXRecordDecl* cxx_r = type->getPointeeCXXRecordDecl())
34         return cxx_r->hasTrivialDestructor();
35     
36       return false;
37     }
38     
39     // Returns the underlying Type for |type| by expanding typedefs and removing
40     // any namespace qualifiers.
41     const Type* UnwrapType(const Type* type) {
42       if (const ElaboratedType* elaborated = dyn_cast<ElaboratedType>(type))
43         return UnwrapType(elaborated->getNamedType().getTypePtr());
44       if (const TypedefType* typedefed = dyn_cast<TypedefType>(type))
45         return UnwrapType(typedefed->desugar().getTypePtr());
46       return type;
47     }

49     // Searches for constructs that we know we don't want in the Chromium code base.
50     class FindBadConstructsConsumer : public ChromeClassTester {
51      public:
52       FindBadConstructsConsumer(CompilerInstance& instance,
53                                 bool check_refcounted_dtors,
54                                 bool check_virtuals_in_implementations)
55           : ChromeClassTester(instance),
56             check_refcounted_dtors_(check_refcounted_dtors),
57             check_virtuals_in_implementations_(check_virtuals_in_implementations) {
58       }
59     
60       virtual void CheckChromeClass(SourceLocation record_location,
61                                     CXXRecordDecl* record) {
62         bool implementation_file = InImplementationFile(record_location);
63     
64         if (!implementation_file) {
65           // Only check for "heavy" constructors/destructors in header files;
66           // within implementation files, there is no performance cost.
67           CheckCtorDtorWeight(record_location, record);
68         }
69     
70         if (!implementation_file || check_virtuals_in_implementations_) {
71           bool warn_on_inline_bodies = !implementation_file;
72     
73           // Check that all virtual methods are marked accordingly with both
74           // virtual and OVERRIDE.
75           CheckVirtualMethods(record_location, record, warn_on_inline_bodies);
76         }
77     
78         if (check_refcounted_dtors_)
79           CheckRefCountedDtors(record_location, record);
80       }
81     
82      private:
83       bool check_refcounted_dtors_;
84       bool check_virtuals_in_implementations_;
85     
86       // Returns true if |base| specifies one of the Chromium reference counted
87       // classes (base::RefCounted / base::RefCountedThreadSafe). |user_data| is
88       // ignored.
89       static bool IsRefCountedCallback(const CXXBaseSpecifier* base,
90                                        CXXBasePath& path,
91                                        void* user_data) {
92         FindBadConstructsConsumer* self =
93             static_cast<FindBadConstructsConsumer*>(user_data);
94     
95         const TemplateSpecializationType* base_type =
96             dyn_cast<TemplateSpecializationType>(
97                 UnwrapType(base->getType().getTypePtr()));
98         if (!base_type) {
99           // Base-most definition is not a template, so this cannot derive from
100           // base::RefCounted. However, it may still be possible to use with a
101           // scoped_refptr<> and support ref-counting, so this is not a perfect
102           // guarantee of safety.
103           return false;
104         }
105     
106         TemplateName name = base_type->getTemplateName();
107         if (TemplateDecl* decl = name.getAsTemplateDecl()) {
108           std::string base_name = decl->getNameAsString();
109     
110           // Check for both base::RefCounted and base::RefCountedThreadSafe.
111           if (base_name.compare(0, 10, "RefCounted") == 0 &&
112               self->GetNamespace(decl) == "base") {
113             return true;
114           }
115         }
116         return false;
117       }

119       // Prints errors if the destructor of a RefCounted class is public.
120       void CheckRefCountedDtors(SourceLocation record_location,
121                                 CXXRecordDecl* record) {
122         // Skip anonymous structs.
123         if (record->getIdentifier() == NULL)
124           return;
125     
126         CXXBasePaths paths;
127         if (!record->lookupInBases(
128                 &FindBadConstructsConsumer::IsRefCountedCallback, this, paths)) {
129           return;  // Class does not derive from a ref-counted base class.
130         }
131     
132         if (!record->hasUserDeclaredDestructor()) {
133           emitWarning(
134               record_location,
135               "Classes that are ref-counted should have explicit "
136               "destructors that are protected or private.");
137         } else if (CXXDestructorDecl* dtor = record->getDestructor()) {
138           if (dtor->getAccess() == AS_public) {
139             emitWarning(
140                 dtor->getInnerLocStart(),
141                 "Classes that are ref-counted should not have "
142                 "public destructors.");
143           }
144         }
145       }
146     
147       // Prints errors if the constructor/destructor weight is too heavy.
148       void CheckCtorDtorWeight(SourceLocation record_location,
149                                CXXRecordDecl* record) {
150         // We don't handle anonymous structs. If this record doesn't have a
151         // name, it's of the form:
152         //
153         // struct {
154         //   ...
155         // } name_;
156         if (record->getIdentifier() == NULL)
157           return;
158     
159         // Count the number of templated base classes as a feature of whether the
160         // destructor can be inlined.
161         int templated_base_classes = 0;
162         for (CXXRecordDecl::base_class_const_iterator it = record->bases_begin();
163              it != record->bases_end(); ++it) {
164           if (it->getTypeSourceInfo()->getTypeLoc().getTypeLocClass() ==
165               TypeLoc::TemplateSpecialization) {
166             ++templated_base_classes;
167           }
168         }
169     
170         // Count the number of trivial and non-trivial member variables.
171         int trivial_member = 0;
172         int non_trivial_member = 0;
173         int templated_non_trivial_member = 0;
174         for (RecordDecl::field_iterator it = record->field_begin();
175              it != record->field_end(); ++it) {
176           CountType(it->getType().getTypePtr(),
177                     &trivial_member,
178                     &non_trivial_member,
179                     &templated_non_trivial_member);
180         }
181     
182         // Check to see if we need to ban inlined/synthesized constructors. Note
183         // that the cutoffs here are kind of arbitrary. Scores over 10 break.
184         int dtor_score = 0;
185         // Deriving from a templated base class shouldn't be enough to trigger
186         // the ctor warning, but if you do *anything* else, it should.
187         //
188         // TODO(erg): This is motivated by templated base classes that don't have
189         // any data members. Somehow detect when templated base classes have data
190         // members and treat them differently.
191         dtor_score += templated_base_classes * 9;
192         // Instantiating a template is an insta-hit.
193         dtor_score += templated_non_trivial_member * 10;
194         // The fourth normal class member should trigger the warning.
195         dtor_score += non_trivial_member * 3;
196     
197         int ctor_score = dtor_score;
198         // You should be able to have 9 ints before we warn you.
199         ctor_score += trivial_member;
200     
201         if (ctor_score >= 10) {
202           if (!record->hasUserDeclaredConstructor()) {
203             emitWarning(record_location,
204                         "Complex class/struct needs an explicit out-of-line "
205                         "constructor.");
206           } else {
207             // Iterate across all the constructors in this file and yell if we
208             // find one that tries to be inline.
209             for (CXXRecordDecl::ctor_iterator it = record->ctor_begin();
210                  it != record->ctor_end(); ++it) {
211               if (it->hasInlineBody()) {
212                 if (it->isCopyConstructor() &&
213                     !record->hasUserDeclaredCopyConstructor()) {
214                   emitWarning(record_location,
215                               "Complex class/struct needs an explicit out-of-line "
216                               "copy constructor.");
217                 } else {
218                   emitWarning(it->getInnerLocStart(),
219                               "Complex constructor has an inlined body.");
220                 }
221               }
222             }
223           }
224         }
225     
226         // The destructor side is equivalent except that we don't check for
227         // trivial members; 20 ints don't need a destructor.
228         if (dtor_score >= 10 && !record->hasTrivialDestructor()) {
229           if (!record->hasUserDeclaredDestructor()) {
230             emitWarning(
231                 record_location,
232                 "Complex class/struct needs an explicit out-of-line "
233                 "destructor.");
234           } else if (CXXDestructorDecl* dtor = record->getDestructor()) {
235             if (dtor->hasInlineBody()) {
236               emitWarning(dtor->getInnerLocStart(),
237                           "Complex destructor has an inline body.");
238             }
239           }
240         }
241       }

243       void CheckVirtualMethod(const CXXMethodDecl* method,
244                               bool warn_on_inline_bodies) {
245         if (!method->isVirtual())
246           return;
247     
248         if (!method->isVirtualAsWritten()) {
249           SourceLocation loc = method->getTypeSpecStartLoc();
250           if (isa<CXXDestructorDecl>(method))
251             loc = method->getInnerLocStart();
252           emitWarning(loc, "Overriding method must have \"virtual\" keyword.");
253         }
254     
255         // Virtual methods should not have inline definitions beyond "{}". This
256         // only matters for header files.
257         if (warn_on_inline_bodies && method->hasBody() &&
258             method->hasInlineBody()) {
259           if (CompoundStmt* cs = dyn_cast<CompoundStmt>(method->getBody())) {
260             if (cs->size()) {
261               emitWarning(
262                   cs->getLBracLoc(),
263                   "virtual methods with non-empty bodies shouldn't be "
264                   "declared inline.");
265             }
266           }
267         }
268       }

270       bool InTestingNamespace(const Decl* record) {
271         return GetNamespace(record).find("testing") != std::string::npos;
272       }
273     
274       bool IsMethodInBannedNamespace(const CXXMethodDecl* method) {
275         if (InBannedNamespace(method))
276           return true;
277         for (CXXMethodDecl::method_iterator i = method->begin_overridden_methods();
278              i != method->end_overridden_methods();
279              ++i) {
280           const CXXMethodDecl* overridden = *i;
281           if (IsMethodInBannedNamespace(overridden))
282             return true;
283         }
284     
285         return false;
286       }
287     
288       void CheckOverriddenMethod(const CXXMethodDecl* method) {
289         if (!method->size_overridden_methods() || method->getAttr<OverrideAttr>())
290           return;
291     
292         if (isa<CXXDestructorDecl>(method) || method->isPure())
293           return;
294     
295         if (IsMethodInBannedNamespace(method))
296           return;
297     
298         SourceLocation loc = method->getTypeSpecStartLoc();
299         emitWarning(loc, "Overriding method must be marked with OVERRIDE.");
300       }
301     
302       // Makes sure there is a "virtual" keyword on virtual methods.
303       //
304       // Gmock objects trigger these for each MOCK_BLAH() macro used. So we have a
305       // trick to get around that. If a class has member variables whose types are
306       // in the "testing" namespace (which is how gmock works behind the scenes),
307       // there's a really high chance we won't care about these errors
308       void CheckVirtualMethods(SourceLocation record_location,
309                                CXXRecordDecl* record,
310                                bool warn_on_inline_bodies) {
311         for (CXXRecordDecl::field_iterator it = record->field_begin();
312              it != record->field_end(); ++it) {
313           CXXRecordDecl* record_type =
314               it->getTypeSourceInfo()->getTypeLoc().getTypePtr()->
315               getAsCXXRecordDecl();
316           if (record_type) {
317             if (InTestingNamespace(record_type)) {
318               return;
319             }
320           }
321         }
322     
323         for (CXXRecordDecl::method_iterator it = record->method_begin();
324              it != record->method_end(); ++it) {
325           if (it->isCopyAssignmentOperator() || isa<CXXConstructorDecl>(*it)) {
326             // Ignore constructors and assignment operators.
327           } else if (isa<CXXDestructorDecl>(*it) &&
328               !record->hasUserDeclaredDestructor()) {
329             // Ignore non-user-declared destructors.
330           } else {
331             CheckVirtualMethod(*it, warn_on_inline_bodies);
332             CheckOverriddenMethod(*it);
333           }
334         }
335       }

337       void CountType(const Type* type,
338                      int* trivial_member,
339                      int* non_trivial_member,
340                      int* templated_non_trivial_member) {
341         switch (type->getTypeClass()) {
342           case Type::Record: {
343             // Simplifying; the whole class isn't trivial if the dtor is, but
344             // we use this as a signal about complexity.
345             if (TypeHasNonTrivialDtor(type))
346               (*trivial_member)++;
347             else
348               (*non_trivial_member)++;
349             break;
350           }
351           case Type::TemplateSpecialization: {
352             TemplateName name =
353                 dyn_cast<TemplateSpecializationType>(type)->getTemplateName();
354             bool whitelisted_template = false;
355     
356             // HACK: I'm at a loss about how to get the syntax checker to get
357             // whether a template is exterened or not. For the first pass here,
358             // just do retarded string comparisons.
359             if (TemplateDecl* decl = name.getAsTemplateDecl()) {
360               std::string base_name = decl->getNameAsString();
361               if (base_name == "basic_string")
362                 whitelisted_template = true;
363             }
364     
365             if (whitelisted_template)
366               (*non_trivial_member)++;
367             else
368               (*templated_non_trivial_member)++;
369             break;
370           }
371           case Type::Elaborated: {
372             CountType(
373                 dyn_cast<ElaboratedType>(type)->getNamedType().getTypePtr(),
374                 trivial_member, non_trivial_member, templated_non_trivial_member);
375             break;
376           }
377           case Type::Typedef: {
378             while (const TypedefType* TT = dyn_cast<TypedefType>(type)) {
379               type = TT->getDecl()->getUnderlyingType().getTypePtr();
380             }
381             CountType(type, trivial_member, non_trivial_member,
382                       templated_non_trivial_member);
383             break;
384           }
385           default: {
386             // Stupid assumption: anything we see that isn't the above is one of
387             // the 20 integer types.
388             (*trivial_member)++;
389             break;
390           }
391         }
392       }

395     class FindBadConstructsAction : public PluginASTAction {
396      public:
397       FindBadConstructsAction()
398           : check_refcounted_dtors_(true),
399             check_virtuals_in_implementations_(true) {
400       }
401     
402      protected:
403       // Overridden from PluginASTAction:
404       virtual ASTConsumer* CreateASTConsumer(CompilerInstance& instance,
405                                              llvm::StringRef ref) {
406         return new FindBadConstructsConsumer(
407             instance, check_refcounted_dtors_, check_virtuals_in_implementations_);
408       }
409     
410       virtual bool ParseArgs(const CompilerInstance& instance,
411                              const std::vector<std::string>& args) {
412         bool parsed = true;
413     
414         for (size_t i = 0; i < args.size() && parsed; ++i) {
415           if (args[i] == "skip-refcounted-dtors") {
416             check_refcounted_dtors_ = false;
417           } else if (args[i] == "skip-virtuals-in-implementations") {
418             check_virtuals_in_implementations_ = false;
419           } else {
420             parsed = false;
421             llvm::errs() << "Unknown clang plugin argument: " << args[i] << "\n";
422           }
423         }
424     
425         return parsed;
426       }
427     
428      private:
429       bool check_refcounted_dtors_;
430       bool check_virtuals_in_implementations_;
431     };
