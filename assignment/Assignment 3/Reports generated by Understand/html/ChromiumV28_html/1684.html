27       * Eliminates unused functions from the linked program.
28       */

36      class signature_entry : public exec_node
37      {
38      public:
39         signature_entry(ir_function_signature *sig)
40         {
41            this->signature = sig;
42            this->used = false;
43         }
44     
45         ir_function_signature *signature;
46         bool used;
47      };
48     
49      class ir_dead_functions_visitor : public ir_hierarchical_visitor {
50      public:
51         ir_dead_functions_visitor()
52         {
53            this->mem_ctx = talloc_new(NULL);
54         }
55     
56         ~ir_dead_functions_visitor()
57         {
58            talloc_free(this->mem_ctx);
59         }
60     
61         virtual ir_visitor_status visit_enter(ir_function_signature *);
62         virtual ir_visitor_status visit_enter(ir_call *);
63     
64         signature_entry *get_signature_entry(ir_function_signature *var);
65     
66         bool (*predicate)(ir_instruction *ir);
67     
68         /* List of signature_entry */
69         exec_list signature_list;
70         void *mem_ctx;
71      };
72     
73     
74      signature_entry *
75      ir_dead_functions_visitor::get_signature_entry(ir_function_signature *sig)
76      {
77         foreach_iter(exec_list_iterator, iter, this->signature_list) {
78            signature_entry *entry = (signature_entry *)iter.get();
79            if (entry->signature == sig)
80     	  return entry;
81         }
82     
83         signature_entry *entry = new(mem_ctx) signature_entry(sig);
84         this->signature_list.push_tail(entry);
85         return entry;
86      }
87     
88     
89      ir_visitor_status
90      ir_dead_functions_visitor::visit_enter(ir_function_signature *ir)
91      {
92         signature_entry *entry = this->get_signature_entry(ir);
93     
94         if (safe_strcmp(ir->function_name(), "main") == 0) {
95            entry->used = true;
96         }
97     
98         return visit_continue;
99      }
100     
101     
102      ir_visitor_status
103      ir_dead_functions_visitor::visit_enter(ir_call *ir)
104      {
105         signature_entry *entry = this->get_signature_entry(ir->get_callee());
106     
107         entry->used = true;
108     
109        return visit_continue;
110     }

112     bool
113     do_dead_functions(exec_list *instructions)
114     {
115        ir_dead_functions_visitor v;
116        bool progress = false;
117     
118        visit_list_elements(&v, instructions);
119     
120        /* Now that we've figured out which function signatures are used, remove
121         * the unused ones, and remove function definitions that have no more
122         * signatures.
123         */
124         foreach_iter(exec_list_iterator, iter, v.signature_list) {
125           signature_entry *entry = (signature_entry *)iter.get();
126     
127           if (!entry->used) {
128     	 entry->signature->remove();
129     	 progress = true;
130           }
131           delete(entry);
132        }
133     
134        /* We don't just do this above when we nuked a signature because of
135         * const pointers.
136         */
137        foreach_iter(exec_list_iterator, iter, *instructions) {
138           ir_instruction *ir = (ir_instruction *)iter.get();
139           ir_function *func = ir->as_function();
140     
141           if (func && func->signatures.is_empty()) {
142     	 /* At this point (post-linking), the symbol table is no
143     	  * longer in use, so not removing the function from the
144     	  * symbol table should be OK.
145     	  */
146     	 func->remove();
147     	 progress = true;
148           }
149        }
150     
151        return progress;
152     }
