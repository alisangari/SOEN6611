24     /**
25      * \file ast_to_hir.c
26      * Convert abstract syntax to to high-level intermediate reprensentation (HIR).
27      *
28      * During the conversion to HIR, the majority of the symantic checking is
29      * preformed on the program.  This includes:
30      *
31      *    * Symbol table management
32      *    * Type checking
33      *    * Function binding
34      *
35      * The majority of this work could be done during parsing, and the parser could
36      * probably generate HIR directly.  However, this results in frequent changes
37      * to the parser code.  Since we do not assume that every system this complier
38      * is built on will have Flex and Bison installed, we have to store the code
39      * generated by these tools in our version control system.  In other parts of
40      * the system we've seen problems where a parser was changed but the generated
41      * code was not committed, merge conflicts where created because two developers
42      * had slightly different versions of Bison installed, etc.
43      *
44      * I have also noticed that running Bison generated parsers in GDB is very
45      * irritating.  When you get a segfault on '$$ = $1->foo', you can't very
46      * well 'print $1' in GDB.
47      *
48      * As a result, my preference is to put as little C code as possible in the
49      * parser (and lexer) sources.
50      */

60     void
61     _mesa_ast_to_hir(exec_list *instructions, struct _mesa_glsl_parse_state *state)
62     {
63        _mesa_glsl_initialize_variables(instructions, state);
64        _mesa_glsl_initialize_functions(instructions, state);
65     
66        state->symbols->language_version = state->language_version;
67     
68        state->current_function = NULL;
69     
70        /* Section 4.2 of the GLSL 1.20 specification states:
71         * "The built-in functions are scoped in a scope outside the global scope
72         *  users declare global variables in.  That is, a shader's global scope,
73         *  available for user-defined functions and global variables, is nested
74         *  inside the scope containing the built-in functions."
75         *
76         * Since built-in functions like ftransform() access built-in variables,
77         * it follows that those must be in the outer scope as well.
78         *
79         * We push scope here to create this nesting effect...but don't pop.
80         * This way, a shader's globals are still in the symbol table for use
81         * by the linker.
82         */
83        state->symbols->push_scope();
84     
85        foreach_list_typed (ast_node, ast, link, & state->translation_unit)
86           ast->hir(instructions, state);
87     }

90     /**
91      * If a conversion is available, convert one operand to a different type
92      *
93      * The \c from \c ir_rvalue is converted "in place".
94      *
95      * \param to     Type that the operand it to be converted to
96      * \param from   Operand that is being converted
97      * \param state  GLSL compiler state
98      *
99      * \return
100      * If a conversion is possible (or unnecessary), \c true is returned.
101      * Otherwise \c false is returned.
102      */
103     bool
104     apply_implicit_conversion(const glsl_type *to, ir_rvalue * &from,
105     			  struct _mesa_glsl_parse_state *state)
106     {
107        void *ctx = state;
108        if (to->base_type == from->type->base_type)
109           return true;
110     
111        /* This conversion was added in GLSL 1.20.  If the compilation mode is
112         * GLSL 1.10, the conversion is skipped.
113         */
114        if (state->language_version < 120)
115           return false;
116     
117        /* From page 27 (page 33 of the PDF) of the GLSL 1.50 spec:
118         *
119         *    "There are no implicit array or structure conversions. For
120         *    example, an array of int cannot be implicitly converted to an
121         *    array of float. There are no implicit conversions between
122         *    signed and unsigned integers."
123         */
124        /* FINISHME: The above comment is partially a lie.  There is int/uint
125         * FINISHME: conversion for immediate constants.
126         */
127        if (!to->is_float() || !from->type->is_numeric())
128           return false;
129     
130        /* Convert to a floating point type with the same number of components
131         * as the original type - i.e. int to float, not int to vec4.
132         */
133        to = glsl_type::get_instance(GLSL_TYPE_FLOAT, from->type->vector_elements,
134     			        from->type->matrix_columns);
135     
136        switch (from->type->base_type) {
137        case GLSL_TYPE_INT:
138           from = new(ctx) ir_expression(ir_unop_i2f, to, from, NULL);
139           break;
140        case GLSL_TYPE_UINT:
141           from = new(ctx) ir_expression(ir_unop_u2f, to, from, NULL);
142           break;
143        case GLSL_TYPE_BOOL:
144           from = new(ctx) ir_expression(ir_unop_b2f, to, from, NULL);
145           break;
146        default:
147           assert(0);
148        }
149     
150        return true;
151     }

154     static const struct glsl_type *
155     arithmetic_result_type(ir_rvalue * &value_a, ir_rvalue * &value_b,
156     		       bool multiply,
157     		       struct _mesa_glsl_parse_state *state, YYLTYPE *loc)
158     {
159        const glsl_type *type_a = value_a->type;
160        const glsl_type *type_b = value_b->type;
161     
162        /* From GLSL 1.50 spec, page 56:
163         *
164         *    "The arithmetic binary operators add (+), subtract (-),
165         *    multiply (*), and divide (/) operate on integer and
166         *    floating-point scalars, vectors, and matrices."
167         */
168        if (!type_a->is_numeric() || !type_b->is_numeric()) {
169           _mesa_glsl_error(loc, state,
170     		       "Operands to arithmetic operators must be numeric");
171           return glsl_type::error_type;
172        }
173     
174     
175        /*    "If one operand is floating-point based and the other is
176         *    not, then the conversions from Section 4.1.10 "Implicit
177         *    Conversions" are applied to the non-floating-point-based operand."
178         */
179        if (!apply_implicit_conversion(type_a, value_b, state)
180            && !apply_implicit_conversion(type_b, value_a, state)) {
181           _mesa_glsl_error(loc, state,
182     		       "Could not implicitly convert operands to "
183     		       "arithmetic operator");
184           return glsl_type::error_type;
185        }
186        type_a = value_a->type;
187        type_b = value_b->type;
188     
189        /*    "If the operands are integer types, they must both be signed or
190         *    both be unsigned."
191         *
192         * From this rule and the preceeding conversion it can be inferred that
193         * both types must be GLSL_TYPE_FLOAT, or GLSL_TYPE_UINT, or GLSL_TYPE_INT.
194         * The is_numeric check above already filtered out the case where either
195         * type is not one of these, so now the base types need only be tested for
196         * equality.
197         */
198        if (type_a->base_type != type_b->base_type) {
199           _mesa_glsl_error(loc, state,
200     		       "base type mismatch for arithmetic operator");
201           return glsl_type::error_type;
202        }
203     
204        /*    "All arithmetic binary operators result in the same fundamental type
205         *    (signed integer, unsigned integer, or floating-point) as the
206         *    operands they operate on, after operand type conversion. After
207         *    conversion, the following cases are valid
208         *
209         *    * The two operands are scalars. In this case the operation is
210         *      applied, resulting in a scalar."
211         */
212        if (type_a->is_scalar() && type_b->is_scalar())
213           return type_a;
214     
215        /*   "* One operand is a scalar, and the other is a vector or matrix.
216         *      In this case, the scalar operation is applied independently to each
217         *      component of the vector or matrix, resulting in the same size
218         *      vector or matrix."
219         */
220        if (type_a->is_scalar()) {
221           if (!type_b->is_scalar())
222     	 return type_b;
223        } else if (type_b->is_scalar()) {
224           return type_a;
225        }
226     
227        /* All of the combinations of <scalar, scalar>, <vector, scalar>,
228         * <scalar, vector>, <scalar, matrix>, and <matrix, scalar> have been
229         * handled.
230         */
231        assert(!type_a->is_scalar());
232        assert(!type_b->is_scalar());
233     
234        /*   "* The two operands are vectors of the same size. In this case, the
235         *      operation is done component-wise resulting in the same size
236         *      vector."
237         */
238        if (type_a->is_vector() && type_b->is_vector()) {
239           if (type_a == type_b) {
240     	 return type_a;
241           } else {
242     	 _mesa_glsl_error(loc, state,
243     			  "vector size mismatch for arithmetic operator");
244     	 return glsl_type::error_type;
245           }
246        }
247     
248        /* All of the combinations of <scalar, scalar>, <vector, scalar>,
249         * <scalar, vector>, <scalar, matrix>, <matrix, scalar>, and
250         * <vector, vector> have been handled.  At least one of the operands must
251         * be matrix.  Further, since there are no integer matrix types, the base
252         * type of both operands must be float.
253         */
254        assert(type_a->is_matrix() || type_b->is_matrix());
255        assert(type_a->base_type == GLSL_TYPE_FLOAT);
256        assert(type_b->base_type == GLSL_TYPE_FLOAT);
257     
258        /*   "* The operator is add (+), subtract (-), or divide (/), and the
259         *      operands are matrices with the same number of rows and the same
260         *      number of columns. In this case, the operation is done component-
261         *      wise resulting in the same size matrix."
262         *    * The operator is multiply (*), where both operands are matrices or
263         *      one operand is a vector and the other a matrix. A right vector
264         *      operand is treated as a column vector and a left vector operand as a
265         *      row vector. In all these cases, it is required that the number of
266         *      columns of the left operand is equal to the number of rows of the
267         *      right operand. Then, the multiply (*) operation does a linear
268         *      algebraic multiply, yielding an object that has the same number of
269         *      rows as the left operand and the same number of columns as the right
270         *      operand. Section 5.10 "Vector and Matrix Operations" explains in
271         *      more detail how vectors and matrices are operated on."
272         */
273        if (! multiply) {
274           if (type_a == type_b)
275     	 return type_a;
276        } else {
277           if (type_a->is_matrix() && type_b->is_matrix()) {
278     	 /* Matrix multiply.  The columns of A must match the rows of B.  Given
279     	  * the other previously tested constraints, this means the vector type
280     	  * of a row from A must be the same as the vector type of a column from
281     	  * B.
282     	  */
283     	 if (type_a->row_type() == type_b->column_type()) {
284     	    /* The resulting matrix has the number of columns of matrix B and
285     	     * the number of rows of matrix A.  We get the row count of A by
286     	     * looking at the size of a vector that makes up a column.  The
287     	     * transpose (size of a row) is done for B.
288     	     */
289     	    const glsl_type *const type =
290     	       glsl_type::get_instance(type_a->base_type,
291     				       type_a->column_type()->vector_elements,
292     				       type_b->row_type()->vector_elements);
293     	    assert(type != glsl_type::error_type);
294     
295     	    return type;
296     	 }
297           } else if (type_a->is_matrix()) {
298     	 /* A is a matrix and B is a column vector.  Columns of A must match
299     	  * rows of B.  Given the other previously tested constraints, this
300     	  * means the vector type of a row from A must be the same as the
301     	  * vector the type of B.
302     	  */
303     	 if (type_a->row_type() == type_b) {
304     	    /* The resulting vector has a number of elements equal to
305     	     * the number of rows of matrix A. */
306     	    const glsl_type *const type =
307     	       glsl_type::get_instance(type_a->base_type,
308     				       type_a->column_type()->vector_elements,
309     				       1);
310     	    assert(type != glsl_type::error_type);
311     
312     	    return type;
313     	 }
314           } else {
315     	 assert(type_b->is_matrix());
316     
317     	 /* A is a row vector and B is a matrix.  Columns of A must match rows
318     	  * of B.  Given the other previously tested constraints, this means
319     	  * the type of A must be the same as the vector type of a column from
320     	  * B.
321     	  */
322     	 if (type_a == type_b->column_type()) {
323     	    /* The resulting vector has a number of elements equal to
324     	     * the number of columns of matrix B. */
325     	    const glsl_type *const type =
326     	       glsl_type::get_instance(type_a->base_type,
327     				       type_b->row_type()->vector_elements,
328     				       1);
329     	    assert(type != glsl_type::error_type);
330     
331     	    return type;
332     	 }
333           }
334     
335           _mesa_glsl_error(loc, state, "size mismatch for matrix multiplication");
336           return glsl_type::error_type;
337        }
338     
339     
340        /*    "All other cases are illegal."
341         */
342        _mesa_glsl_error(loc, state, "type mismatch");
343        return glsl_type::error_type;
344     }

347     static const struct glsl_type *
348     unary_arithmetic_result_type(const struct glsl_type *type,
349     			     struct _mesa_glsl_parse_state *state, YYLTYPE *loc)
350     {
351        /* From GLSL 1.50 spec, page 57:
352         *
353         *    "The arithmetic unary operators negate (-), post- and pre-increment
354         *     and decrement (-- and ++) operate on integer or floating-point
355         *     values (including vectors and matrices). All unary operators work
356         *     component-wise on their operands. These result with the same type
357         *     they operated on."
358         */
359        if (!type->is_numeric()) {
360           _mesa_glsl_error(loc, state,
361     		       "Operands to arithmetic operators must be numeric");
362           return glsl_type::error_type;
363        }
364     
365        return type;
366     }

369     static const struct glsl_type *
370     modulus_result_type(const struct glsl_type *type_a,
371     		    const struct glsl_type *type_b,
372     		    struct _mesa_glsl_parse_state *state, YYLTYPE *loc)
373     {
374        /* From GLSL 1.50 spec, page 56:
375         *    "The operator modulus (%) operates on signed or unsigned integers or
376         *    integer vectors. The operand types must both be signed or both be
377         *    unsigned."
378         */
379        if (!type_a->is_integer() || !type_b->is_integer()
380            || (type_a->base_type != type_b->base_type)) {
381           _mesa_glsl_error(loc, state, "type mismatch");
382           return glsl_type::error_type;
383        }
384     
385        /*    "The operands cannot be vectors of differing size. If one operand is
386         *    a scalar and the other vector, then the scalar is applied component-
387         *    wise to the vector, resulting in the same type as the vector. If both
388         *    are vectors of the same size, the result is computed component-wise."
389         */
390        if (type_a->is_vector()) {
391           if (!type_b->is_vector()
392     	  || (type_a->vector_elements == type_b->vector_elements))
393     	 return type_a;
394        } else
395           return type_b;
396     
397        /*    "The operator modulus (%) is not defined for any other data types
398         *    (non-integer types)."
399         */
400        _mesa_glsl_error(loc, state, "type mismatch");
401        return glsl_type::error_type;
402     }

405     static const struct glsl_type *
406     relational_result_type(ir_rvalue * &value_a, ir_rvalue * &value_b,
407     		       struct _mesa_glsl_parse_state *state, YYLTYPE *loc)
408     {
409        const glsl_type *type_a = value_a->type;
410        const glsl_type *type_b = value_b->type;
411     
412        /* From GLSL 1.50 spec, page 56:
413         *    "The relational operators greater than (>), less than (<), greater
414         *    than or equal (>=), and less than or equal (<=) operate only on
415         *    scalar integer and scalar floating-point expressions."
416         */
417        if (!type_a->is_numeric()
418            || !type_b->is_numeric()
419            || !type_a->is_scalar()
420            || !type_b->is_scalar()) {
421           _mesa_glsl_error(loc, state,
422     		       "Operands to relational operators must be scalar and "
423     		       "numeric");
424           return glsl_type::error_type;
425        }
426     
427        /*    "Either the operands' types must match, or the conversions from
428         *    Section 4.1.10 "Implicit Conversions" will be applied to the integer
429         *    operand, after which the types must match."
430         */
431        if (!apply_implicit_conversion(type_a, value_b, state)
432            && !apply_implicit_conversion(type_b, value_a, state)) {
433           _mesa_glsl_error(loc, state,
434     		       "Could not implicitly convert operands to "
435     		       "relational operator");
436           return glsl_type::error_type;
437        }
438        type_a = value_a->type;
439        type_b = value_b->type;
440     
441        if (type_a->base_type != type_b->base_type) {
442           _mesa_glsl_error(loc, state, "base type mismatch");
443           return glsl_type::error_type;
444        }
445     
446        /*    "The result is scalar Boolean."
447         */
448        return glsl_type::bool_type;
449     }

452     /**
453      * Validates that a value can be assigned to a location with a specified type
454      *
455      * Validates that \c rhs can be assigned to some location.  If the types are
456      * not an exact match but an automatic conversion is possible, \c rhs will be
457      * converted.
458      *
459      * \return
460      * \c NULL if \c rhs cannot be assigned to a location with type \c lhs_type.
461      * Otherwise the actual RHS to be assigned will be returned.  This may be
462      * \c rhs, or it may be \c rhs after some type conversion.
463      *
464      * \note
465      * In addition to being used for assignments, this function is used to
466      * type-check return values.
467      */
468     ir_rvalue *
469     validate_assignment(struct _mesa_glsl_parse_state *state,
470     		    const glsl_type *lhs_type, ir_rvalue *rhs)
471     {
472        const glsl_type *rhs_type = rhs->type;
473     
474        /* If there is already some error in the RHS, just return it.  Anything
475         * else will lead to an avalanche of error message back to the user.
476         */
477        if (rhs_type->is_error())
478           return rhs;
479     
480        /* If the types are identical, the assignment can trivially proceed.
481         */
482        if (rhs_type == lhs_type)
483           return rhs;
484     
485        /* If the array element types are the same and the size of the LHS is zero,
486         * the assignment is okay.
487         *
488         * Note: Whole-array assignments are not permitted in GLSL 1.10, but this
489         * is handled by ir_dereference::is_lvalue.
490         */
491        if (lhs_type->is_array() && rhs->type->is_array()
492            && (lhs_type->element_type() == rhs->type->element_type())
493            && (lhs_type->array_size() == 0)) {
494           return rhs;
495        }
496     
497        /* Check for implicit conversion in GLSL 1.20 */
498        if (apply_implicit_conversion(lhs_type, rhs, state)) {
499           rhs_type = rhs->type;
500           if (rhs_type == lhs_type)
501     	 return rhs;
502        }
503     
504        return NULL;
505     }

507     ir_rvalue *
508     do_assignment(exec_list *instructions, struct _mesa_glsl_parse_state *state,
509     	      ir_rvalue *lhs, ir_rvalue *rhs,
510     	      YYLTYPE lhs_loc)
511     {
512        void *ctx = state;
513        bool error_emitted = (lhs->type->is_error() || rhs->type->is_error());
514     
515        if (!error_emitted) {
516           if (!lhs->is_lvalue()) {
517     	 _mesa_glsl_error(& lhs_loc, state, "non-lvalue in assignment");
518     	 error_emitted = true;
519           }
520     
521           if (state->es_shader && lhs->type->is_array()) {
522     	 _mesa_glsl_error(&lhs_loc, state, "whole array assignment is not "
523     			  "allowed in GLSL ES 1.00.");
524     	 error_emitted = true;
525           }
526        }
527     
528        ir_rvalue *new_rhs = validate_assignment(state, lhs->type, rhs);
529        if (new_rhs == NULL) {
530           _mesa_glsl_error(& lhs_loc, state, "type mismatch");
531        } else {
532           rhs = new_rhs;
533     
534           /* If the LHS array was not declared with a size, it takes it size from
535            * the RHS.  If the LHS is an l-value and a whole array, it must be a
536            * dereference of a variable.  Any other case would require that the LHS
537            * is either not an l-value or not a whole array.
538            */
539           if (lhs->type->array_size() == 0) {
540     	 ir_dereference *const d = lhs->as_dereference();
541     
542     	 assert(d != NULL);
543     
544     	 ir_variable *const var = d->variable_referenced();
545     
546     	 assert(var != NULL);
547     
548     	 if (var->max_array_access >= unsigned(rhs->type->array_size())) {
549     	    /* FINISHME: This should actually log the location of the RHS. */
550     	    _mesa_glsl_error(& lhs_loc, state, "array size must be > %u due to "
551     			     "previous access",
552     			     var->max_array_access);
553     	 }
554     
555     	 var->type = glsl_type::get_array_instance(lhs->type->element_type(),
556     						   rhs->type->array_size());
557     	 d->type = var->type;
558           }
559        }
560     
561        /* Most callers of do_assignment (assign, add_assign, pre_inc/dec,
562         * but not post_inc) need the converted assigned value as an rvalue
563         * to handle things like:
564         *
565         * i = j += 1;
566         *
567         * So we always just store the computed value being assigned to a
568         * temporary and return a deref of that temporary.  If the rvalue
569         * ends up not being used, the temp will get copy-propagated out.
570         */
571        ir_variable *var = new(ctx) ir_variable(rhs->type, "assignment_tmp",
572     					   ir_var_temporary);
573        ir_dereference_variable *deref_var = new(ctx) ir_dereference_variable(var);
574        instructions->push_tail(var);
575        instructions->push_tail(new(ctx) ir_assignment(deref_var,
576     						  rhs,
577     						  NULL));
578        deref_var = new(ctx) ir_dereference_variable(var);
579     
580        if (!error_emitted)
581           instructions->push_tail(new(ctx) ir_assignment(lhs, deref_var, NULL));
582     
583        return new(ctx) ir_dereference_variable(var);
584     }

586     static ir_rvalue *
587     get_lvalue_copy(exec_list *instructions, ir_rvalue *lvalue)
588     {
589        void *ctx = talloc_parent(lvalue);
590        ir_variable *var;
591     
592        var = new(ctx) ir_variable(lvalue->type, "_post_incdec_tmp",
593     			      ir_var_temporary);
594        instructions->push_tail(var);
595        var->mode = ir_var_auto;
596     
597        instructions->push_tail(new(ctx) ir_assignment(new(ctx) ir_dereference_variable(var),
598     						  lvalue, NULL));
599     
600        /* Once we've created this temporary, mark it read only so it's no
601         * longer considered an lvalue.
602         */
603        var->read_only = true;
604     
605        return new(ctx) ir_dereference_variable(var);
606     }

619     
620     ir_rvalue *
621     ast_expression::hir(exec_list *instructions,
622     		    struct _mesa_glsl_parse_state *state)
623     {
624        void *ctx = state;
625        static const int operations[AST_NUM_OPERATORS] = {
626           -1,               /* ast_assign doesn't convert to ir_expression. */
627           -1,               /* ast_plus doesn't convert to ir_expression. */
628           ir_unop_neg,
629           ir_binop_add,
630           ir_binop_sub,
631           ir_binop_mul,
632           ir_binop_div,
633           ir_binop_mod,
634           ir_binop_lshift,
635           ir_binop_rshift,
636           ir_binop_less,
637           ir_binop_greater,
638           ir_binop_lequal,
639           ir_binop_gequal,
640           ir_binop_all_equal,
641           ir_binop_any_nequal,
642           ir_binop_bit_and,
643           ir_binop_bit_xor,
644           ir_binop_bit_or,
645           ir_unop_bit_not,
646           ir_binop_logic_and,
647           ir_binop_logic_xor,
648           ir_binop_logic_or,
649           ir_unop_logic_not,
650     
651           /* Note: The following block of expression types actually convert
652            * to multiple IR instructions.
653            */
654           ir_binop_mul,     /* ast_mul_assign */
655           ir_binop_div,     /* ast_div_assign */
656           ir_binop_mod,     /* ast_mod_assign */
657           ir_binop_add,     /* ast_add_assign */
658           ir_binop_sub,     /* ast_sub_assign */
659           ir_binop_lshift,  /* ast_ls_assign */
660           ir_binop_rshift,  /* ast_rs_assign */
661           ir_binop_bit_and, /* ast_and_assign */
662           ir_binop_bit_xor, /* ast_xor_assign */
663           ir_binop_bit_or,  /* ast_or_assign */
664     
665           -1,               /* ast_conditional doesn't convert to ir_expression. */
666           ir_binop_add,     /* ast_pre_inc. */
667           ir_binop_sub,     /* ast_pre_dec. */
668           ir_binop_add,     /* ast_post_inc. */
669           ir_binop_sub,     /* ast_post_dec. */
670           -1,               /* ast_field_selection doesn't conv to ir_expression. */
671           -1,               /* ast_array_index doesn't convert to ir_expression. */
672           -1,               /* ast_function_call doesn't conv to ir_expression. */
673           -1,               /* ast_identifier doesn't convert to ir_expression. */
674           -1,               /* ast_int_constant doesn't convert to ir_expression. */
675           -1,               /* ast_uint_constant doesn't conv to ir_expression. */
676           -1,               /* ast_float_constant doesn't conv to ir_expression. */
677           -1,               /* ast_bool_constant doesn't conv to ir_expression. */
678           -1,               /* ast_sequence doesn't convert to ir_expression. */
679        };
680        ir_rvalue *result = NULL;
681        ir_rvalue *op[3];
682        const struct glsl_type *type = glsl_type::error_type;
683        bool error_emitted = false;
684        YYLTYPE loc;
685     
686        loc = this->get_location();
687     
688        switch (this->oper) {
689        case ast_assign: {
690           op[0] = this->subexpressions[0]->hir(instructions, state);
691           op[1] = this->subexpressions[1]->hir(instructions, state);
692     
693           result = do_assignment(instructions, state, op[0], op[1],
694     			     this->subexpressions[0]->get_location());
695           error_emitted = result->type->is_error();
696           type = result->type;
697           break;
698        }
699     
700        case ast_plus:
701           op[0] = this->subexpressions[0]->hir(instructions, state);
702     
703           type = unary_arithmetic_result_type(op[0]->type, state, & loc);
704     
705           error_emitted = type->is_error();
706     
707           result = op[0];
708           break;
709     
710        case ast_neg:
711           op[0] = this->subexpressions[0]->hir(instructions, state);
712     
713           type = unary_arithmetic_result_type(op[0]->type, state, & loc);
714     
715           error_emitted = type->is_error();
716     
717           result = new(ctx) ir_expression(operations[this->oper], type,
718     				      op[0], NULL);
719           break;
720     
721        case ast_add:
722        case ast_sub:
723        case ast_mul:
724        case ast_div:
725           op[0] = this->subexpressions[0]->hir(instructions, state);
726           op[1] = this->subexpressions[1]->hir(instructions, state);
727     
728           type = arithmetic_result_type(op[0], op[1],
729     				    (this->oper == ast_mul),
730     				    state, & loc);
731           error_emitted = type->is_error();
732     
733           result = new(ctx) ir_expression(operations[this->oper], type,
734     				      op[0], op[1]);
735           break;
736     
737        case ast_mod:
738           op[0] = this->subexpressions[0]->hir(instructions, state);
739           op[1] = this->subexpressions[1]->hir(instructions, state);
740     
741           type = modulus_result_type(op[0]->type, op[1]->type, state, & loc);
742     
743           assert(operations[this->oper] == ir_binop_mod);
744     
745           result = new(ctx) ir_expression(operations[this->oper], type,
746     				      op[0], op[1]);
747           error_emitted = type->is_error();
748           break;
749     
750        case ast_lshift:
751        case ast_rshift:
752           _mesa_glsl_error(& loc, state, "FINISHME: implement bit-shift operators");
753           error_emitted = true;
754           break;
755     
756        case ast_less:
757        case ast_greater:
758        case ast_lequal:
759        case ast_gequal:
760           op[0] = this->subexpressions[0]->hir(instructions, state);
761           op[1] = this->subexpressions[1]->hir(instructions, state);
762     
763           type = relational_result_type(op[0], op[1], state, & loc);
764     
765           /* The relational operators must either generate an error or result
766            * in a scalar boolean.  See page 57 of the GLSL 1.50 spec.
767            */
768           assert(type->is_error()
769     	     || ((type->base_type == GLSL_TYPE_BOOL)
770     		 && type->is_scalar()));
771     
772           result = new(ctx) ir_expression(operations[this->oper], type,
773     				      op[0], op[1]);
774           error_emitted = type->is_error();
775           break;
776     
777        case ast_nequal:
778        case ast_equal:
779           op[0] = this->subexpressions[0]->hir(instructions, state);
780           op[1] = this->subexpressions[1]->hir(instructions, state);
781     
782           /* From page 58 (page 64 of the PDF) of the GLSL 1.50 spec:
783            *
784            *    "The equality operators equal (==), and not equal (!=)
785            *    operate on all types. They result in a scalar Boolean. If
786            *    the operand types do not match, then there must be a
787            *    conversion from Section 4.1.10 "Implicit Conversions"
788            *    applied to one operand that can make them match, in which
789            *    case this conversion is done."
790            */
791           if ((!apply_implicit_conversion(op[0]->type, op[1], state)
792     	   && !apply_implicit_conversion(op[1]->type, op[0], state))
793     	  || (op[0]->type != op[1]->type)) {
794     	 _mesa_glsl_error(& loc, state, "operands of `%s' must have the same "
795     			  "type", (this->oper == ast_equal) ? "==" : "!=");
796     	 error_emitted = true;
797           } else if ((state->language_version <= 110)
798     		 && (op[0]->type->is_array() || op[1]->type->is_array())) {
799     	 _mesa_glsl_error(& loc, state, "array comparisons forbidden in "
800     			  "GLSL 1.10");
801     	 error_emitted = true;
802           }
803     
804           result = new(ctx) ir_expression(operations[this->oper], glsl_type::bool_type,
805     				      op[0], op[1]);
806           type = glsl_type::bool_type;
807     
808           assert(result->type == glsl_type::bool_type);
809           break;
810     
811        case ast_bit_and:
812        case ast_bit_xor:
813        case ast_bit_or:
814           op[0] = this->subexpressions[0]->hir(instructions, state);
815           op[1] = this->subexpressions[1]->hir(instructions, state);
816     
817           if (state->language_version < 130) {
818     	 _mesa_glsl_error(&loc, state, "bit-wise operations require GLSL 1.30");
819     	 error_emitted = true;
820           }
821     
822           if (!op[0]->type->is_integer()) {
823     	 _mesa_glsl_error(&loc, state, "LHS of `%s' must be an integer",
824     			  operator_string(this->oper));
825     	 error_emitted = true;
826           }
827     
828           if (!op[1]->type->is_integer()) {
829     	 _mesa_glsl_error(&loc, state, "RHS of `%s' must be an integer",
830     			  operator_string(this->oper));
831     	 error_emitted = true;
832           }
833     
834           if (op[0]->type->base_type != op[1]->type->base_type) {
835     	 _mesa_glsl_error(&loc, state, "operands of `%s' must have the same "
836     			  "base type", operator_string(this->oper));
837     	 error_emitted = true;
838           }
839     
840           if (op[0]->type->is_vector() && op[1]->type->is_vector()
841     	  && op[0]->type->vector_elements != op[1]->type->vector_elements) {
842     	 _mesa_glsl_error(&loc, state, "operands of `%s' cannot be vectors of "
843     			  "different sizes", operator_string(this->oper));
844     	 error_emitted = true;
845           }
846     
847           type = op[0]->type->is_scalar() ? op[1]->type : op[0]->type;
848           result = new(ctx) ir_expression(operations[this->oper], type,
849     				      op[0], op[1]);
850           error_emitted = op[0]->type->is_error() || op[1]->type->is_error();
851           break;
852     
853        case ast_bit_not:
854           op[0] = this->subexpressions[0]->hir(instructions, state);
855     
856           if (state->language_version < 130) {
857     	 _mesa_glsl_error(&loc, state, "bit-wise operations require GLSL 1.30");
858     	 error_emitted = true;
859           }
860     
861           if (!op[0]->type->is_integer()) {
862     	 _mesa_glsl_error(&loc, state, "operand of `~' must be an integer");
863     	 error_emitted = true;
864           }
865     
866           type = op[0]->type;
867           result = new(ctx) ir_expression(ir_unop_bit_not, type, op[0], NULL);
868           break;
869     
870        case ast_logic_and: {
871           op[0] = this->subexpressions[0]->hir(instructions, state);
872     
873           if (!op[0]->type->is_boolean() || !op[0]->type->is_scalar()) {
874     	 YYLTYPE loc = this->subexpressions[0]->get_location();
875     
876     	 _mesa_glsl_error(& loc, state, "LHS of `%s' must be scalar boolean",
877     			  operator_string(this->oper));
878     	 error_emitted = true;
879           }
880     
881           ir_constant *op0_const = op[0]->constant_expression_value();
882           if (op0_const) {
883     	 if (op0_const->value.b[0]) {
884     	    op[1] = this->subexpressions[1]->hir(instructions, state);
885     
886     	    if (!op[1]->type->is_boolean() || !op[1]->type->is_scalar()) {
887     	       YYLTYPE loc = this->subexpressions[1]->get_location();
888     
889     	       _mesa_glsl_error(& loc, state,
890     				"RHS of `%s' must be scalar boolean",
891     				operator_string(this->oper));
892     	       error_emitted = true;
893     	    }
894     	    result = op[1];
895     	 } else {
896     	    result = op0_const;
897     	 }
898     	 type = glsl_type::bool_type;
899           } else {
900     	 ir_variable *const tmp = new(ctx) ir_variable(glsl_type::bool_type,
901     						       "and_tmp",
902     						       ir_var_temporary);
903     	 instructions->push_tail(tmp);
904     
905     	 ir_if *const stmt = new(ctx) ir_if(op[0]);
906     	 instructions->push_tail(stmt);
907     
908     	 op[1] = this->subexpressions[1]->hir(&stmt->then_instructions, state);
909     
910     	 if (!op[1]->type->is_boolean() || !op[1]->type->is_scalar()) {
911     	    YYLTYPE loc = this->subexpressions[1]->get_location();
912     
913     	    _mesa_glsl_error(& loc, state,
914     			     "RHS of `%s' must be scalar boolean",
915     			     operator_string(this->oper));
916     	    error_emitted = true;
917     	 }
918     
919     	 ir_dereference *const then_deref = new(ctx) ir_dereference_variable(tmp);
920     	 ir_assignment *const then_assign =
921     	    new(ctx) ir_assignment(then_deref, op[1], NULL);
922     	 stmt->then_instructions.push_tail(then_assign);
923     
924     	 ir_dereference *const else_deref = new(ctx) ir_dereference_variable(tmp);
925     	 ir_assignment *const else_assign =
926     	    new(ctx) ir_assignment(else_deref, new(ctx) ir_constant(false), NULL);
927     	 stmt->else_instructions.push_tail(else_assign);
928     
929     	 result = new(ctx) ir_dereference_variable(tmp);
930     	 type = tmp->type;
931           }
932           break;
933        }
934     
935        case ast_logic_or: {
936           op[0] = this->subexpressions[0]->hir(instructions, state);
937     
938           if (!op[0]->type->is_boolean() || !op[0]->type->is_scalar()) {
939     	 YYLTYPE loc = this->subexpressions[0]->get_location();
940     
941     	 _mesa_glsl_error(& loc, state, "LHS of `%s' must be scalar boolean",
942     			  operator_string(this->oper));
943     	 error_emitted = true;
944           }
945     
946           ir_constant *op0_const = op[0]->constant_expression_value();
947           if (op0_const) {
948     	 if (op0_const->value.b[0]) {
949     	    result = op0_const;
950     	 } else {
951     	    op[1] = this->subexpressions[1]->hir(instructions, state);
952     
953     	    if (!op[1]->type->is_boolean() || !op[1]->type->is_scalar()) {
954     	       YYLTYPE loc = this->subexpressions[1]->get_location();
955     
956     	       _mesa_glsl_error(& loc, state,
957     				"RHS of `%s' must be scalar boolean",
958     				operator_string(this->oper));
959     	       error_emitted = true;
960     	    }
961     	    result = op[1];
962     	 }
963     	 type = glsl_type::bool_type;
964           } else {
965     	 ir_variable *const tmp = new(ctx) ir_variable(glsl_type::bool_type,
966     						       "or_tmp",
967     						       ir_var_temporary);
968     	 instructions->push_tail(tmp);
969     
970     	 ir_if *const stmt = new(ctx) ir_if(op[0]);
971     	 instructions->push_tail(stmt);
972     
973     	 op[1] = this->subexpressions[1]->hir(&stmt->else_instructions, state);
974     
975     	 if (!op[1]->type->is_boolean() || !op[1]->type->is_scalar()) {
976     	    YYLTYPE loc = this->subexpressions[1]->get_location();
977     
978     	    _mesa_glsl_error(& loc, state, "RHS of `%s' must be scalar boolean",
979     			     operator_string(this->oper));
980     	    error_emitted = true;
981     	 }
982     
983     	 ir_dereference *const then_deref = new(ctx) ir_dereference_variable(tmp);
984     	 ir_assignment *const then_assign =
985     	    new(ctx) ir_assignment(then_deref, new(ctx) ir_constant(true), NULL);
986     	 stmt->then_instructions.push_tail(then_assign);
987     
988     	 ir_dereference *const else_deref = new(ctx) ir_dereference_variable(tmp);
989     	 ir_assignment *const else_assign =
990     	    new(ctx) ir_assignment(else_deref, op[1], NULL);
991     	 stmt->else_instructions.push_tail(else_assign);
992     
993     	 result = new(ctx) ir_dereference_variable(tmp);
994     	 type = tmp->type;
995           }
996           break;
997        }
998     
999        case ast_logic_xor:
1000           op[0] = this->subexpressions[0]->hir(instructions, state);
1001           op[1] = this->subexpressions[1]->hir(instructions, state);
1002     
1003     
1004           result = new(ctx) ir_expression(operations[this->oper], glsl_type::bool_type,
1005     				      op[0], op[1]);
1006           type = glsl_type::bool_type;
1007           break;
1008     
1009        case ast_logic_not:
1010           op[0] = this->subexpressions[0]->hir(instructions, state);
1011     
1012           if (!op[0]->type->is_boolean() || !op[0]->type->is_scalar()) {
1013     	 YYLTYPE loc = this->subexpressions[0]->get_location();
1014     
1015     	 _mesa_glsl_error(& loc, state,
1016     			  "operand of `!' must be scalar boolean");
1017     	 error_emitted = true;
1018           }
1019     
1020           result = new(ctx) ir_expression(operations[this->oper], glsl_type::bool_type,
1021     				      op[0], NULL);
1022           type = glsl_type::bool_type;
1023           break;
1024     
1025        case ast_mul_assign:
1026        case ast_div_assign:
1027        case ast_add_assign:
1028        case ast_sub_assign: {
1029           op[0] = this->subexpressions[0]->hir(instructions, state);
1030           op[1] = this->subexpressions[1]->hir(instructions, state);
1031     
1032           type = arithmetic_result_type(op[0], op[1],
1033     				    (this->oper == ast_mul_assign),
1034     				    state, & loc);
1035     
1036           ir_rvalue *temp_rhs = new(ctx) ir_expression(operations[this->oper], type,
1037     						   op[0], op[1]);
1038     
1039           result = do_assignment(instructions, state,
1040     			     op[0]->clone(ctx, NULL), temp_rhs,
1041     			     this->subexpressions[0]->get_location());
1042           type = result->type;
1043           error_emitted = (op[0]->type->is_error());
1044     
1045           /* GLSL 1.10 does not allow array assignment.  However, we don't have to
1046            * explicitly test for this because none of the binary expression
1047            * operators allow array operands either.
1048            */
1049     
1050           break;
1051        }
1052     
1053        case ast_mod_assign: {
1054           op[0] = this->subexpressions[0]->hir(instructions, state);
1055           op[1] = this->subexpressions[1]->hir(instructions, state);
1056     
1057           type = modulus_result_type(op[0]->type, op[1]->type, state, & loc);
1058     
1059           assert(operations[this->oper] == ir_binop_mod);
1060     
1061           ir_rvalue *temp_rhs;
1062           temp_rhs = new(ctx) ir_expression(operations[this->oper], type,
1063     					op[0], op[1]);
1064     
1065           result = do_assignment(instructions, state,
1066     			     op[0]->clone(ctx, NULL), temp_rhs,
1067     			     this->subexpressions[0]->get_location());
1068           type = result->type;
1069           error_emitted = type->is_error();
1070           break;
1071        }
1072     
1073        case ast_ls_assign:
1074        case ast_rs_assign:
1075           _mesa_glsl_error(& loc, state,
1076     		       "FINISHME: implement bit-shift assignment operators");
1077           error_emitted = true;
1078           break;
1079     
1080        case ast_and_assign:
1081        case ast_xor_assign:
1082        case ast_or_assign:
1083           _mesa_glsl_error(& loc, state,
1084     		       "FINISHME: implement logic assignment operators");
1085           error_emitted = true;
1086           break;
1087     
1088        case ast_conditional: {
1089           op[0] = this->subexpressions[0]->hir(instructions, state);
1090     
1091           /* From page 59 (page 65 of the PDF) of the GLSL 1.50 spec:
1092            *
1093            *    "The ternary selection operator (?:). It operates on three
1094            *    expressions (exp1 ? exp2 : exp3). This operator evaluates the
1095            *    first expression, which must result in a scalar Boolean."
1096            */
1097           if (!op[0]->type->is_boolean() || !op[0]->type->is_scalar()) {
1098     	 YYLTYPE loc = this->subexpressions[0]->get_location();
1099     
1100     	 _mesa_glsl_error(& loc, state, "?: condition must be scalar boolean");
1101     	 error_emitted = true;
1102           }
1103     
1104           /* The :? operator is implemented by generating an anonymous temporary
1105            * followed by an if-statement.  The last instruction in each branch of
1106            * the if-statement assigns a value to the anonymous temporary.  This
1107            * temporary is the r-value of the expression.
1108            */
1109           exec_list then_instructions;
1110           exec_list else_instructions;
1111     
1112           op[1] = this->subexpressions[1]->hir(&then_instructions, state);
1113           op[2] = this->subexpressions[2]->hir(&else_instructions, state);
1114     
1115           /* From page 59 (page 65 of the PDF) of the GLSL 1.50 spec:
1116            *
1117            *     "The second and third expressions can be any type, as
1118            *     long their types match, or there is a conversion in
1119            *     Section 4.1.10 "Implicit Conversions" that can be applied
1120            *     to one of the expressions to make their types match. This
1121            *     resulting matching type is the type of the entire
1122            *     expression."
1123            */
1124           if ((!apply_implicit_conversion(op[1]->type, op[2], state)
1125     	   && !apply_implicit_conversion(op[2]->type, op[1], state))
1126     	  || (op[1]->type != op[2]->type)) {
1127     	 YYLTYPE loc = this->subexpressions[1]->get_location();
1128     
1129     	 _mesa_glsl_error(& loc, state, "Second and third operands of ?: "
1130     			  "operator must have matching types.");
1131     	 error_emitted = true;
1132     	 type = glsl_type::error_type;
1133           } else {
1134     	 type = op[1]->type;
1135           }
1136     
1137           /* From page 33 (page 39 of the PDF) of the GLSL 1.10 spec:
1138            *
1139            *    "The second and third expressions must be the same type, but can
1140            *    be of any type other than an array."
1141            */
1142           if ((state->language_version <= 110) && type->is_array()) {
1143     	 _mesa_glsl_error(& loc, state, "Second and third operands of ?: "
1144     			  "operator must not be arrays.");
1145     	 error_emitted = true;
1146           }
1147     
1148           ir_constant *cond_val = op[0]->constant_expression_value();
1149           ir_constant *then_val = op[1]->constant_expression_value();
1150           ir_constant *else_val = op[2]->constant_expression_value();
1151     
1152           if (then_instructions.is_empty()
1153     	  && else_instructions.is_empty()
1154     	  && (cond_val != NULL) && (then_val != NULL) && (else_val != NULL)) {
1155     	 result = (cond_val->value.b[0]) ? then_val : else_val;
1156           } else {
1157     	 ir_variable *const tmp =
1158     	    new(ctx) ir_variable(type, "conditional_tmp", ir_var_temporary);
1159     	 instructions->push_tail(tmp);
1160     
1161     	 ir_if *const stmt = new(ctx) ir_if(op[0]);
1162     	 instructions->push_tail(stmt);
1163     
1164     	 then_instructions.move_nodes_to(& stmt->then_instructions);
1165     	 ir_dereference *const then_deref =
1166     	    new(ctx) ir_dereference_variable(tmp);
1167     	 ir_assignment *const then_assign =
1168     	    new(ctx) ir_assignment(then_deref, op[1], NULL);
1169     	 stmt->then_instructions.push_tail(then_assign);
1170     
1171     	 else_instructions.move_nodes_to(& stmt->else_instructions);
1172     	 ir_dereference *const else_deref =
1173     	    new(ctx) ir_dereference_variable(tmp);
1174     	 ir_assignment *const else_assign =
1175     	    new(ctx) ir_assignment(else_deref, op[2], NULL);
1176     	 stmt->else_instructions.push_tail(else_assign);
1177     
1178     	 result = new(ctx) ir_dereference_variable(tmp);
1179           }
1180           break;
1181        }
1182     
1183        case ast_pre_inc:
1184        case ast_pre_dec: {
1185           op[0] = this->subexpressions[0]->hir(instructions, state);
1186           if (op[0]->type->base_type == GLSL_TYPE_FLOAT)
1187     	 op[1] = new(ctx) ir_constant(1.0f);
1188           else
1189     	 op[1] = new(ctx) ir_constant(1);
1190     
1191           type = arithmetic_result_type(op[0], op[1], false, state, & loc);
1192     
1193           ir_rvalue *temp_rhs;
1194           temp_rhs = new(ctx) ir_expression(operations[this->oper], type,
1195     					op[0], op[1]);
1196     
1197           result = do_assignment(instructions, state,
1198     			     op[0]->clone(ctx, NULL), temp_rhs,
1199     			     this->subexpressions[0]->get_location());
1200           type = result->type;
1201           error_emitted = op[0]->type->is_error();
1202           break;
1203        }
1204     
1205        case ast_post_inc:
1206        case ast_post_dec: {
1207           op[0] = this->subexpressions[0]->hir(instructions, state);
1208           if (op[0]->type->base_type == GLSL_TYPE_FLOAT)
1209     	 op[1] = new(ctx) ir_constant(1.0f);
1210           else
1211     	 op[1] = new(ctx) ir_constant(1);
1212     
1213           error_emitted = op[0]->type->is_error() || op[1]->type->is_error();
1214     
1215           type = arithmetic_result_type(op[0], op[1], false, state, & loc);
1216     
1217           ir_rvalue *temp_rhs;
1218           temp_rhs = new(ctx) ir_expression(operations[this->oper], type,
1219     					op[0], op[1]);
1220     
1221           /* Get a temporary of a copy of the lvalue before it's modified.
1222            * This may get thrown away later.
1223            */
1224           result = get_lvalue_copy(instructions, op[0]->clone(ctx, NULL));
1225     
1226           (void)do_assignment(instructions, state,
1227     			  op[0]->clone(ctx, NULL), temp_rhs,
1228     			  this->subexpressions[0]->get_location());
1229     
1230           type = result->type;
1231           error_emitted = op[0]->type->is_error();
1232           break;
1233        }
1234     
1235        case ast_field_selection:
1236           result = _mesa_ast_field_selection_to_hir(this, instructions, state);
1237           type = result->type;
1238           break;
1239     
1240        case ast_array_index: {
1241           YYLTYPE index_loc = subexpressions[1]->get_location();
1242     
1243           op[0] = subexpressions[0]->hir(instructions, state);
1244           op[1] = subexpressions[1]->hir(instructions, state);
1245     
1246           error_emitted = op[0]->type->is_error() || op[1]->type->is_error();
1247     
1248           ir_rvalue *const array = op[0];
1249     
1250           result = new(ctx) ir_dereference_array(op[0], op[1]);
1251     
1252           /* Do not use op[0] after this point.  Use array.
1253            */
1254           op[0] = NULL;
1255     
1256     
1257           if (error_emitted)
1258     	 break;
1259     
1260           if (!array->type->is_array()
1261     	  && !array->type->is_matrix()
1262     	  && !array->type->is_vector()) {
1263     	 _mesa_glsl_error(& index_loc, state,
1264     			  "cannot dereference non-array / non-matrix / "
1265     			  "non-vector");
1266     	 error_emitted = true;
1267           }
1268     
1269           if (!op[1]->type->is_integer()) {
1270     	 _mesa_glsl_error(& index_loc, state,
1271     			  "array index must be integer type");
1272     	 error_emitted = true;
1273           } else if (!op[1]->type->is_scalar()) {
1274     	 _mesa_glsl_error(& index_loc, state,
1275     			  "array index must be scalar");
1276     	 error_emitted = true;
1277           }
1278     
1279           /* If the array index is a constant expression and the array has a
1280            * declared size, ensure that the access is in-bounds.  If the array
1281            * index is not a constant expression, ensure that the array has a
1282            * declared size.
1283            */
1284           ir_constant *const const_index = op[1]->constant_expression_value();
1285           if (const_index != NULL) {
1286     	 const int idx = const_index->value.i[0];
1287     	 const char *type_name;
1288     	 unsigned bound = 0;
1289     
1290     	 if (array->type->is_matrix()) {
1291     	    type_name = "matrix";
1292     	 } else if (array->type->is_vector()) {
1293     	    type_name = "vector";
1294     	 } else {
1295     	    type_name = "array";
1296     	 }
1297     
1298     	 /* From page 24 (page 30 of the PDF) of the GLSL 1.50 spec:
1299     	  *
1300     	  *    "It is illegal to declare an array with a size, and then
1301     	  *    later (in the same shader) index the same array with an
1302     	  *    integral constant expression greater than or equal to the
1303     	  *    declared size. It is also illegal to index an array with a
1304     	  *    negative constant expression."
1305     	  */
1306     	 if (array->type->is_matrix()) {
1307     	    if (array->type->row_type()->vector_elements <= idx) {
1308     	       bound = array->type->row_type()->vector_elements;
1309     	    }
1310     	 } else if (array->type->is_vector()) {
1311     	    if (array->type->vector_elements <= idx) {
1312     	       bound = array->type->vector_elements;
1313     	    }
1314     	 } else {
1315     	    if ((array->type->array_size() > 0)
1316     		&& (array->type->array_size() <= idx)) {
1317     	       bound = array->type->array_size();
1318     	    }
1319     	 }
1320     
1321     	 if (bound > 0) {
1322     	    _mesa_glsl_error(& loc, state, "%s index must be < %u",
1323     			     type_name, bound);
1324     	    error_emitted = true;
1325     	 } else if (idx < 0) {
1326     	    _mesa_glsl_error(& loc, state, "%s index must be >= 0",
1327     			     type_name);
1328     	    error_emitted = true;
1329     	 }
1330     
1331     	 if (array->type->is_array()) {
1332     	    /* If the array is a variable dereference, it dereferences the
1333     	     * whole array, by definition.  Use this to get the variable.
1334     	     *
1335     	     * FINISHME: Should some methods for getting / setting / testing
1336     	     * FINISHME: array access limits be added to ir_dereference?
1337     	     */
1338     	    ir_variable *const v = array->whole_variable_referenced();
1339     	    if ((v != NULL) && (unsigned(idx) > v->max_array_access))
1340     	       v->max_array_access = idx;
1341     	 }
1342           } else if (array->type->array_size() == 0) {
1343     	 _mesa_glsl_error(&loc, state, "unsized array index must be constant");
1344           } else {
1345     	 if (array->type->is_array()) {
1346     	    /* whole_variable_referenced can return NULL if the array is a
1347     	     * member of a structure.  In this case it is safe to not update
1348     	     * the max_array_access field because it is never used for fields
1349     	     * of structures.
1350     	     */
1351     	    ir_variable *v = array->whole_variable_referenced();
1352     	    if (v != NULL)
1353     	       v->max_array_access = array->type->array_size();
1354     	 }
1355           }
1356     
1357           if (error_emitted)
1358     	 result->type = glsl_type::error_type;
1359     
1360           type = result->type;
1361           break;
1362        }
1363     
1364        case ast_function_call:
1365           /* Should *NEVER* get here.  ast_function_call should always be handled
1366            * by ast_function_expression::hir.
1367            */
1368           assert(0);
1369           break;
1370     
1371        case ast_identifier: {
1372           /* ast_identifier can appear several places in a full abstract syntax
1373            * tree.  This particular use must be at location specified in the grammar
1374            * as 'variable_identifier'.
1375            */
1376           ir_variable *var = 
1377     	 state->symbols->get_variable(this->primary_expression.identifier);
1378     
1379           result = new(ctx) ir_dereference_variable(var);
1380     
1381           if (var != NULL) {
1382     	 type = result->type;
1383           } else {
1384     	 _mesa_glsl_error(& loc, state, "`%s' undeclared",
1385     			  this->primary_expression.identifier);
1386     
1387     	 error_emitted = true;
1388           }
1389           break;
1390        }
1391     
1392        case ast_int_constant:
1393           type = glsl_type::int_type;
1394           result = new(ctx) ir_constant(this->primary_expression.int_constant);
1395           break;
1396     
1397        case ast_uint_constant:
1398           type = glsl_type::uint_type;
1399           result = new(ctx) ir_constant(this->primary_expression.uint_constant);
1400           break;
1401     
1402        case ast_float_constant:
1403           type = glsl_type::float_type;
1404           result = new(ctx) ir_constant(this->primary_expression.float_constant);
1405           break;
1406     
1407        case ast_bool_constant:
1408           type = glsl_type::bool_type;
1409           result = new(ctx) ir_constant(bool(this->primary_expression.bool_constant));
1410           break;
1411     
1412        case ast_sequence: {
1413           /* It should not be possible to generate a sequence in the AST without
1414            * any expressions in it.
1415            */
1416           assert(!this->expressions.is_empty());
1417     
1418           /* The r-value of a sequence is the last expression in the sequence.  If
1419            * the other expressions in the sequence do not have side-effects (and
1420            * therefore add instructions to the instruction list), they get dropped
1421            * on the floor.
1422            */
1423           foreach_list_typed (ast_node, ast, link, &this->expressions)
1424     	 result = ast->hir(instructions, state);
1425     
1426           type = result->type;
1427     
1428           /* Any errors should have already been emitted in the loop above.
1429            */
1430           error_emitted = true;
1431           break;
1432        }
1433        }
1434     
1435        if (type->is_error() && !error_emitted)
1436           _mesa_glsl_error(& loc, state, "type mismatch");
1437     
1438        return result;
1439     }

1483     static const glsl_type *
1484     process_array_type(YYLTYPE *loc, const glsl_type *base, ast_node *array_size,
1485     		   struct _mesa_glsl_parse_state *state)
1486     {
1487        unsigned length = 0;
1488     
1489        /* FINISHME: Reject delcarations of multidimensional arrays. */
1490     
1491        if (array_size != NULL) {
1492           exec_list dummy_instructions;
1493           ir_rvalue *const ir = array_size->hir(& dummy_instructions, state);
1494           YYLTYPE loc = array_size->get_location();
1495     
1496           /* FINISHME: Verify that the grammar forbids side-effects in array
1497            * FINISHME: sizes.   i.e., 'vec4 [x = 12] data'
1498            */
1499           assert(dummy_instructions.is_empty());
1500     
1501           if (ir != NULL) {
1502     	 if (!ir->type->is_integer()) {
1503     	    _mesa_glsl_error(& loc, state, "array size must be integer type");
1504     	 } else if (!ir->type->is_scalar()) {
1505     	    _mesa_glsl_error(& loc, state, "array size must be scalar type");
1506     	 } else {
1507     	    ir_constant *const size = ir->constant_expression_value();
1508     
1509     	    if (size == NULL) {
1510     	       _mesa_glsl_error(& loc, state, "array size must be a "
1511     				"constant valued expression");
1512     	    } else if (size->value.i[0] <= 0) {
1513     	       _mesa_glsl_error(& loc, state, "array size must be > 0");
1514     	    } else {
1515     	       assert(size->type == ir->type);
1516     	       length = size->value.u[0];
1517     	    }
1518     	 }
1519           }
1520        } else if (state->es_shader) {
1521           /* Section 10.17 of the GLSL ES 1.00 specification states that unsized
1522            * array declarations have been removed from the language.
1523            */
1524           _mesa_glsl_error(loc, state, "unsized array declarations are not "
1525     		       "allowed in GLSL ES 1.00.");
1526        }
1527     
1528        return glsl_type::get_array_instance(base, length);
1529     }

1550     static void
1551     apply_type_qualifier_to_variable(const struct ast_type_qualifier *qual,
1552     				 ir_variable *var,
1553     				 struct _mesa_glsl_parse_state *state,
1554     				 YYLTYPE *loc)
1555     {
1556        if (qual->invariant)
1557           var->invariant = 1;
1558     
1559        /* FINISHME: Mark 'in' variables at global scope as read-only. */
1560        if (qual->constant || qual->attribute || qual->uniform
1561            || (qual->varying && (state->target == fragment_shader)))
1562           var->read_only = 1;
1563     
1564        if (qual->centroid)
1565           var->centroid = 1;
1566     
1567        if (qual->attribute && state->target != vertex_shader) {
1568           var->type = glsl_type::error_type;
1569           _mesa_glsl_error(loc, state,
1570     		       "`attribute' variables may not be declared in the "
1571     		       "%s shader",
1572     		       _mesa_glsl_shader_target_name(state->target));
1573        }
1574     
1575        /* From page 25 (page 31 of the PDF) of the GLSL 1.10 spec:
1576         *
1577         *     "The varying qualifier can be used only with the data types
1578         *     float, vec2, vec3, vec4, mat2, mat3, and mat4, or arrays of
1579         *     these."
1580         */
1581        if (qual->varying) {
1582           const glsl_type *non_array_type;
1583     
1584           if (var->type && var->type->is_array())
1585     	 non_array_type = var->type->fields.array;
1586           else
1587     	 non_array_type = var->type;
1588     
1589           if (non_array_type && non_array_type->base_type != GLSL_TYPE_FLOAT) {
1590     	 var->type = glsl_type::error_type;
1591     	 _mesa_glsl_error(loc, state,
1592     			  "varying variables must be of base type float");
1593           }
1594        }
1595     
1596        /* If there is no qualifier that changes the mode of the variable, leave
1597         * the setting alone.
1598         */
1599        if (qual->in && qual->out)
1600           var->mode = ir_var_inout;
1601        else if (qual->attribute || qual->in
1602     	    || (qual->varying && (state->target == fragment_shader)))
1603           var->mode = ir_var_in;
1604        else if (qual->out || (qual->varying && (state->target == vertex_shader)))
1605           var->mode = ir_var_out;
1606        else if (qual->uniform)
1607           var->mode = ir_var_uniform;
1608     
1609        if (qual->flat)
1610           var->interpolation = ir_var_flat;
1611        else if (qual->noperspective)
1612           var->interpolation = ir_var_noperspective;
1613        else
1614           var->interpolation = ir_var_smooth;
1615     
1616        var->pixel_center_integer = qual->pixel_center_integer;
1617        var->origin_upper_left = qual->origin_upper_left;
1618        if ((qual->origin_upper_left || qual->pixel_center_integer)
1619            && (safe_strcmp(var->name, "gl_FragCoord") != 0)) {
1620           const char *const qual_string = (qual->origin_upper_left)
1621     	 ? "origin_upper_left" : "pixel_center_integer";
1622     
1623           _mesa_glsl_error(loc, state,
1624     		       "layout qualifier `%s' can only be applied to "
1625     		       "fragment shader input `gl_FragCoord'",
1626     		       qual_string);
1627        }
1628     
1629        if (var->type->is_array() && state->language_version != 110) {
1630           var->array_lvalue = true;
1631        }
1632     }

1634     
1635     ir_rvalue *
1636     ast_declarator_list::hir(exec_list *instructions,
1637     			 struct _mesa_glsl_parse_state *state)
1638     {
1639        void *ctx = state;
1640        const struct glsl_type *decl_type;
1641        const char *type_name = NULL;
1642        ir_rvalue *result = NULL;
1643        YYLTYPE loc = this->get_location();
1644     
1645        /* From page 46 (page 52 of the PDF) of the GLSL 1.50 spec:
1646         *
1647         *     "To ensure that a particular output variable is invariant, it is
1648         *     necessary to use the invariant qualifier. It can either be used to
1649         *     qualify a previously declared variable as being invariant
1650         *
1651         *         invariant gl_Position; // make existing gl_Position be invariant"
1652         *
1653         * In these cases the parser will set the 'invariant' flag in the declarator
1654         * list, and the type will be NULL.
1655         */
1656        if (this->invariant) {
1657           assert(this->type == NULL);
1658     
1659           if (state->current_function != NULL) {
1660     	 _mesa_glsl_error(& loc, state,
1661     			  "All uses of `invariant' keyword must be at global "
1662     			  "scope\n");
1663           }
1664     
1665           foreach_list_typed (ast_declaration, decl, link, &this->declarations) {
1666     	 assert(!decl->is_array);
1667     	 assert(decl->array_size == NULL);
1668     	 assert(decl->initializer == NULL);
1669     
1670     	 ir_variable *const earlier =
1671     	    state->symbols->get_variable(decl->identifier);
1672     	 if (earlier == NULL) {
1673     	    _mesa_glsl_error(& loc, state,
1674     			     "Undeclared variable `%s' cannot be marked "
1675     			     "invariant\n", decl->identifier);
1676     	 } else if ((state->target == vertex_shader)
1677     	       && (earlier->mode != ir_var_out)) {
1678     	    _mesa_glsl_error(& loc, state,
1679     			     "`%s' cannot be marked invariant, vertex shader "
1680     			     "outputs only\n", decl->identifier);
1681     	 } else if ((state->target == fragment_shader)
1682     	       && (earlier->mode != ir_var_in)) {
1683     	    _mesa_glsl_error(& loc, state,
1684     			     "`%s' cannot be marked invariant, fragment shader "
1685     			     "inputs only\n", decl->identifier);
1686     	 } else {
1687     	    earlier->invariant = true;
1688     	 }
1689           }
1690     
1691           /* Invariant redeclarations do not have r-values.
1692            */
1693           return NULL;
1694        }
1695     
1696        assert(this->type != NULL);
1697        assert(!this->invariant);
1698     
1699        /* The type specifier may contain a structure definition.  Process that
1700         * before any of the variable declarations.
1701         */
1702        (void) this->type->specifier->hir(instructions, state);
1703     
1704        decl_type = this->type->specifier->glsl_type(& type_name, state);
1705        if (this->declarations.is_empty()) {
1706           /* The only valid case where the declaration list can be empty is when
1707            * the declaration is setting the default precision of a built-in type
1708            * (e.g., 'precision highp vec4;').
1709            */
1710     
1711           if (decl_type != NULL) {
1712           } else {
1713     	    _mesa_glsl_error(& loc, state, "incomplete declaration");
1714           }
1715        }
1716     
1717        foreach_list_typed (ast_declaration, decl, link, &this->declarations) {
1718           const struct glsl_type *var_type;
1719           ir_variable *var;
1720     
1721           /* FINISHME: Emit a warning if a variable declaration shadows a
1722            * FINISHME: declaration at a higher scope.
1723            */
1724     
1725           if ((decl_type == NULL) || decl_type->is_void()) {
1726     	 if (type_name != NULL) {
1727     	    _mesa_glsl_error(& loc, state,
1728     			     "invalid type `%s' in declaration of `%s'",
1729     			     type_name, decl->identifier);
1730     	 } else {
1731     	    _mesa_glsl_error(& loc, state,
1732     			     "invalid type in declaration of `%s'",
1733     			     decl->identifier);
1734     	 }
1735     	 continue;
1736           }
1737     
1738           if (decl->is_array) {
1739     	 var_type = process_array_type(&loc, decl_type, decl->array_size,
1740     				       state);
1741           } else {
1742     	 var_type = decl_type;
1743           }
1744     
1745           var = new(ctx) ir_variable(var_type, decl->identifier, ir_var_auto);
1746     
1747           /* From page 22 (page 28 of the PDF) of the GLSL 1.10 specification;
1748            *
1749            *     "Global variables can only use the qualifiers const,
1750            *     attribute, uni form, or varying. Only one may be
1751            *     specified.
1752            *
1753            *     Local variables can only use the qualifier const."
1754            *
1755            * This is relaxed in GLSL 1.30.
1756            */
1757           if (state->language_version < 120) {
1758     	 if (this->type->qualifier.out) {
1759     	    _mesa_glsl_error(& loc, state,
1760     			     "`out' qualifier in declaration of `%s' "
1761     			     "only valid for function parameters in GLSL 1.10.",
1762     			     decl->identifier);
1763     	 }
1764     	 if (this->type->qualifier.in) {
1765     	    _mesa_glsl_error(& loc, state,
1766     			     "`in' qualifier in declaration of `%s' "
1767     			     "only valid for function parameters in GLSL 1.10.",
1768     			     decl->identifier);
1769     	 }
1770     	 /* FINISHME: Test for other invalid qualifiers. */
1771           }
1772     
1773           apply_type_qualifier_to_variable(& this->type->qualifier, var, state,
1774     				       & loc);
1775     
1776           if (this->type->qualifier.invariant) {
1777     	 if ((state->target == vertex_shader) && !(var->mode == ir_var_out ||
1778     						   var->mode == ir_var_inout)) {
1779     	    /* FINISHME: Note that this doesn't work for invariant on
1780     	     * a function signature outval
1781     	     */
1782     	    _mesa_glsl_error(& loc, state,
1783     			     "`%s' cannot be marked invariant, vertex shader "
1784     			     "outputs only\n", var->name);
1785     	 } else if ((state->target == fragment_shader) &&
1786     		    !(var->mode == ir_var_in || var->mode == ir_var_inout)) {
1787     	    /* FINISHME: Note that this doesn't work for invariant on
1788     	     * a function signature inval
1789     	     */
1790     	    _mesa_glsl_error(& loc, state,
1791     			     "`%s' cannot be marked invariant, fragment shader "
1792     			     "inputs only\n", var->name);
1793     	 }
1794           }
1795     
1796           if (state->current_function != NULL) {
1797     	 const char *mode = NULL;
1798     	 const char *extra = "";
1799     
1800     	 /* There is no need to check for 'inout' here because the parser will
1801     	  * only allow that in function parameter lists.
1802     	  */
1803     	 if (this->type->qualifier.attribute) {
1804     	    mode = "attribute";
1805     	 } else if (this->type->qualifier.uniform) {
1806     	    mode = "uniform";
1807     	 } else if (this->type->qualifier.varying) {
1808     	    mode = "varying";
1809     	 } else if (this->type->qualifier.in) {
1810     	    mode = "in";
1811     	    extra = " or in function parameter list";
1812     	 } else if (this->type->qualifier.out) {
1813     	    mode = "out";
1814     	    extra = " or in function parameter list";
1815     	 }
1816     
1817     	 if (mode) {
1818     	    _mesa_glsl_error(& loc, state,
1819     			     "%s variable `%s' must be declared at "
1820     			     "global scope%s",
1821     			     mode, var->name, extra);
1822     	 }
1823           } else if (var->mode == ir_var_in) {
1824     	 if (state->target == vertex_shader) {
1825     	    bool error_emitted = false;
1826     
1827     	    /* From page 31 (page 37 of the PDF) of the GLSL 1.50 spec:
1828     	     *
1829     	     *    "Vertex shader inputs can only be float, floating-point
1830     	     *    vectors, matrices, signed and unsigned integers and integer
1831     	     *    vectors. Vertex shader inputs can also form arrays of these
1832     	     *    types, but not structures."
1833     	     *
1834     	     * From page 31 (page 27 of the PDF) of the GLSL 1.30 spec:
1835     	     *
1836     	     *    "Vertex shader inputs can only be float, floating-point
1837     	     *    vectors, matrices, signed and unsigned integers and integer
1838     	     *    vectors. They cannot be arrays or structures."
1839     	     *
1840     	     * From page 23 (page 29 of the PDF) of the GLSL 1.20 spec:
1841     	     *
1842     	     *    "The attribute qualifier can be used only with float,
1843     	     *    floating-point vectors, and matrices. Attribute variables
1844     	     *    cannot be declared as arrays or structures."
1845     	     */
1846     	    const glsl_type *check_type = var->type->is_array()
1847     	       ? var->type->fields.array : var->type;
1848     
1849     	    switch (check_type->base_type) {
1850     	    case GLSL_TYPE_FLOAT:
1851     	       break;
1852     	    case GLSL_TYPE_UINT:
1853     	    case GLSL_TYPE_INT:
1854     	       if (state->language_version > 120)
1855     		  break;
1856     	       /* FALLTHROUGH */
1857     	    default:
1858     	       _mesa_glsl_error(& loc, state,
1859     				"vertex shader input / attribute cannot have "
1860     				"type %s`%s'",
1861     				var->type->is_array() ? "array of " : "",
1862     				check_type->name);
1863     	       error_emitted = true;
1864     	    }
1865     
1866     	    if (!error_emitted && (state->language_version <= 130)
1867     		&& var->type->is_array()) {
1868     	       _mesa_glsl_error(& loc, state,
1869     				"vertex shader input / attribute cannot have "
1870     				"array type");
1871     	       error_emitted = true;
1872     	    }
1873     	 }
1874           }
1875     
1876           /* Process the initializer and add its instructions to a temporary
1877            * list.  This list will be added to the instruction stream (below) after
1878            * the declaration is added.  This is done because in some cases (such as
1879            * redeclarations) the declaration may not actually be added to the
1880            * instruction stream.
1881            */
1882           exec_list initializer_instructions;
1883           if (decl->initializer != NULL) {
1884     	 YYLTYPE initializer_loc = decl->initializer->get_location();
1885     
1886     	 /* From page 24 (page 30 of the PDF) of the GLSL 1.10 spec:
1887     	  *
1888     	  *    "All uniform variables are read-only and are initialized either
1889     	  *    directly by an application via API commands, or indirectly by
1890     	  *    OpenGL."
1891     	  */
1892     	 if ((state->language_version <= 110)
1893     	     && (var->mode == ir_var_uniform)) {
1894     	    _mesa_glsl_error(& initializer_loc, state,
1895     			     "cannot initialize uniforms in GLSL 1.10");
1896     	 }
1897     
1898     	 if (var->type->is_sampler()) {
1899     	    _mesa_glsl_error(& initializer_loc, state,
1900     			     "cannot initialize samplers");
1901     	 }
1902     
1903     	 if ((var->mode == ir_var_in) && (state->current_function == NULL)) {
1904     	    _mesa_glsl_error(& initializer_loc, state,
1905     			     "cannot initialize %s shader input / %s",
1906     			     _mesa_glsl_shader_target_name(state->target),
1907     			     (state->target == vertex_shader)
1908     			     ? "attribute" : "varying");
1909     	 }
1910     
1911     	 ir_dereference *const lhs = new(ctx) ir_dereference_variable(var);
1912     	 ir_rvalue *rhs = decl->initializer->hir(&initializer_instructions,
1913     						 state);
1914     
1915     	 /* Calculate the constant value if this is a const or uniform
1916     	  * declaration.
1917     	  */
1918     	 if (this->type->qualifier.constant || this->type->qualifier.uniform) {
1919     	    ir_rvalue *new_rhs = validate_assignment(state, var->type, rhs);
1920     	    if (new_rhs != NULL) {
1921     	       rhs = new_rhs;
1922     
1923     	       ir_constant *constant_value = rhs->constant_expression_value();
1924     	       if (!constant_value) {
1925     		  _mesa_glsl_error(& initializer_loc, state,
1926     				   "initializer of %s variable `%s' must be a "
1927     				   "constant expression",
1928     				   (this->type->qualifier.constant)
1929     				   ? "const" : "uniform",
1930     				   decl->identifier);
1931     		  if (var->type->is_numeric()) {
1932     		     /* Reduce cascading errors. */
1933     		     var->constant_value = ir_constant::zero(ctx, var->type);
1934     		  }
1935     	       } else {
1936     		  rhs = constant_value;
1937     		  var->constant_value = constant_value;
1938     	       }
1939     	    } else {
1940     	       _mesa_glsl_error(&initializer_loc, state,
1941     			        "initializer of type %s cannot be assigned to "
1942     				"variable of type %s",
1943     				rhs->type->name, var->type->name);
1944     	       if (var->type->is_numeric()) {
1945     		  /* Reduce cascading errors. */
1946     		  var->constant_value = ir_constant::zero(ctx, var->type);
1947     	       }
1948     	    }
1949     	 }
1950     
1951     	 if (rhs && !rhs->type->is_error()) {
1952     	    bool temp = var->read_only;
1953     	    if (this->type->qualifier.constant)
1954     	       var->read_only = false;
1955     
1956     	    /* Never emit code to initialize a uniform.
1957     	     */
1958     	    if (!this->type->qualifier.uniform)
1959     	       result = do_assignment(&initializer_instructions, state,
1960     				      lhs, rhs,
1961     				      this->get_location());
1962     	    var->read_only = temp;
1963     	 }
1964           }
1965     
1966           /* From page 23 (page 29 of the PDF) of the GLSL 1.10 spec:
1967            *
1968            *     "It is an error to write to a const variable outside of
1969            *      its declaration, so they must be initialized when
1970            *      declared."
1971            */
1972           if (this->type->qualifier.constant && decl->initializer == NULL) {
1973     	 _mesa_glsl_error(& loc, state,
1974     			  "const declaration of `%s' must be initialized");
1975           }
1976     
1977           /* Check if this declaration is actually a re-declaration, either to
1978            * resize an array or add qualifiers to an existing variable.
1979            *
1980            * This is allowed for variables in the current scope, or when at
1981            * global scope (for built-ins in the implicit outer scope).
1982            */
1983           ir_variable *earlier = state->symbols->get_variable(decl->identifier);
1984           if (earlier != NULL && (state->current_function == NULL ||
1985     	  state->symbols->name_declared_this_scope(decl->identifier))) {
1986     
1987     	 /* From page 24 (page 30 of the PDF) of the GLSL 1.50 spec,
1988     	  *
1989     	  * "It is legal to declare an array without a size and then
1990     	  *  later re-declare the same name as an array of the same
1991     	  *  type and specify a size."
1992     	  */
1993     	 if ((earlier->type->array_size() == 0)
1994     	     && var->type->is_array()
1995     	     && (var->type->element_type() == earlier->type->element_type())) {
1996     	    /* FINISHME: This doesn't match the qualifiers on the two
1997     	     * FINISHME: declarations.  It's not 100% clear whether this is
1998     	     * FINISHME: required or not.
1999     	     */
2000     
2001     	    /* From page 54 (page 60 of the PDF) of the GLSL 1.20 spec:
2002     	     *
2003     	     *     "The size [of gl_TexCoord] can be at most
2004     	     *     gl_MaxTextureCoords."
2005     	     */
2006     	    const unsigned size = unsigned(var->type->array_size());
2007     	    if ((safe_strcmp("gl_TexCoord", var->name) == 0)
2008     		&& (size > state->Const.MaxTextureCoords)) {
2009     	       YYLTYPE loc = this->get_location();
2010     
2011     	       _mesa_glsl_error(& loc, state, "`gl_TexCoord' array size cannot "
2012     				"be larger than gl_MaxTextureCoords (%u)\n",
2013     				state->Const.MaxTextureCoords);
2014     	    } else if ((size > 0) && (size <= earlier->max_array_access)) {
2015     	       YYLTYPE loc = this->get_location();
2016     
2017     	       _mesa_glsl_error(& loc, state, "array size must be > %u due to "
2018     				"previous access",
2019     				earlier->max_array_access);
2020     	    }
2021     
2022     	    earlier->type = var->type;
2023     	    delete var;
2024     	    var = NULL;
2025     	 } else if (state->extensions->ARB_fragment_coord_conventions
2026     		    && safe_strcmp(var->name, "gl_FragCoord") == 0
2027     		    && earlier->type == var->type
2028     		    && earlier->mode == var->mode) {
2029     	    /* Allow redeclaration of gl_FragCoord for ARB_fcc layout
2030     	     * qualifiers.
2031     	     */
2032     	    earlier->origin_upper_left = var->origin_upper_left;
2033     	    earlier->pixel_center_integer = var->pixel_center_integer;
2034     	 } else {
2035     	    YYLTYPE loc = this->get_location();
2036     	    _mesa_glsl_error(&loc, state, "`%s' redeclared", decl->identifier);
2037     	 }
2038     
2039     	 continue;
2040           }
2041     
2042           /* By now, we know it's a new variable declaration (we didn't hit the
2043            * above "continue").
2044            *
2045            * From page 15 (page 21 of the PDF) of the GLSL 1.10 spec,
2046            *
2047            *   "Identifiers starting with "gl_" are reserved for use by
2048            *   OpenGL, and may not be declared in a shader as either a
2049            *   variable or a function."
2050            */
2051           if (strncmp(decl->identifier, "gl_", 3) == 0)
2052     	 _mesa_glsl_error(& loc, state,
2053     			  "identifier `%s' uses reserved `gl_' prefix",
2054     			  decl->identifier);
2055     
2056           /* Add the variable to the symbol table.  Note that the initializer's
2057            * IR was already processed earlier (though it hasn't been emitted yet),
2058            * without the variable in scope.
2059            *
2060            * This differs from most C-like languages, but it follows the GLSL
2061            * specification.  From page 28 (page 34 of the PDF) of the GLSL 1.50
2062            * spec:
2063            *
2064            *     "Within a declaration, the scope of a name starts immediately
2065            *     after the initializer if present or immediately after the name
2066            *     being declared if not."
2067            */
2068           if (!state->symbols->add_variable(var->name, var)) {
2069     	 YYLTYPE loc = this->get_location();
2070     	 _mesa_glsl_error(&loc, state, "name `%s' already taken in the "
2071     			  "current scope", decl->identifier);
2072     	 continue;
2073           }
2074     
2075           /* Push the variable declaration to the top.  It means that all
2076            * the variable declarations will appear in a funny
2077            * last-to-first order, but otherwise we run into trouble if a
2078            * function is prototyped, a global var is decled, then the
2079            * function is defined with usage of the global var.  See
2080            * glslparsertest's CorrectModule.frag.
2081            */
2082           instructions->push_head(var);
2083           instructions->append_list(&initializer_instructions);
2084        }
2085     
2086     
2087        /* Generally, variable declarations do not have r-values.  However,
2088         * one is used for the declaration in
2089         *
2090         * while (bool b = some_condition()) {
2091         *   ...
2092         * }
2093         *
2094         * so we return the rvalue from the last seen declaration here.
2095         */
2096        return result;
2097     }

2100     ir_rvalue *
2101     ast_parameter_declarator::hir(exec_list *instructions,
2102     			      struct _mesa_glsl_parse_state *state)
2103     {
2104        void *ctx = state;
2105        const struct glsl_type *type;
2106        const char *name = NULL;
2107        YYLTYPE loc = this->get_location();
2108     
2109        type = this->type->specifier->glsl_type(& name, state);
2110     
2111        if (type == NULL) {
2112           if (name != NULL) {
2113     	 _mesa_glsl_error(& loc, state,
2114     			  "invalid type `%s' in declaration of `%s'",
2115     			  name, this->identifier);
2116           } else {
2117     	 _mesa_glsl_error(& loc, state,
2118     			  "invalid type in declaration of `%s'",
2119     			  this->identifier);
2120           }
2121     
2122           type = glsl_type::error_type;
2123        }
2124     
2125        /* From page 62 (page 68 of the PDF) of the GLSL 1.50 spec:
2126         *
2127         *    "Functions that accept no input arguments need not use void in the
2128         *    argument list because prototypes (or definitions) are required and
2129         *    therefore there is no ambiguity when an empty argument list "( )" is
2130         *    declared. The idiom "(void)" as a parameter list is provided for
2131         *    convenience."
2132         *
2133         * Placing this check here prevents a void parameter being set up
2134         * for a function, which avoids tripping up checks for main taking
2135         * parameters and lookups of an unnamed symbol.
2136         */
2137        if (type->is_void()) {
2138           if (this->identifier != NULL)
2139     	 _mesa_glsl_error(& loc, state,
2140     			  "named parameter cannot have type `void'");
2141     
2142           is_void = true;
2143           return NULL;
2144        }
2145     
2146        if (formal_parameter && (this->identifier == NULL)) {
2147           _mesa_glsl_error(& loc, state, "formal parameter lacks a name");
2148           return NULL;
2149        }
2150     
2151        /* This only handles "vec4 foo[..]".  The earlier specifier->glsl_type(...)
2152         * call already handled the "vec4[..] foo" case.
2153         */
2154        if (this->is_array) {
2155           type = process_array_type(&loc, type, this->array_size, state);
2156        }
2157     
2158        if (type->array_size() == 0) {
2159           _mesa_glsl_error(&loc, state, "arrays passed as parameters must have "
2160     		       "a declared size.");
2161           type = glsl_type::error_type;
2162        }
2163     
2164        is_void = false;
2165        ir_variable *var = new(ctx) ir_variable(type, this->identifier, ir_var_in);
2166     
2167        /* Apply any specified qualifiers to the parameter declaration.  Note that
2168         * for function parameters the default mode is 'in'.
2169         */
2170        apply_type_qualifier_to_variable(& this->type->qualifier, var, state, & loc);
2171     
2172        instructions->push_tail(var);
2173     
2174        /* Parameter declarations do not have r-values.
2175         */
2176        return NULL;
2177     }

2182     					    bool formal,
2183     					    exec_list *ir_parameters,

2200           YYLTYPE loc = void_param->get_location();

2208     ir_rvalue *
2209     ast_function::hir(exec_list *instructions,
2210     		  struct _mesa_glsl_parse_state *state)
2211     {
2212        void *ctx = state;
2213        ir_function *f = NULL;
2214        ir_function_signature *sig = NULL;
2215        exec_list hir_parameters;
2216     
2217        const char *const name = identifier;
2218     
2219        /* From page 21 (page 27 of the PDF) of the GLSL 1.20 spec,
2220         *
2221         *   "Function declarations (prototypes) cannot occur inside of functions;
2222         *   they must be at global scope, or for the built-in functions, outside
2223         *   the global scope."
2224         *
2225         * From page 27 (page 33 of the PDF) of the GLSL ES 1.00.16 spec,
2226         *
2227         *   "User defined functions may only be defined within the global scope."
2228         *
2229         * Note that this language does not appear in GLSL 1.10.
2230         */
2231        if ((state->current_function != NULL) && (state->language_version != 110)) {
2232           YYLTYPE loc = this->get_location();
2233           _mesa_glsl_error(&loc, state,
2234     		       "declaration of function `%s' not allowed within "
2235     		       "function body", name);
2236        }
2237     
2238        /* From page 15 (page 21 of the PDF) of the GLSL 1.10 spec,
2239         *
2240         *   "Identifiers starting with "gl_" are reserved for use by
2241         *   OpenGL, and may not be declared in a shader as either a
2242         *   variable or a function."
2243         */
2244        if (strncmp(name, "gl_", 3) == 0) {
2245           YYLTYPE loc = this->get_location();
2246           _mesa_glsl_error(&loc, state,
2247     		       "identifier `%s' uses reserved `gl_' prefix", name);
2248        }
2249     
2250        /* Convert the list of function parameters to HIR now so that they can be
2251         * used below to compare this function's signature with previously seen
2252         * signatures for functions with the same name.
2253         */
2254        ast_parameter_declarator::parameters_to_hir(& this->parameters,
2255     					       is_definition,
2256     					       & hir_parameters, state);
2257     
2258        const char *return_type_name;
2259        const glsl_type *return_type =
2260           this->return_type->specifier->glsl_type(& return_type_name, state);
2261     
2262        if (!return_type) {
2263           YYLTYPE loc = this->get_location();
2264           _mesa_glsl_error(&loc, state,
2265     		       "function `%s' has undeclared return type `%s'",
2266     		       name, return_type_name);
2267           return_type = glsl_type::error_type;
2268        }
2269     
2270        /* From page 56 (page 62 of the PDF) of the GLSL 1.30 spec:
2271         * "No qualifier is allowed on the return type of a function."
2272         */
2273        if (this->return_type->has_qualifiers()) {
2274           YYLTYPE loc = this->get_location();
2275           _mesa_glsl_error(& loc, state,
2276     		       "function `%s' return type has qualifiers", name);
2277        }
2278     
2279        /* Verify that this function's signature either doesn't match a previously
2280         * seen signature for a function with the same name, or, if a match is found,
2281         * that the previously seen signature does not have an associated definition.
2282         */
2283        f = state->symbols->get_function(name);
2284        if (f != NULL && (state->es_shader || f->has_user_signature())) {
2285           sig = f->exact_matching_signature(&hir_parameters);
2286           if (sig != NULL) {
2287     	 const char *badvar = sig->qualifiers_match(&hir_parameters);
2288     	 if (badvar != NULL) {
2289     	    YYLTYPE loc = this->get_location();
2290     
2291     	    _mesa_glsl_error(&loc, state, "function `%s' parameter `%s' "
2292     			     "qualifiers don't match prototype", name, badvar);
2293     	 }
2294     
2295     	 if (sig->return_type != return_type) {
2296     	    YYLTYPE loc = this->get_location();
2297     
2298     	    _mesa_glsl_error(&loc, state, "function `%s' return type doesn't "
2299     			     "match prototype", name);
2300     	 }
2301     
2302     	 if (is_definition && sig->is_defined) {
2303     	    YYLTYPE loc = this->get_location();
2304     
2305     	    _mesa_glsl_error(& loc, state, "function `%s' redefined", name);
2306     	 }
2307           }
2308        } else {
2309           f = new(ctx) ir_function(name);
2310           if (!state->symbols->add_function(f->name, f)) {
2311     	 /* This function name shadows a non-function use of the same name. */
2312     	 YYLTYPE loc = this->get_location();
2313     
2314     	 _mesa_glsl_error(&loc, state, "function name `%s' conflicts with "
2315     			  "non-function", name);
2316     	 return NULL;
2317           }
2318     
2319           /* Emit the new function header */
2320           if (state->current_function == NULL)
2321     	 instructions->push_tail(f);
2322           else {
2323     	 /* IR invariants disallow function declarations or definitions nested
2324     	  * within other function definitions.  Insert the new ir_function
2325     	  * block in the instruction sequence before the ir_function block
2326     	  * containing the current ir_function_signature.
2327     	  *
2328     	  * This can only happen in a GLSL 1.10 shader.  In all other GLSL
2329     	  * versions this nesting is disallowed.  There is a check for this at
2330     	  * the top of this function.
2331     	  */
2332     	 ir_function *const curr =
2333     	    const_cast<ir_function *>(state->current_function->function());
2334     
2335     	 curr->insert_before(f);
2336           }
2337        }
2338     
2339        /* Verify the return type of main() */
2340        if (safe_strcmp(name, "main") == 0) {
2341           if (! return_type->is_void()) {
2342     	 YYLTYPE loc = this->get_location();
2343     
2344     	 _mesa_glsl_error(& loc, state, "main() must return void");
2345           }
2346     
2347           if (!hir_parameters.is_empty()) {
2348     	 YYLTYPE loc = this->get_location();
2349     
2350     	 _mesa_glsl_error(& loc, state, "main() must not take any parameters");
2351           }
2352        }
2353     
2354        /* Finish storing the information about this new function in its signature.
2355         */
2356        if (sig == NULL) {
2357           sig = new(ctx) ir_function_signature(return_type);
2358           f->add_signature(sig);
2359        }
2360     
2361        sig->replace_parameters(&hir_parameters);
2362        signature = sig;
2363     
2364        /* Function declarations (prototypes) do not have r-values.
2365         */
2366        return NULL;
2367     }

2369     
2370     ir_rvalue *
2371     ast_function_definition::hir(exec_list *instructions,
2372     			     struct _mesa_glsl_parse_state *state)
2373     {
2374        prototype->is_definition = true;
2375        prototype->hir(instructions, state);
2376     
2377        ir_function_signature *signature = prototype->signature;
2378        if (signature == NULL)
2379           return NULL;
2380     
2381        assert(state->current_function == NULL);
2382        state->current_function = signature;
2383        state->found_return = false;
2384     
2385        /* Duplicate parameters declared in the prototype as concrete variables.
2386         * Add these to the symbol table.
2387         */
2388        state->symbols->push_scope();
2389        foreach_iter(exec_list_iterator, iter, signature->parameters) {
2390           ir_variable *const var = ((ir_instruction *) iter.get())->as_variable();
2391     
2392           assert(var != NULL);
2393     
2394           /* The only way a parameter would "exist" is if two parameters have
2395            * the same name.
2396            */
2397           if (state->symbols->name_declared_this_scope(var->name)) {
2398     	 YYLTYPE loc = this->get_location();
2399     
2400     	 _mesa_glsl_error(& loc, state, "parameter `%s' redeclared", var->name);
2401           } else {
2402     	 state->symbols->add_variable(var->name, var);
2403           }
2404        }
2405     
2406        /* Convert the body of the function to HIR. */
2407        this->body->hir(&signature->body, state);
2408        signature->is_defined = true;
2409     
2410        state->symbols->pop_scope();
2411     
2412        assert(state->current_function == signature);
2413        state->current_function = NULL;
2414     
2415        if (!signature->return_type->is_void() && !state->found_return) {
2416           YYLTYPE loc = this->get_location();
2417           _mesa_glsl_error(& loc, state, "function `%s' has non-void return type "
2418     		       "%s, but no return statement",
2419     		       signature->function_name(),
2420     		       signature->return_type->name);
2421        }
2422     
2423        /* Function definitions do not have r-values.
2424         */
2425        return NULL;
2426     }

2429     ir_rvalue *
2430     ast_jump_statement::hir(exec_list *instructions,
2431     			struct _mesa_glsl_parse_state *state)
2432     {
2433        void *ctx = state;
2434     
2435        switch (mode) {
2436        case ast_return: {
2437           ir_return *inst;
2438           assert(state->current_function);
2439     
2440           if (opt_return_value) {
2441     	 if (state->current_function->return_type->base_type ==
2442     	     GLSL_TYPE_VOID) {
2443     	    YYLTYPE loc = this->get_location();
2444     
2445     	    _mesa_glsl_error(& loc, state,
2446     			     "`return` with a value, in function `%s' "
2447     			     "returning void",
2448     			     state->current_function->function_name());
2449     	 }
2450     
2451     	 ir_expression *const ret = (ir_expression *)
2452     	    opt_return_value->hir(instructions, state);
2453     	 assert(ret != NULL);
2454     
2455     	 /* Implicit conversions are not allowed for return values. */
2456     	 if (state->current_function->return_type != ret->type) {
2457     	    YYLTYPE loc = this->get_location();
2458     
2459     	    _mesa_glsl_error(& loc, state,
2460     			     "`return' with wrong type %s, in function `%s' "
2461     			     "returning %s",
2462     			     ret->type->name,
2463     			     state->current_function->function_name(),
2464     			     state->current_function->return_type->name);
2465     	 }
2466     
2467     	 inst = new(ctx) ir_return(ret);
2468           } else {
2469     	 if (state->current_function->return_type->base_type !=
2470     	     GLSL_TYPE_VOID) {
2471     	    YYLTYPE loc = this->get_location();
2472     
2473     	    _mesa_glsl_error(& loc, state,
2474     			     "`return' with no value, in function %s returning "
2475     			     "non-void",
2476     			     state->current_function->function_name());
2477     	 }
2478     	 inst = new(ctx) ir_return;
2479           }
2480     
2481           state->found_return = true;
2482           instructions->push_tail(inst);
2483           break;
2484        }
2485     
2486        case ast_discard:
2487           if (state->target != fragment_shader) {
2488     	 YYLTYPE loc = this->get_location();
2489     
2490     	 _mesa_glsl_error(& loc, state,
2491     			  "`discard' may only appear in a fragment shader");
2492           }
2493           instructions->push_tail(new(ctx) ir_discard);
2494           break;
2495     
2496        case ast_break:
2497        case ast_continue:
2498           /* FINISHME: Handle switch-statements.  They cannot contain 'continue',
2499            * FINISHME: and they use a different IR instruction for 'break'.
2500            */
2501           /* FINISHME: Correctly handle the nesting.  If a switch-statement is
2502            * FINISHME: inside a loop, a 'continue' is valid and will bind to the
2503            * FINISHME: loop.
2504            */
2505           if (state->loop_or_switch_nesting == NULL) {
2506     	 YYLTYPE loc = this->get_location();
2507     
2508     	 _mesa_glsl_error(& loc, state,
2509     			  "`%s' may only appear in a loop",
2510     			  (mode == ast_break) ? "break" : "continue");
2511           } else {
2512     	 ir_loop *const loop = state->loop_or_switch_nesting->as_loop();
2513     
2514     	 /* Inline the for loop expression again, since we don't know
2515     	  * where near the end of the loop body the normal copy of it
2516     	  * is going to be placed.
2517     	  */
2518     	 if (mode == ast_continue &&
2519     	     state->loop_or_switch_nesting_ast->rest_expression) {
2520     	    state->loop_or_switch_nesting_ast->rest_expression->hir(instructions,
2521     								    state);
2522     	 }
2523     
2524     	 if (loop != NULL) {
2525     	    ir_loop_jump *const jump =
2526     	       new(ctx) ir_loop_jump((mode == ast_break)
2527     				     ? ir_loop_jump::jump_break
2528     				     : ir_loop_jump::jump_continue);
2529     	    instructions->push_tail(jump);
2530     	 }
2531           }
2532     
2533           break;
2534        }
2535     
2536        /* Jump instructions do not have r-values.
2537         */
2538        return NULL;
2539     }

2542     ir_rvalue *
2543     ast_selection_statement::hir(exec_list *instructions,
2544     			     struct _mesa_glsl_parse_state *state)
2545     {
2546        void *ctx = state;
2547     
2548        ir_rvalue *const condition = this->condition->hir(instructions, state);
2549     
2550        /* From page 66 (page 72 of the PDF) of the GLSL 1.50 spec:
2551         *
2552         *    "Any expression whose type evaluates to a Boolean can be used as the
2553         *    conditional expression bool-expression. Vector types are not accepted
2554         *    as the expression to if."
2555         *
2556         * The checks are separated so that higher quality diagnostics can be
2557         * generated for cases where both rules are violated.
2558         */
2559        if (!condition->type->is_boolean() || !condition->type->is_scalar()) {
2560           YYLTYPE loc = this->condition->get_location();
2561     
2562           _mesa_glsl_error(& loc, state, "if-statement condition must be scalar "
2563     		       "boolean");
2564        }
2565     
2566        ir_if *const stmt = new(ctx) ir_if(condition);
2567     
2568        if (then_statement != NULL) {
2569           state->symbols->push_scope();
2570           then_statement->hir(& stmt->then_instructions, state);
2571           state->symbols->pop_scope();
2572        }
2573     
2574        if (else_statement != NULL) {
2575           state->symbols->push_scope();
2576           else_statement->hir(& stmt->else_instructions, state);
2577           state->symbols->pop_scope();
2578        }
2579     
2580        instructions->push_tail(stmt);
2581     
2582        /* if-statements do not have r-values.
2583         */
2584        return NULL;
2585     }

2588     void
2589     ast_iteration_statement::condition_to_hir(ir_loop *stmt,
2590     					  struct _mesa_glsl_parse_state *state)
2591     {
2592        void *ctx = state;
2593     
2594        if (condition != NULL) {
2595           ir_rvalue *const cond =
2596     	 condition->hir(& stmt->body_instructions, state);
2597     
2598           if ((cond == NULL)
2599     	  || !cond->type->is_boolean() || !cond->type->is_scalar()) {
2600     	 YYLTYPE loc = condition->get_location();
2601     
2602     	 _mesa_glsl_error(& loc, state,
2603     			  "loop condition must be scalar boolean");
2604           } else {
2605     	 /* As the first code in the loop body, generate a block that looks
2606     	  * like 'if (!condition) break;' as the loop termination condition.
2607     	  */
2608     	 ir_rvalue *const not_cond =
2609     	    new(ctx) ir_expression(ir_unop_logic_not, glsl_type::bool_type, cond,
2610     				   NULL);
2611     
2612     	 ir_if *const if_stmt = new(ctx) ir_if(not_cond);
2613     
2614     	 ir_jump *const break_stmt =
2615     	    new(ctx) ir_loop_jump(ir_loop_jump::jump_break);
2616     
2617     	 if_stmt->then_instructions.push_tail(break_stmt);
2618     	 stmt->body_instructions.push_tail(if_stmt);
2619           }
2620        }
2621     }

2624     ir_rvalue *
2625     ast_iteration_statement::hir(exec_list *instructions,
2626     			     struct _mesa_glsl_parse_state *state)
2627     {
2628        void *ctx = state;
2629     
2630        /* For-loops and while-loops start a new scope, but do-while loops do not.
2631         */
2632        if (mode != ast_do_while)
2633           state->symbols->push_scope();
2634     
2635        if (init_statement != NULL)
2636           init_statement->hir(instructions, state);
2637     
2638        ir_loop *const stmt = new(ctx) ir_loop();
2639        instructions->push_tail(stmt);
2640     
2641        /* Track the current loop and / or switch-statement nesting.
2642         */
2643        ir_instruction *const nesting = state->loop_or_switch_nesting;
2644        ast_iteration_statement *nesting_ast = state->loop_or_switch_nesting_ast;
2645     
2646        state->loop_or_switch_nesting = stmt;
2647        state->loop_or_switch_nesting_ast = this;
2648     
2649        if (mode != ast_do_while)
2650           condition_to_hir(stmt, state);
2651     
2652        if (body != NULL)
2653           body->hir(& stmt->body_instructions, state);
2654     
2655        if (rest_expression != NULL)
2656           rest_expression->hir(& stmt->body_instructions, state);
2657     
2658        if (mode == ast_do_while)
2659           condition_to_hir(stmt, state);
2660     
2661        if (mode != ast_do_while)
2662           state->symbols->pop_scope();
2663     
2664        /* Restore previous nesting before returning.
2665         */
2666        state->loop_or_switch_nesting = nesting;
2667        state->loop_or_switch_nesting_ast = nesting_ast;
2668     
2669        /* Loops do not have r-values.
2670         */
2671        return NULL;
2672     }

2686     ir_rvalue *
2687     ast_struct_specifier::hir(exec_list *instructions,
2688     			  struct _mesa_glsl_parse_state *state)
2689     {
2690        unsigned decl_count = 0;
2691     
2692        /* Make an initial pass over the list of structure fields to determine how
2693         * many there are.  Each element in this list is an ast_declarator_list.
2694         * This means that we actually need to count the number of elements in the
2695         * 'declarations' list in each of the elements.
2696         */
2697        foreach_list_typed (ast_declarator_list, decl_list, link,
2698     		       &this->declarations) {
2699           foreach_list_const (decl_ptr, & decl_list->declarations) {
2700     	 decl_count++;
2701           }
2702        }
2703     
2704        /* Allocate storage for the structure fields and process the field
2705         * declarations.  As the declarations are processed, try to also convert
2706         * the types to HIR.  This ensures that structure definitions embedded in
2707         * other structure definitions are processed.
2708         */
2709        glsl_struct_field *const fields = talloc_array(state, glsl_struct_field,
2710     						  decl_count);
2711     
2712        unsigned i = 0;
2713        foreach_list_typed (ast_declarator_list, decl_list, link,
2714     		       &this->declarations) {
2715           const char *type_name;
2716     
2717           decl_list->type->specifier->hir(instructions, state);
2718     
2719           /* Section 10.9 of the GLSL ES 1.00 specification states that
2720            * embedded structure definitions have been removed from the language.
2721            */
2722           if (state->es_shader && decl_list->type->specifier->structure != NULL) {
2723     	 YYLTYPE loc = this->get_location();
2724     	 _mesa_glsl_error(&loc, state, "Embedded structure definitions are "
2725     			  "not allowed in GLSL ES 1.00.");
2726           }
2727     
2728           const glsl_type *decl_type =
2729     	 decl_list->type->specifier->glsl_type(& type_name, state);
2730     
2731           foreach_list_typed (ast_declaration, decl, link,
2732     			  &decl_list->declarations) {
2733     	 const struct glsl_type *field_type = decl_type;
2734     	 if (decl->is_array) {
2735     	    YYLTYPE loc = decl->get_location();
2736     	    field_type = process_array_type(&loc, decl_type, decl->array_size,
2737     					    state);
2738     	 }
2739     	 fields[i].type = (field_type != NULL)
2740     	    ? field_type : glsl_type::error_type;
2741     	 fields[i].name = decl->identifier;
2742     	 i++;
2743           }
2744        }
2745     
2746        assert(i == decl_count);
2747     
2748        const glsl_type *t =
2749           glsl_type::get_record_instance(fields, decl_count, this->name);
2750     
2751        YYLTYPE loc = this->get_location();
2752        if (!state->symbols->add_type(name, t)) {
2753           _mesa_glsl_error(& loc, state, "struct `%s' previously defined", name);
2754        } else {
2755     
2756           const glsl_type **s = (const glsl_type **)
2757     	 realloc(state->user_structures,
2758     		 sizeof(state->user_structures[0]) *
2759     		 (state->num_user_structures + 1));
2760           if (s != NULL) {
2761     	 s[state->num_user_structures] = t;
2762     	 state->user_structures = s;
2763     	 state->num_user_structures++;
2764           }
2765        }
2766     
2767        /* Structure type definitions do not have r-values.
2768         */
2769        return NULL;
2770     }
