46     static void onApply(HWND hWnd)
47     {
48       Logger * logger = (Logger *)GetWindowLong(hWnd, DWL_USER);
49       if(!logger)
50         return;
51     
52       logger->bOnTop = (BST_CHECKED == IsDlgButtonChecked(hWnd, IDC_CHECK_ONTOP));
53       logger->bSaveSettings = TRUE;
54     }
55     
56     static void onNotify(HWND hWnd, int idCtrl, LPNMHDR lpNMHdr)
57     {
58       switch(lpNMHdr->code)
59       {
60         case PSN_RESET:
61           break;
62         case PSN_APPLY:
63           onApply(hWnd);
64           break;
65       }
66     }
67     
68     static BOOL onInitDialog(HWND hWnd, HWND hWndFocus, LPARAM lParam)
69     {
70       Logger * logger = NULL;
71     
72       if(lParam)
73       {
74         logger = (Logger *)(((PROPSHEETPAGE *)lParam)->lParam);
75         SetWindowLong(hWnd, DWL_USER, (long)logger);
76       }
77     
78       if(logger)
79       {
80         CheckDlgButton(hWnd, IDC_CHECK_ONTOP, logger->bOnTop ? BST_CHECKED : BST_UNCHECKED);
81       }
82     
83       return TRUE;
84     }
85     
86     BOOL CALLBACK GeneralPageProc(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam)
87     {
88       switch(msg)
89       {
90         case WM_INITDIALOG:
91           return (BOOL)HANDLE_WM_INITDIALOG(hWnd, wParam, lParam, onInitDialog);
92         case WM_NOTIFY:
93           HANDLE_WM_NOTIFY(hWnd, wParam, lParam, onNotify);
94           break;
95     
96         default:
97           return FALSE;
98       }
99       return TRUE;
100     }
