24     // Find the full path of the current executable. On success return true
25     // and sets |exe_path|. On failure, return false and sets errno.
26     bool FindExecutablePath(String* exe_path) {
27       // /proc/self/exe is a symlink to the full path. Read it with
28       // readlink().
29       exe_path->Resize(512);
30       ssize_t ret = TEMP_FAILURE_RETRY(
31           readlink("/proc/self/exe", exe_path->ptr(), exe_path->size()));
32       if (ret < 0) {
33         LOG_ERRNO("%s: Could not get /proc/self/exe link", __FUNCTION__);
34         return false;
35       }
36     
37       exe_path->Resize(static_cast<size_t>(ret));
38       LOG("%s: Current executable: %s\n", __FUNCTION__, exe_path->c_str());
39       return true;
40     }

42     // Given an ELF binary at |path| that is _already_ mapped in the process,
43     // find the address of its dynamic section and its size.
44     // |path| is the full path of the binary (as it appears in /proc/self/maps.
45     // |self_maps| is an instance of ProcMaps that is used to inspect
46     // /proc/self/maps. The function rewind + iterates over it.
47     // On success, return true and set |*dynamic_offset| and |*dynamic_size|.
48     bool FindElfDynamicSection(const char* path,
49                                ProcMaps* self_maps,
50                                size_t* dynamic_address,
51                                size_t* dynamic_size) {
52       // Read the ELF header first.
53       ELF::Ehdr header[1];
54     
55       crazy::FileDescriptor fd;
56       if (!fd.OpenReadOnly(path) ||
57           fd.Read(header, sizeof(header)) != static_cast<int>(sizeof(header))) {
58         LOG_ERRNO("%s: Could not load ELF binary header", __FUNCTION__);
59         return false;
60       }
61     
62       // Sanity check.
63       if (header->e_ident[0] != 127 || header->e_ident[1] != 'E' ||
64           header->e_ident[2] != 'L' || header->e_ident[3] != 'F' ||
65           header->e_ident[4] != ELF::kElfClass) {
66         LOG("%s: Not a %d-bit ELF binary: %s\n",
67             __FUNCTION__,
68             ELF::kElfBits,
69             path);
70         return false;
71       }
72     
73       if (header->e_phoff == 0 || header->e_phentsize != sizeof(ELF::Phdr)) {
74         LOG("%s: Invalid program header values: %s\n", __FUNCTION__, path);
75         return false;
76       }
77     
78       // Scan the program header table.
79       if (fd.SeekTo(header->e_phoff) < 0) {
80         LOG_ERRNO("%s: Could not find ELF program header table", __FUNCTION__);
81         return false;
82       }
83     
84       ELF::Phdr phdr_load0 = {0, };
85       ELF::Phdr phdr_dyn = {0, };
86       bool found_load0 = false;
87       bool found_dyn = false;
88     
89       for (size_t n = 0; n < header->e_phnum; ++n) {
90         ELF::Phdr phdr;
91         if (fd.Read(&phdr, sizeof(phdr)) != sizeof(phdr)) {
92           LOG_ERRNO("%s: Could not read program header entry", __FUNCTION__);
93           return false;
94         }
95     
96         if (phdr.p_type == PT_LOAD && !found_load0) {
97           phdr_load0 = phdr;
98           found_load0 = true;
99         } else if (phdr.p_type == PT_DYNAMIC && !found_dyn) {
100           phdr_dyn = phdr;
101           found_dyn = true;
102         }
103       }
104     
105       if (!found_load0) {
106         LOG("%s: Could not find loadable segment!?\n", __FUNCTION__);
107         return false;
108       }
109       if (!found_dyn) {
110         LOG("%s: Could not find dynamic segment!?\n", __FUNCTION__);
111         return false;
112       }
113     
114       LOG("%s: Found first loadable segment [offset=%p vaddr=%p]\n",
115           __FUNCTION__,
116           (void*)phdr_load0.p_offset,
117           (void*)phdr_load0.p_vaddr);
118     
119       LOG("%s: Found dynamic segment [offset=%p vaddr=%p size=%p]\n",
120           __FUNCTION__,
121           (void*)phdr_dyn.p_offset,
122           (void*)phdr_dyn.p_vaddr,
123           (void*)phdr_dyn.p_memsz);
124     
125       // Parse /proc/self/maps to find the load address of the first
126       // loadable segment.
127       size_t path_len = strlen(path);
128       self_maps->Rewind();
129       ProcMaps::Entry entry;
130       while (self_maps->GetNextEntry(&entry)) {
131         if (!entry.path || entry.path_len != path_len ||
132             memcmp(entry.path, path, path_len) != 0)
133           continue;
134     
135         LOG("%s: Found executable segment mapped [%p-%p offset=%p]\n",
136             __FUNCTION__,
137             (void*)entry.vma_start,
138             (void*)entry.vma_end,
139             (void*)entry.load_offset);
140     
141         size_t load_bias = entry.vma_start - phdr_load0.p_vaddr;
142         LOG("%s: Load bias is %p\n", __FUNCTION__, (void*)load_bias);
143     
144         *dynamic_address = load_bias + phdr_dyn.p_vaddr;
145         *dynamic_size = phdr_dyn.p_memsz;
146         LOG("%s: Dynamic section addr=%p size=%p\n",
147             __FUNCTION__,
148             (void*)*dynamic_address,
149             (void*)*dynamic_size);
150         return true;
151       }
152     
153       LOG("%s: Executable is not mapped in current process.\n", __FUNCTION__);
154       return false;
155     }

157     // Helper class to temporarily remap a page to readable+writable until
158     // scope exit.
159     class ScopedPageReadWriteRemapper {
160      public:
161       ScopedPageReadWriteRemapper(void* address);
162       ~ScopedPageReadWriteRemapper();
163     
164       // Releases the page so that the destructor does not undo the remapping.
165       void Release();
166     
167      private:
168       static const uintptr_t kPageSize = 4096;
169       uintptr_t page_address_;
170       int page_prot_;
171     };
172     
173     ScopedPageReadWriteRemapper::ScopedPageReadWriteRemapper(void* address) {
174       page_address_ = reinterpret_cast<uintptr_t>(address) & ~(kPageSize - 1);
175       page_prot_ = 0;
176       if (!FindProtectionFlagsForAddress(address, &page_prot_)) {
177         LOG("Could not find protection flags for %p\n", address);
178         page_address_ = 0;
179         return;
180       }
181     
182       // Note: page_prot_ may already indicate read/write, but because of
183       // possible races with the system linker we cannot be confident that
184       // this is reliable. So we always set read/write here.
185       //
186       // See commentary in WriteLinkMapField for more.
187       int new_page_prot = page_prot_ | PROT_READ | PROT_WRITE;
188       int ret = mprotect(
189           reinterpret_cast<void*>(page_address_), kPageSize, new_page_prot);
190       if (ret < 0) {
191         LOG_ERRNO("Could not remap page to read/write");
192         page_address_ = 0;
193       }
194     }
195     
196     ScopedPageReadWriteRemapper::~ScopedPageReadWriteRemapper() {
197       if (page_address_) {
198         int ret =
199             mprotect(reinterpret_cast<void*>(page_address_), kPageSize, page_prot_);
200         if (ret < 0)
201           LOG_ERRNO("Could not remap page to old protection flags");
202       }
203     }
204     
205     void ScopedPageReadWriteRemapper::Release() {
206       page_address_ = 0;
207       page_prot_ = 0;
208     }

212     bool RDebug::Init() {
213       // The address of '_r_debug' is in the DT_DEBUG entry of the current
214       // executable.
215       init_ = true;
216     
217       size_t dynamic_addr = 0;
218       size_t dynamic_size = 0;
219       String path;
220     
221       // Find the current executable's full path, and its dynamic section
222       // information.
223       if (!FindExecutablePath(&path))
224         return false;
225     
226       ProcMaps self_maps;
227       if (!FindElfDynamicSection(
228                path.c_str(), &self_maps, &dynamic_addr, &dynamic_size)) {
229         return false;
230       }
231     
232       // Parse the dynamic table and find the DT_DEBUG entry.
233       const ELF::Dyn* dyn_section = reinterpret_cast<const ELF::Dyn*>(dynamic_addr);
234     
235       while (dynamic_size >= sizeof(*dyn_section)) {
236         if (dyn_section->d_tag == DT_DEBUG) {
237           // Found it!
238           LOG("%s: Found DT_DEBUG entry inside %s at %p, pointing to %p\n",
239               __FUNCTION__,
240               path.c_str(),
241               dyn_section,
242               dyn_section->d_un.d_ptr);
243           if (dyn_section->d_un.d_ptr) {
244             r_debug_ = reinterpret_cast<r_debug*>(dyn_section->d_un.d_ptr);
245             LOG("%s: r_debug [r_version=%d r_map=%p r_brk=%p r_ldbase=%p]\n",
246                 __FUNCTION__,
247                 r_debug_->r_version,
248                 r_debug_->r_map,
249                 r_debug_->r_brk,
250                 r_debug_->r_ldbase);
251             // Only version 1 of the struct is supported.
252             if (r_debug_->r_version != 1) {
253               LOG("%s: r_debug.r_version is %d, 1 expected.\n",
254                   __FUNCTION__,
255                   r_debug_->r_version);
256               r_debug_ = NULL;
257             }
258     
259             // The linker of recent Android releases maps its link map entries
260             // in read-only pages. Determine if this is the case and record it
261             // for later. The first entry in the list corresponds to the
262             // executable.
263             int prot = self_maps.GetProtectionFlagsForAddress(r_debug_->r_map);
264             readonly_entries_ = (prot & PROT_WRITE) == 0;
265     
266             LOG("%s: r_debug.readonly_entries=%s\n",
267                 __FUNCTION__,
268                 readonly_entries_ ? "true" : "false");
269             return true;
270           }
271         }
272         dyn_section++;
273         dynamic_size -= sizeof(*dyn_section);
274       }
275     
276       LOG("%s: There is no non-0 DT_DEBUG entry in this process\n", __FUNCTION__);
277       return false;
278     }

296     // Helper runnable class. Handler is one of the two static functions
297     // AddEntryInternal() or DelEntryInternal(). Calling these invokes
298     // AddEntryImpl() or DelEntryImpl() respectively on rdebug.
299     class RDebugRunnable {
300      public:
301       RDebugRunnable(rdebug_callback_handler_t handler,
302                      RDebug* rdebug,
303                      link_map_t* entry,
304                      bool is_blocking)
305           : handler_(handler), rdebug_(rdebug),
306             entry_(entry), is_blocking_(is_blocking), has_run_(false) {
307         pthread_mutex_init(&mutex_, NULL);
308         pthread_cond_init(&cond_, NULL);
309       }
310     
311       static void Run(void* opaque);
312       static void WaitForCallback(void* opaque);
313     
314      private:
315       rdebug_callback_handler_t handler_;
316       RDebug* rdebug_;
317       link_map_t* entry_;
318       bool is_blocking_;
319       bool has_run_;
320       pthread_mutex_t mutex_;
321       pthread_cond_t cond_;
322     };
323     
324     // Callback entry point.
325     void RDebugRunnable::Run(void* opaque) {
326       RDebugRunnable* runnable = static_cast<RDebugRunnable*>(opaque);
327     
328       LOG("%s: Callback received, runnable=%p\n", __FUNCTION__, runnable);
329       (*runnable->handler_)(runnable->rdebug_, runnable->entry_);
330     
331       if (!runnable->is_blocking_) {
332         delete runnable;
333         return;
334       }
335     
336       LOG("%s: Signalling callback, runnable=%p\n", __FUNCTION__, runnable);
337       {
338         ScopedMutexLock m(&runnable->mutex_);
339         runnable->has_run_ = true;
340         pthread_cond_signal(&runnable->cond_);
341       }
342     }
343     
344     // For blocking callbacks, wait for the call to Run().
345     void RDebugRunnable::WaitForCallback(void* opaque) {
346       RDebugRunnable* runnable = static_cast<RDebugRunnable*>(opaque);
347     
348       if (!runnable->is_blocking_) {
349         LOG("%s: Non-blocking, not waiting, runnable=%p\n", __FUNCTION__, runnable);
350         return;
351       }
352     
353       LOG("%s: Waiting for signal, runnable=%p\n", __FUNCTION__, runnable);
354       {
355         ScopedMutexLock m(&runnable->mutex_);
356         while (!runnable->has_run_)
357           pthread_cond_wait(&runnable->cond_, &runnable->mutex_);
358       }
359     
360       delete runnable;
361     }

365     // Helper function to schedule AddEntry() and DelEntry() calls onto another
366     // thread where possible. Running them there avoids races with the system
367     // linker, which expects to be able to set r_map pages readonly when it
368     // is not using them and which may run simultaneously on the main thread.
369     bool RDebug::PostCallback(rdebug_callback_handler_t handler,
370                               link_map_t* entry,
371                               bool is_blocking) {
372       if (!post_for_later_execution_) {
373         LOG("%s: Deferred execution disabled\n", __FUNCTION__);
374         return false;
375       }
376     
377       RDebugRunnable* runnable =
378           new RDebugRunnable(handler, this, entry, is_blocking);
379       void* context = post_for_later_execution_context_;
380     
381       if (!(*post_for_later_execution_)(context, &RDebugRunnable::Run, runnable)) {
382         LOG("%s: Deferred execution enabled, but posting failed\n", __FUNCTION__);
383         delete runnable;
384         return false;
385       }
386     
387       LOG("%s: Posted for later execution, runnable=%p\n", __FUNCTION__, runnable);
388     
389       if (is_blocking) {
390         RDebugRunnable::WaitForCallback(runnable);
391         LOG("%s: Completed execution, runnable=%p\n", __FUNCTION__, runnable);
392       }
393     
394       return true;
395     }

397     // Helper function for AddEntryImpl and DelEntryImpl.
398     // Sets *link_pointer to entry.  link_pointer is either an 'l_prev' or an
399     // 'l_next' field in a neighbouring linkmap_t.  If link_pointer is in a
400     // page that is mapped readonly, the page is remapped to be writable before
401     // assignment.
402     void RDebug::WriteLinkMapField(link_map_t** link_pointer, link_map_t* entry) {
403       ScopedPageReadWriteRemapper mapper(link_pointer);
404       LOG("%s: Remapped page for %p for read/write\n", __FUNCTION__, link_pointer);
405     
406       *link_pointer = entry;
407     
408       // We always mprotect the page containing link_pointer to read/write,
409       // then write the entry. The page may already be read/write, but on
410       // recent Android release is most likely readonly. Because of the way
411       // the system linker operates we cannot tell with certainty what its
412       // correct setting should be.
413       //
414       // Now, we always leave the page read/write. Here is why. If we set it
415       // back to readonly at the point between where the system linker sets
416       // it to read/write and where it writes to the address, this will cause
417       // the system linker to crash. Clearly that is undesirable. From
418       // observations this occurs most frequently on the gpu process.
419       //
420       // TODO(simonb): Revisit this, details in:
421       // https://code.google.com/p/chromium/issues/detail?id=450659
422       // https://code.google.com/p/chromium/issues/detail?id=458346
423       mapper.Release();
424       LOG("%s: Released mapper, leaving page read/write\n", __FUNCTION__);
425     }

427     void RDebug::AddEntryImpl(link_map_t* entry) {
428       ScopedGlobalLock lock;
429       LOG("%s: Adding: %s\n", __FUNCTION__, entry->l_name);
430       if (!init_)
431         Init();
432     
433       if (!r_debug_) {
434         LOG("%s: Nothing to do\n", __FUNCTION__);
435         return;
436       }
437     
438       // Tell GDB the list is going to be modified.
439       r_debug_->r_state = RT_ADD;
440       r_debug_->r_brk();
441     
442       // IMPORTANT: GDB expects the first entry in the list to correspond
443       // to the executable. So add our new entry just after it. This is ok
444       // because by default, the linker is always the second entry, as in:
445       //
446       //   [<executable>, /system/bin/linker, libc.so, libm.so, ...]
447       //
448       // By design, the first two entries should never be removed since they
449       // can't be unloaded from the process (they are loaded by the kernel
450       // when invoking the program).
451       //
452       // TODO(digit): Does GDB expect the linker to be the second entry?
453       // It doesn't seem so, but have a look at the GDB sources to confirm
454       // this. No problem appear experimentally.
455       //
456       // What happens for static binaries? They don't have an .interp section,
457       // and don't have a r_debug variable on Android, so GDB should not be
458       // able to debug shared libraries at all for them (assuming one
459       // statically links a linker into the executable).
460       if (!r_debug_->r_map || !r_debug_->r_map->l_next ||
461           !r_debug_->r_map->l_next->l_next) {
462         // Sanity check: Must have at least two items in the list.
463         LOG("%s: Malformed r_debug.r_map list\n", __FUNCTION__);
464         r_debug_ = NULL;
465         return;
466       }
467     
468       link_map_t* before = r_debug_->r_map->l_next;
469       link_map_t* after = before->l_next;
470     
471       // Prepare the new entry.
472       entry->l_prev = before;
473       entry->l_next = after;
474     
475       // IMPORTANT: Before modifying the previous and next entries in the
476       // list, ensure that they are writable. This avoids crashing when
477       // updating the 'l_prev' or 'l_next' fields of a system linker entry,
478       // which are mapped read-only.
479       WriteLinkMapField(&before->l_next, entry);
480       WriteLinkMapField(&after->l_prev, entry);
481     
482       // Tell GDB that the list modification has completed.
483       r_debug_->r_state = RT_CONSISTENT;
484       r_debug_->r_brk();
485     }
486     
487     void RDebug::DelEntryImpl(link_map_t* entry) {
488       ScopedGlobalLock lock;
489       LOG("%s: Deleting: %s\n", __FUNCTION__, entry->l_name);
490       if (!r_debug_)
491         return;
492     
493       // Tell GDB the list is going to be modified.
494       r_debug_->r_state = RT_DELETE;
495       r_debug_->r_brk();
496     
497       // IMPORTANT: Before modifying the previous and next entries in the
498       // list, ensure that they are writable. See comment above for more
499       // details.
500       if (entry->l_prev)
501         WriteLinkMapField(&entry->l_prev->l_next, entry->l_next);
502       if (entry->l_next)
503         WriteLinkMapField(&entry->l_next->l_prev, entry->l_prev);
504     
505       if (r_debug_->r_map == entry)
506         r_debug_->r_map = entry->l_next;
507     
508       entry->l_prev = NULL;
509       entry->l_next = NULL;
510     
511       // Tell GDB the list modification has completed.
512       r_debug_->r_state = RT_CONSISTENT;
513       r_debug_->r_brk();
514     }
