36     // Passing a different NPP struct that has the same ndata value as the one passed to NPP_New should
37     // not trigger an assertion failure.
38     
39     class PassDifferentNPPStruct : public PluginTest {
40     public:
41         PassDifferentNPPStruct(NPP npp, const string& identifier)
42             : PluginTest(npp, identifier)
43             , m_didReceiveInitialSetWindowCall(false)
44         {
45         }
46     
47     private:
48      NPError NPP_SetWindow(NPWindow* window) override {
49             if (m_didReceiveInitialSetWindowCall)
50                 return NPERR_NO_ERROR;
51             m_didReceiveInitialSetWindowCall = true;
52     
53             NPP oldNPP = m_npp;
54             NPP_t differentNPP = *m_npp;
55             m_npp = &differentNPP;
56     
57             NPBool privateMode;
58             NPError error = NPN_GetValue(NPNVprivateModeBool, &privateMode);
59     
60             m_npp = oldNPP;
61     
62             if (error != NPERR_NO_ERROR) {
63                 log("NPN_GetValue(NPNVprivateModeBool) with a different NPP struct failed with error %d", error);
64                 notifyDone();
65                 return NPERR_GENERIC_ERROR;
66             }
67             log("NPN_GetValue(NPNVprivateModeBool) with a different NPP struct succeeded");
68             notifyDone();
69             return NPERR_NO_ERROR;
70         }
71     
72         bool m_didReceiveInitialSetWindowCall;
73     };
74
