18     // All offsets are given in bytes relative to the start of the header.
19     // Arithmetic is used to indicate the size of small fields that are skipped.
20     
21     // http://www.pkware.com/documents/casestudies/APPNOTE.TXT Section 4.3.16
22     // This marker appears at the start of the end of central directory record
23     const uint32_t kEndOfCentralDirectoryMarker = 0x06054b50;
24     
25     // Offsets of fields in End of Central Directory.
26     const int kOffsetNumOfEntriesInEndOfCentralDirectory = 4 + 2 + 2;
27     const int kOffsetOfCentralDirLengthInEndOfCentralDirectory =
28         kOffsetNumOfEntriesInEndOfCentralDirectory + 2 + 2;
29     const int kOffsetOfStartOfCentralDirInEndOfCentralDirectory =
30         kOffsetOfCentralDirLengthInEndOfCentralDirectory + 4;
31     
32     // http://www.pkware.com/documents/casestudies/APPNOTE.TXT Section 4.3.12
33     // This marker appears at the start of the central directory
34     const uint32_t kCentralDirHeaderMarker = 0x2014b50;
35     
36     // Offsets of fields in Central Directory Header.
37     const int kOffsetFilenameLengthInCentralDirectory =
38         4 + 2 + 2 + 2 + 2 + 2 + 2 + 4 + 4 + 4;
39     const int kOffsetExtraFieldLengthInCentralDirectory =
40         kOffsetFilenameLengthInCentralDirectory + 2;
41     const int kOffsetCommentLengthInCentralDirectory =
42         kOffsetExtraFieldLengthInCentralDirectory + 2;
43     const int kOffsetLocalHeaderOffsetInCentralDirectory =
44         kOffsetCommentLengthInCentralDirectory + 2 + 2 + 2 + 4;
45     const int kOffsetFilenameInCentralDirectory =
46         kOffsetLocalHeaderOffsetInCentralDirectory + 4;
47     
48     // http://www.pkware.com/documents/casestudies/APPNOTE.TXT Section 4.3.7
49     // This marker appears at the start of local header
50     const uint32_t kLocalHeaderMarker = 0x04034b50;
51     
52     // http://www.pkware.com/documents/casestudies/APPNOTE.TXT Section 4.4.5
53     // This value denotes that the file is stored (no compression).
54     const uint32_t kCompressionMethodStored = 0;
55     
56     // Offsets of fields in the Local Header.
57     const int kOffsetCompressionMethodInLocalHeader = 4 + 2 + 2;
58     const int kOffsetFilenameLengthInLocalHeader =
59         kOffsetCompressionMethodInLocalHeader + 2 + 2 + 2 + 4 + 4 + 4;
60     const int kOffsetExtraFieldLengthInLocalHeader =
61         kOffsetFilenameLengthInLocalHeader + 2;
62     const int kOffsetFilenameInLocalHeader =
63         kOffsetExtraFieldLengthInLocalHeader + 2;

65     // RAII pattern for unmapping and closing the mapped file.
66     class ScopedMMap {
67      public:
68       ScopedMMap(void* mem, uint32_t len) : mem_(mem), len_(len) {}
69       ~ScopedMMap() {
70         if (munmap(mem_, len_) == -1) {
71           LOG_ERRNO("%s: munmap failed when trying to unmap zip file\n",
72                     __FUNCTION__);
73         }
74       }
75      private:
76       void* mem_;
77       uint32_t len_;
78     };

80     inline uint32_t ReadUInt16(uint8_t* mem_bytes, int offset) {
81       return
82           static_cast<uint32_t>(mem_bytes[offset]) |
83           (static_cast<uint32_t>(mem_bytes[offset + 1]) << 8);
84     }
85     
86     inline uint32_t ReadUInt32(uint8_t* mem_bytes, int offset) {
87       return
88           static_cast<uint32_t>(mem_bytes[offset]) |
89           (static_cast<uint32_t>(mem_bytes[offset + 1]) << 8) |
90           (static_cast<uint32_t>(mem_bytes[offset + 2]) << 16) |
91           (static_cast<uint32_t>(mem_bytes[offset + 3]) << 24);
92     }

98     const uint32_t kMaxZipFileLength = 1U << 31;  // 2GB
99     
100     int FindStartOffsetOfFileInZipFile(const char* zip_file, const char* filename) {
101       // Open the file
102       FileDescriptor fd;
103       if (!fd.OpenReadOnly(zip_file)) {
104         LOG_ERRNO("%s: open failed trying to open zip file %s\n",
105                   __FUNCTION__, zip_file);
106         return CRAZY_OFFSET_FAILED;
107       }
108     
109       // Find the length of the file.
110       struct stat stat_buf;
111       if (stat(zip_file, &stat_buf) == -1) {
112         LOG_ERRNO("%s: stat failed trying to stat zip file %s\n",
113                   __FUNCTION__, zip_file);
114         return CRAZY_OFFSET_FAILED;
115       }
116     
117       if (stat_buf.st_size > kMaxZipFileLength) {
118         LOG("%s: The size %ld of %s is too large to map\n",
119             __FUNCTION__, stat_buf.st_size, zip_file);
120         return CRAZY_OFFSET_FAILED;
121       }
122     
123       // Map the file into memory.
124       void* mem = fd.Map(NULL, stat_buf.st_size, PROT_READ, MAP_PRIVATE, 0);
125       if (mem == MAP_FAILED) {
126         LOG_ERRNO("%s: mmap failed trying to mmap zip file %s\n",
127                   __FUNCTION__, zip_file);
128         return CRAZY_OFFSET_FAILED;
129       }
130       ScopedMMap scoped_mmap(mem, stat_buf.st_size);
131     
132       // Scan backwards from the end of the file searching for the end of
133       // central directory marker.
134       uint8_t* mem_bytes = static_cast<uint8_t*>(mem);
135       int off;
136       for (off = stat_buf.st_size - sizeof(kEndOfCentralDirectoryMarker);
137            off >= 0; --off) {
138         if (ReadUInt32(mem_bytes, off) == kEndOfCentralDirectoryMarker) {
139           break;
140         }
141       }
142       if (off == -1) {
143         LOG("%s: Failed to find end of central directory in %s\n",
144             __FUNCTION__, zip_file);
145         return CRAZY_OFFSET_FAILED;
146       }
147     
148       // We have located the end of central directory record, now locate
149       // the central directory by reading the end of central directory record.
150     
151       uint32_t length_of_central_dir = ReadUInt32(
152           mem_bytes, off + kOffsetOfCentralDirLengthInEndOfCentralDirectory);
153       uint32_t start_of_central_dir = ReadUInt32(
154           mem_bytes, off + kOffsetOfStartOfCentralDirInEndOfCentralDirectory);
155     
156       if (start_of_central_dir > off) {
157         LOG("%s: Found out of range offset %u for start of directory in %s\n",
158             __FUNCTION__, start_of_central_dir, zip_file);
159         return CRAZY_OFFSET_FAILED;
160       }
161     
162       uint32_t end_of_central_dir = start_of_central_dir + length_of_central_dir;
163       if (end_of_central_dir > off) {
164         LOG("%s: Found out of range offset %u for end of directory in %s\n",
165             __FUNCTION__, end_of_central_dir, zip_file);
166         return CRAZY_OFFSET_FAILED;
167       }
168     
169       uint32_t num_entries = ReadUInt16(
170           mem_bytes, off + kOffsetNumOfEntriesInEndOfCentralDirectory);
171     
172       // Read the headers in the central directory and locate the file.
173       off = start_of_central_dir;
174       const int target_len = strlen(filename);
175       int n = 0;
176       for (; n < num_entries && off < end_of_central_dir; ++n) {
177         uint32_t marker = ReadUInt32(mem_bytes, off);
178         if (marker != kCentralDirHeaderMarker) {
179           LOG("%s: Failed to find central directory header marker in %s. "
180               "Found 0x%x but expected 0x%x\n", __FUNCTION__,
181               zip_file, marker, kCentralDirHeaderMarker);
182           return CRAZY_OFFSET_FAILED;
183         }
184         uint32_t file_name_length =
185             ReadUInt16(mem_bytes, off + kOffsetFilenameLengthInCentralDirectory);
186         uint32_t extra_field_length =
187             ReadUInt16(mem_bytes, off + kOffsetExtraFieldLengthInCentralDirectory);
188         uint32_t comment_field_length =
189             ReadUInt16(mem_bytes, off + kOffsetCommentLengthInCentralDirectory);
190         uint32_t header_length = kOffsetFilenameInCentralDirectory +
191             file_name_length + extra_field_length + comment_field_length;
192     
193         uint32_t local_header_offset =
194             ReadUInt32(mem_bytes, off + kOffsetLocalHeaderOffsetInCentralDirectory);
195     
196         uint8_t* filename_bytes =
197             mem_bytes + off + kOffsetFilenameInCentralDirectory;
198     
199         if (file_name_length == target_len &&
200             memcmp(filename_bytes, filename, target_len) == 0) {
201           // Filename matches. Read the local header and compute the offset.
202           uint32_t marker = ReadUInt32(mem_bytes, local_header_offset);
203           if (marker != kLocalHeaderMarker) {
204             LOG("%s: Failed to find local file header marker in %s. "
205                 "Found 0x%x but expected 0x%x\n", __FUNCTION__,
206                 zip_file, marker, kLocalHeaderMarker);
207             return CRAZY_OFFSET_FAILED;
208           }
209     
210           uint32_t compression_method =
211               ReadUInt16(
212                   mem_bytes,
213                   local_header_offset + kOffsetCompressionMethodInLocalHeader);
214           if (compression_method != kCompressionMethodStored) {
215             LOG("%s: %s is compressed within %s. "
216                 "Found compression method %u but expected %u\n", __FUNCTION__,
217                 filename, zip_file, compression_method, kCompressionMethodStored);
218             return CRAZY_OFFSET_FAILED;
219           }
220     
221           uint32_t file_name_length =
222               ReadUInt16(
223                   mem_bytes,
224                   local_header_offset + kOffsetFilenameLengthInLocalHeader);
225           uint32_t extra_field_length =
226               ReadUInt16(
227                   mem_bytes,
228                   local_header_offset + kOffsetExtraFieldLengthInLocalHeader);
229           uint32_t header_length =
230               kOffsetFilenameInLocalHeader + file_name_length + extra_field_length;
231     
232           return local_header_offset + header_length;
233         }
234     
235         off += header_length;
236       }
237     
238       if (n < num_entries) {
239         LOG("%s: Did not find all the expected entries in the central directory. "
240             "Found %d but expected %d\n", __FUNCTION__, n, num_entries);
241       }
242     
243       if (off < end_of_central_dir) {
244         LOG("%s: There are %d extra bytes at the end of the central directory.\n",
245             __FUNCTION__, end_of_central_dir - off);
246       }
247     
248       LOG("%s: Did not find %s in %s\n", __FUNCTION__, filename, zip_file);
249       return CRAZY_OFFSET_FAILED;
250     }
