309     static int swizzle_for_size(int size)
310     {
311        int size_swizzles[4] = {
312           MAKE_SWIZZLE4(SWIZZLE_X, SWIZZLE_X, SWIZZLE_X, SWIZZLE_X),
313           MAKE_SWIZZLE4(SWIZZLE_X, SWIZZLE_Y, SWIZZLE_Y, SWIZZLE_Y),
314           MAKE_SWIZZLE4(SWIZZLE_X, SWIZZLE_Y, SWIZZLE_Z, SWIZZLE_Z),
315           MAKE_SWIZZLE4(SWIZZLE_X, SWIZZLE_Y, SWIZZLE_Z, SWIZZLE_W),
316        };
317     
318        return size_swizzles[size - 1];
319     }

321     ir_to_mesa_instruction *
322     ir_to_mesa_visitor::ir_to_mesa_emit_op3(ir_instruction *ir,
323     					enum prog_opcode op,
324     					ir_to_mesa_dst_reg dst,
325     					ir_to_mesa_src_reg src0,
326     					ir_to_mesa_src_reg src1,
327     					ir_to_mesa_src_reg src2)
328     {
329        ir_to_mesa_instruction *inst = new(mem_ctx) ir_to_mesa_instruction();
330        int num_reladdr = 0;
331     
332        /* If we have to do relative addressing, we want to load the ARL
333         * reg directly for one of the regs, and preload the other reladdr
334         * sources into temps.
335         */
336        num_reladdr += dst.reladdr != NULL;
337        num_reladdr += src0.reladdr != NULL;
338        num_reladdr += src1.reladdr != NULL;
339        num_reladdr += src2.reladdr != NULL;
340     
341        reladdr_to_temp(ir, &src2, &num_reladdr);
342        reladdr_to_temp(ir, &src1, &num_reladdr);
343        reladdr_to_temp(ir, &src0, &num_reladdr);
344     
345        if (dst.reladdr) {
346           ir_to_mesa_emit_op1(ir, OPCODE_ARL, ir_to_mesa_address_reg,
347                               *dst.reladdr);
348     
349           num_reladdr--;
350        }
351        assert(num_reladdr == 0);
352     
353        inst->op = op;
354        inst->dst_reg = dst;
355        inst->src_reg[0] = src0;
356        inst->src_reg[1] = src1;
357        inst->src_reg[2] = src2;
358        inst->ir = ir;
359     
360        inst->function = NULL;
361     
362        this->instructions.push_tail(inst);
363     
364        return inst;
365     }

419     /**
420      * Emits Mesa scalar opcodes to produce unique answers across channels.
421      *
422      * Some Mesa opcodes are scalar-only, like ARB_fp/vp.  The src X
423      * channel determines the result across all channels.  So to do a vec4
424      * of this operation, we want to emit a scalar per source channel used
425      * to produce dest channels.
426      */
427     void
428     ir_to_mesa_visitor::ir_to_mesa_emit_scalar_op2(ir_instruction *ir,
429     					       enum prog_opcode op,
430     					       ir_to_mesa_dst_reg dst,
431     					       ir_to_mesa_src_reg orig_src0,
432     					       ir_to_mesa_src_reg orig_src1)
433     {
434        int i, j;
435        int done_mask = ~dst.writemask;
436     
437        /* Mesa RCP is a scalar operation splatting results to all channels,
438         * like ARB_fp/vp.  So emit as many RCPs as necessary to cover our
439         * dst channels.
440         */
441        for (i = 0; i < 4; i++) {
442           GLuint this_mask = (1 << i);
443           ir_to_mesa_instruction *inst;
444           ir_to_mesa_src_reg src0 = orig_src0;
445           ir_to_mesa_src_reg src1 = orig_src1;
446     
447           if (done_mask & this_mask)
448     	 continue;
449     
450           GLuint src0_swiz = GET_SWZ(src0.swizzle, i);
451           GLuint src1_swiz = GET_SWZ(src1.swizzle, i);
452           for (j = i + 1; j < 4; j++) {
453     	 if (!(done_mask & (1 << j)) &&
454     	     GET_SWZ(src0.swizzle, j) == src0_swiz &&
455     	     GET_SWZ(src1.swizzle, j) == src1_swiz) {
456     	    this_mask |= (1 << j);
457     	 }
458           }
459           src0.swizzle = MAKE_SWIZZLE4(src0_swiz, src0_swiz,
460     				   src0_swiz, src0_swiz);
461           src1.swizzle = MAKE_SWIZZLE4(src1_swiz, src1_swiz,
462     				  src1_swiz, src1_swiz);
463     
464           inst = ir_to_mesa_emit_op2(ir, op,
465     				 dst,
466     				 src0,
467     				 src1);
468           inst->dst_reg.writemask = this_mask;
469           done_mask |= this_mask;
470        }
471     }

497     static int
498     type_size(const struct glsl_type *type)
499     {
500        unsigned int i;
501        int size;
502     
503        switch (type->base_type) {
504        case GLSL_TYPE_UINT:
505        case GLSL_TYPE_INT:
506        case GLSL_TYPE_FLOAT:
507        case GLSL_TYPE_BOOL:
508           if (type->is_matrix()) {
509     	 return type->matrix_columns;
510           } else {
511     	 /* Regardless of size of vector, it gets a vec4. This is bad
512     	  * packing for things like floats, but otherwise arrays become a
513     	  * mess.  Hopefully a later pass over the code can pack scalars
514     	  * down if appropriate.
515     	  */
516     	 return 1;
517           }
518        case GLSL_TYPE_ARRAY:
519           return type_size(type->fields.array) * type->length;
520        case GLSL_TYPE_STRUCT:
521           size = 0;
522           for (i = 0; i < type->length; i++) {
523     	 size += type_size(type->fields.structure[i].type);
524           }
525           return size;
526        case GLSL_TYPE_SAMPLER:
527           /* Samplers take up one slot in UNIFORMS[], but they're baked in
528            * at link time.
529            */
530           return 1;
531        default:
532           assert(0);
533           return 0;
534        }
535     }

537     /**
538      * In the initial pass of codegen, we assign temporary numbers to
539      * intermediate results.  (not SSA -- variable assignments will reuse
540      * storage).  Actual register allocation for the Mesa VM occurs in a
541      * pass over the Mesa IR later.
542      */
543     ir_to_mesa_src_reg
544     ir_to_mesa_visitor::get_temp(const glsl_type *type)
545     {
546        ir_to_mesa_src_reg src_reg;
547        int swizzle[4];
548        int i;
549     
550        src_reg.file = PROGRAM_TEMPORARY;
551        src_reg.index = next_temp;
552        src_reg.reladdr = NULL;
553        next_temp += type_size(type);
554     
555        if (type->is_array() || type->is_record()) {
556           src_reg.swizzle = SWIZZLE_NOOP;
557        } else {
558           for (i = 0; i < type->vector_elements; i++)
559     	 swizzle[i] = i;
560           for (; i < 4; i++)
561     	 swizzle[i] = type->vector_elements - 1;
562           src_reg.swizzle = MAKE_SWIZZLE4(swizzle[0], swizzle[1],
563     				      swizzle[2], swizzle[3]);
564        }
565        src_reg.negate = 0;
566     
567        return src_reg;
568     }

823     void
824     ir_to_mesa_visitor::visit(ir_variable *ir)
825     {
826        if (strcmp(ir->name, "gl_FragCoord") == 0) {
827           struct gl_fragment_program *fp = (struct gl_fragment_program *)this->prog;
828     
829           fp->OriginUpperLeft = ir->origin_upper_left;
830           fp->PixelCenterInteger = ir->pixel_center_integer;
831        }
832     
833        if (ir->mode == ir_var_uniform && strncmp(ir->name, "gl_", 3) == 0) {
834           unsigned int i;
835     
836           for (i = 0; i < Elements(statevars); i++) {
837     	 if (strcmp(ir->name, statevars[i].name) == 0)
838     	    break;
839           }
840     
841           if (i == Elements(statevars)) {
842     	 fail_link(this->shader_program,
843     		   "Failed to find builtin uniform `%s'\n", ir->name);
844     	 return;
845           }
846     
847           const struct statevar *statevar = &statevars[i];
848     
849           int array_count;
850           if (ir->type->is_array()) {
851     	 array_count = ir->type->length;
852           } else {
853     	 array_count = 1;
854           }
855     
856           /* Check if this statevar's setup in the STATE file exactly
857            * matches how we'll want to reference it as a
858            * struct/array/whatever.  If not, then we need to move it into
859            * temporary storage and hope that it'll get copy-propagated
860            * out.
861            */
862           for (i = 0; i < statevar->num_elements; i++) {
863     	 if (statevar->elements[i].swizzle != SWIZZLE_XYZW) {
864     	    break;
865     	 }
866           }
867     
868           struct variable_storage *storage;
869           ir_to_mesa_dst_reg dst;
870           if (i == statevar->num_elements) {
871     	 /* We'll set the index later. */
872     	 storage = new(mem_ctx) variable_storage(ir, PROGRAM_STATE_VAR, -1);
873     	 this->variables.push_tail(storage);
874     
875     	 dst = ir_to_mesa_undef_dst;
876           } else {
877     	 storage = new(mem_ctx) variable_storage(ir, PROGRAM_TEMPORARY,
878     						 this->next_temp);
879     	 this->variables.push_tail(storage);
880     	 this->next_temp += type_size(ir->type);
881     
882     	 dst = ir_to_mesa_dst_reg_from_src(ir_to_mesa_src_reg(PROGRAM_TEMPORARY,
883     							      storage->index,
884     							      NULL));
885           }
886     
887     
888           for (int a = 0; a < array_count; a++) {
889     	 for (unsigned int i = 0; i < statevar->num_elements; i++) {
890     	    struct statevar_element *element = &statevar->elements[i];
891     	    int tokens[STATE_LENGTH];
892     
893     	    memcpy(tokens, element->tokens, sizeof(element->tokens));
894     	    if (ir->type->is_array()) {
895     	       tokens[1] = a;
896     	    }
897     
898     	    int index = _mesa_add_state_reference(this->prog->Parameters,
899     						  (gl_state_index *)tokens);
900     
901     	    if (storage->file == PROGRAM_STATE_VAR) {
902     	       if (storage->index == -1) {
903     		  storage->index = index;
904     	       } else {
905     		  assert(index ==
906                              (int)(storage->index + a * statevar->num_elements + i));
907     	       }
908     	    } else {
909     	       ir_to_mesa_src_reg src(PROGRAM_STATE_VAR, index, NULL);
910     	       src.swizzle = element->swizzle;
911     	       ir_to_mesa_emit_op1(ir, OPCODE_MOV, dst, src);
912     	       /* even a float takes up a whole vec4 reg in a struct/array. */
913     	       dst.index++;
914     	    }
915     	 }
916           }
917           if (storage->file == PROGRAM_TEMPORARY &&
918     	  dst.index != storage->index + type_size(ir->type)) {
919     	 fail_link(this->shader_program,
920     		   "failed to load builtin uniform `%s'  (%d/%d regs loaded)\n",
921     		   ir->name, dst.index - storage->index,
922     		   type_size(ir->type));
923           }
924        }
925     }

927     void
928     ir_to_mesa_visitor::visit(ir_loop *ir)
929     {
930        ir_dereference_variable *counter = NULL;
931     
932        if (ir->counter != NULL)
933           counter = new(ir) ir_dereference_variable(ir->counter);
934     
935        if (ir->from != NULL) {
936           assert(ir->counter != NULL);
937     
938           ir_assignment *a = new(ir) ir_assignment(counter, ir->from, NULL);
939     
940           a->accept(this);
941           delete a;
942        }
943     
944        ir_to_mesa_emit_op0(NULL, OPCODE_BGNLOOP);
945     
946        if (ir->to) {
947           ir_expression *e =
948     	 new(ir) ir_expression(ir->cmp, glsl_type::bool_type,
949     			       counter, ir->to);
950           ir_if *if_stmt =  new(ir) ir_if(e);
951     
952           ir_loop_jump *brk = new(ir) ir_loop_jump(ir_loop_jump::jump_break);
953     
954           if_stmt->then_instructions.push_tail(brk);
955     
956           if_stmt->accept(this);
957     
958           delete if_stmt;
959           delete e;
960           delete brk;
961        }
962     
963        visit_exec_list(&ir->body_instructions, this);
964     
965        if (ir->increment) {
966           ir_expression *e =
967     	 new(ir) ir_expression(ir_binop_add, counter->type,
968     			       counter, ir->increment);
969     
970           ir_assignment *a = new(ir) ir_assignment(counter, e, NULL);
971     
972           a->accept(this);
973           delete a;
974           delete e;
975        }
976     
977        ir_to_mesa_emit_op0(NULL, OPCODE_ENDLOOP);
978     }

1023     GLboolean
1024     ir_to_mesa_visitor::try_emit_mad(ir_expression *ir, int mul_operand)
1025     {
1026        int nonmul_operand = 1 - mul_operand;
1027        ir_to_mesa_src_reg a, b, c;
1028     
1029        ir_expression *expr = ir->operands[mul_operand]->as_expression();
1030        if (!expr || expr->operation != ir_binop_mul)
1031           return false;
1032     
1033        expr->operands[0]->accept(this);
1034        a = this->result;
1035        expr->operands[1]->accept(this);
1036        b = this->result;
1037        ir->operands[nonmul_operand]->accept(this);
1038        c = this->result;
1039     
1040        this->result = get_temp(ir->type);
1041        ir_to_mesa_emit_op3(ir, OPCODE_MAD,
1042     		       ir_to_mesa_dst_reg_from_src(this->result), a, b, c);
1043     
1044        return true;
1045     }

1047     void
1048     ir_to_mesa_visitor::reladdr_to_temp(ir_instruction *ir,
1049     				    ir_to_mesa_src_reg *reg, int *num_reladdr)
1050     {
1051        if (!reg->reladdr)
1052           return;
1053     
1054        ir_to_mesa_emit_op1(ir, OPCODE_ARL, ir_to_mesa_address_reg, *reg->reladdr);
1055     
1056        if (*num_reladdr != 1) {
1057           ir_to_mesa_src_reg temp = get_temp(glsl_type::vec4_type);
1058     
1059           ir_to_mesa_emit_op1(ir, OPCODE_MOV,
1060     			  ir_to_mesa_dst_reg_from_src(temp), *reg);
1061           *reg = temp;
1062        }
1063     
1064        (*num_reladdr)--;
1065     }

1067     void
1068     ir_to_mesa_visitor::visit(ir_expression *ir)
1069     {
1070        unsigned int operand;
1071        struct ir_to_mesa_src_reg op[2];
1072        struct ir_to_mesa_src_reg result_src;
1073        struct ir_to_mesa_dst_reg result_dst;
1074        const glsl_type *vec4_type = glsl_type::get_instance(GLSL_TYPE_FLOAT, 4, 1);
1075        const glsl_type *vec3_type = glsl_type::get_instance(GLSL_TYPE_FLOAT, 3, 1);
1076        const glsl_type *vec2_type = glsl_type::get_instance(GLSL_TYPE_FLOAT, 2, 1);
1077     
1078        /* Quick peephole: Emit OPCODE_MAD(a, b, c) instead of ADD(MUL(a, b), c)
1079         */
1080        if (ir->operation == ir_binop_add) {
1081           if (try_emit_mad(ir, 1))
1082     	 return;
1083           if (try_emit_mad(ir, 0))
1084     	 return;
1085        }
1086     
1087        for (operand = 0; operand < ir->get_num_operands(); operand++) {
1088           this->result.file = PROGRAM_UNDEFINED;
1089           ir->operands[operand]->accept(this);
1090           if (this->result.file == PROGRAM_UNDEFINED) {
1091     	 ir_print_visitor v;
1092     	 printf("Failed to get tree for expression operand:\n");
1093     	 ir->operands[operand]->accept(&v);
1094     	 exit(1);
1095           }
1096           op[operand] = this->result;
1097     
1098           /* Matrix expression operands should have been broken down to vector
1099            * operations already.
1100            */
1101           assert(!ir->operands[operand]->type->is_matrix());
1102        }
1103     
1104        int vector_elements = ir->operands[0]->type->vector_elements;
1105        if (ir->operands[1]) {
1106           vector_elements = MAX2(vector_elements,
1107     			     ir->operands[1]->type->vector_elements);
1108        }
1109     
1110        this->result.file = PROGRAM_UNDEFINED;
1111     
1112        /* Storage for our result.  Ideally for an assignment we'd be using
1113         * the actual storage for the result here, instead.
1114         */
1115        result_src = get_temp(ir->type);
1116        /* convenience for the emit functions below. */
1117        result_dst = ir_to_mesa_dst_reg_from_src(result_src);
1118        /* Limit writes to the channels that will be used by result_src later.
1119         * This does limit this temp's use as a temporary for multi-instruction
1120         * sequences.
1121         */
1122        result_dst.writemask = (1 << ir->type->vector_elements) - 1;
1123     
1124        switch (ir->operation) {
1125        case ir_unop_logic_not:
1126           ir_to_mesa_emit_op2(ir, OPCODE_SEQ, result_dst,
1127     			  op[0], src_reg_for_float(0.0));
1128           break;
1129        case ir_unop_neg:
1130           op[0].negate = ~op[0].negate;
1131           result_src = op[0];
1132           break;
1133        case ir_unop_abs:
1134           ir_to_mesa_emit_op1(ir, OPCODE_ABS, result_dst, op[0]);
1135           break;
1136        case ir_unop_sign:
1137           ir_to_mesa_emit_op1(ir, OPCODE_SSG, result_dst, op[0]);
1138           break;
1139        case ir_unop_rcp:
1140           ir_to_mesa_emit_scalar_op1(ir, OPCODE_RCP, result_dst, op[0]);
1141           break;
1142     
1143        case ir_unop_exp2:
1144           ir_to_mesa_emit_scalar_op1(ir, OPCODE_EX2, result_dst, op[0]);
1145           break;
1146        case ir_unop_exp:
1147        case ir_unop_log:
1148           assert(!"not reached: should be handled by ir_explog_to_explog2");
1149           break;
1150        case ir_unop_log2:
1151           ir_to_mesa_emit_scalar_op1(ir, OPCODE_LG2, result_dst, op[0]);
1152           break;
1153        case ir_unop_sin:

1380     void
1381     ir_to_mesa_visitor::visit(ir_swizzle *ir)
1382     {
1383        ir_to_mesa_src_reg src_reg;
1384        int i;
1385        int swizzle[4];
1386     
1387        /* Note that this is only swizzles in expressions, not those on the left
1388         * hand side of an assignment, which do write masking.  See ir_assignment
1389         * for that.
1390         */
1391     
1392        ir->val->accept(this);
1393        src_reg = this->result;
1394        assert(src_reg.file != PROGRAM_UNDEFINED);
1395     
1396        for (i = 0; i < 4; i++) {
1397           if (i < ir->type->vector_elements) {
1398     	 switch (i) {
1399     	 case 0:
1400     	    swizzle[i] = GET_SWZ(src_reg.swizzle, ir->mask.x);
1401     	    break;
1402     	 case 1:
1403     	    swizzle[i] = GET_SWZ(src_reg.swizzle, ir->mask.y);
1404     	    break;
1405     	 case 2:
1406     	    swizzle[i] = GET_SWZ(src_reg.swizzle, ir->mask.z);
1407     	    break;
1408     	 case 3:
1409     	    swizzle[i] = GET_SWZ(src_reg.swizzle, ir->mask.w);
1410     	    break;
1411     	 }
1412           } else {
1413     	 /* If the type is smaller than a vec4, replicate the last
1414     	  * channel out.
1415     	  */
1416     	 swizzle[i] = swizzle[ir->type->vector_elements - 1];
1417           }
1418        }
1419     
1420        src_reg.swizzle = MAKE_SWIZZLE4(swizzle[0],
1421     				   swizzle[1],
1422     				   swizzle[2],
1423     				   swizzle[3]);
1424     
1425        this->result = src_reg;
1426     }

1429     ir_to_mesa_visitor::visit(ir_dereference_variable *ir)
1430     {
1431        variable_storage *entry = find_variable_storage(ir->var);
1432     
1433        if (!entry) {
1434           switch (ir->var->mode) {
1435           case ir_var_uniform:
1436     	 entry = new(mem_ctx) variable_storage(ir->var, PROGRAM_UNIFORM,
1437     					       ir->var->location);

1439     	 break;

1446     	  * user-defined varyings.
1447     	  *
1448     	  * FINISHME: We would hit this path for function arguments.  Fix!

1452     	     ir->var->mode == ir_var_inout) {

1466     	    entry = new(mem_ctx) variable_storage(ir->var,
1467     						  PROGRAM_OUTPUT,
1468     						  ir->var->location);
1469     	 }
1470     
1471     	 break;
1472           case ir_var_auto:
1473           case ir_var_temporary:

1491     void
1492     ir_to_mesa_visitor::visit(ir_dereference_array *ir)
1493     {
1494        ir_constant *index;
1495        ir_to_mesa_src_reg src_reg;
1496        int element_size = type_size(ir->type);
1497     
1498        index = ir->array_index->constant_expression_value();
1499     
1500        ir->array->accept(this);
1501        src_reg = this->result;
1502     
1503        if (index) {
1504           src_reg.index += index->value.i[0] * element_size;
1505        } else {
1506           ir_to_mesa_src_reg array_base = this->result;
1507           /* Variable index array dereference.  It eats the "vec4" of the
1508            * base of the array and an index that offsets the Mesa register
1509            * index.
1510            */
1511           ir->array_index->accept(this);
1512     
1513           ir_to_mesa_src_reg index_reg;
1514     
1515           if (element_size == 1) {
1516     	 index_reg = this->result;
1517           } else {
1518     	 index_reg = get_temp(glsl_type::float_type);
1519     
1520     	 ir_to_mesa_emit_op2(ir, OPCODE_MUL,
1521     			     ir_to_mesa_dst_reg_from_src(index_reg),
1522     			     this->result, src_reg_for_float(element_size));
1523           }
1524     
1525           src_reg.reladdr = talloc(mem_ctx, ir_to_mesa_src_reg);
1526           memcpy(src_reg.reladdr, &index_reg, sizeof(index_reg));
1527        }
1528     
1529        /* If the type is smaller than a vec4, replicate the last channel out. */
1530        if (ir->type->is_scalar() || ir->type->is_vector())
1531           src_reg.swizzle = swizzle_for_size(ir->type->vector_elements);
1532        else
1533           src_reg.swizzle = SWIZZLE_NOOP;
1534     
1535        this->result = src_reg;
1536     }

1556     /**
1557      * We want to be careful in assignment setup to hit the actual storage
1558      * instead of potentially using a temporary like we might with the
1559      * ir_dereference handler.
1560      */
1561     static struct ir_to_mesa_dst_reg
1562     get_assignment_lhs(ir_dereference *ir, ir_to_mesa_visitor *v)
1563     {
1564        /* The LHS must be a dereference.  If the LHS is a variable indexed array
1565         * access of a vector, it must be separated into a series conditional moves
1566         * before reaching this point (see ir_vec_index_to_cond_assign).
1567         */
1568        assert(ir->as_dereference());
1569        ir_dereference_array *deref_array = ir->as_dereference_array();
1570        if (deref_array) {
1571           assert(!deref_array->array->type->is_vector());
1572        }
1573     
1574        /* Use the rvalue deref handler for the most part.  We'll ignore
1575         * swizzles in it and write swizzles using writemask, though.
1576         */
1577        ir->accept(v);
1578        return ir_to_mesa_dst_reg_from_src(v->result);
1579     }

1581     void
1582     ir_to_mesa_visitor::visit(ir_assignment *ir)
1583     {
1584        struct ir_to_mesa_dst_reg l;
1585        struct ir_to_mesa_src_reg r;
1586        int i;
1587     
1588        ir->rhs->accept(this);
1589        r = this->result;
1590     
1591        l = get_assignment_lhs(ir->lhs, this);
1592     
1593        /* FINISHME: This should really set to the correct maximal writemask for each
1594         * FINISHME: component written (in the loops below).  This case can only
1595         * FINISHME: occur for matrices, arrays, and structures.
1596         */
1597        if (ir->write_mask == 0) {
1598           assert(!ir->lhs->type->is_scalar() && !ir->lhs->type->is_vector());
1599           l.writemask = WRITEMASK_XYZW;
1600        } else if (ir->lhs->type->is_scalar()) {
1601           /* FINISHME: This hack makes writing to gl_FragDepth, which lives in the
1602            * FINISHME: W component of fragment shader output zero, work correctly.
1603            */
1604           l.writemask = WRITEMASK_XYZW;
1605        } else {
1606           int swizzles[4];
1607           int first_enabled_chan = 0;
1608           int rhs_chan = 0;
1609     
1610           assert(ir->lhs->type->is_vector());
1611           l.writemask = ir->write_mask;
1612     
1613           for (int i = 0; i < 4; i++) {
1614     	 if (l.writemask & (1 << i)) {
1615     	    first_enabled_chan = GET_SWZ(r.swizzle, i);
1616     	    break;
1617     	 }
1618           }
1619     
1620           /* Swizzle a small RHS vector into the channels being written.
1621            *
1622            * glsl ir treats write_mask as dictating how many channels are
1623            * present on the RHS while Mesa IR treats write_mask as just
1624            * showing which channels of the vec4 RHS get written.
1625            */
1626           for (int i = 0; i < 4; i++) {
1627     	 if (l.writemask & (1 << i))
1628     	    swizzles[i] = GET_SWZ(r.swizzle, rhs_chan++);
1629     	 else
1630     	    swizzles[i] = first_enabled_chan;
1631           }
1632           r.swizzle = MAKE_SWIZZLE4(swizzles[0], swizzles[1],
1633     				swizzles[2], swizzles[3]);
1634        }
1635     
1636        assert(l.file != PROGRAM_UNDEFINED);
1637        assert(r.file != PROGRAM_UNDEFINED);
1638     
1639        if (ir->condition) {
1640           ir_to_mesa_src_reg condition;
1641     
1642           ir->condition->accept(this);
1643           condition = this->result;
1644     
1645           /* We use the OPCODE_CMP (a < 0 ? b : c) for conditional moves,
1646            * and the condition we produced is 0.0 or 1.0.  By flipping the
1647            * sign, we can choose which value OPCODE_CMP produces without
1648            * an extra computing the condition.
1649            */
1650           condition.negate = ~condition.negate;
1651           for (i = 0; i < type_size(ir->lhs->type); i++) {
1652     	 ir_to_mesa_emit_op3(ir, OPCODE_CMP, l,
1653     			     condition, r, ir_to_mesa_src_reg_from_dst(l));
1654     	 l.index++;
1655     	 r.index++;
1656           }
1657        } else {
1658           for (i = 0; i < type_size(ir->lhs->type); i++) {
1659     	 ir_to_mesa_emit_op1(ir, OPCODE_MOV, l, r);
1660     	 l.index++;
1661     	 r.index++;
1662           }
1663        }
1664     }

1669     {
1670        ir_to_mesa_src_reg src_reg;
1671        GLfloat stack_vals[4] = { 0 };
1672        GLfloat *values = stack_vals;
1673        unsigned int i;
1674     
1675        /* Unfortunately, 4 floats is all we can get into
1676         * _mesa_add_unnamed_constant.  So, make a temp to store an
1677         * aggregate constant and move each constant value into it.  If we
1678         * get lucky, copy propagation will eliminate the extra moves.
1679         */
1680     
1681        if (ir->type->base_type == GLSL_TYPE_STRUCT) {
1682           ir_to_mesa_src_reg temp_base = get_temp(ir->type);
1683           ir_to_mesa_dst_reg temp = ir_to_mesa_dst_reg_from_src(temp_base);
1684     
1685           foreach_iter(exec_list_iterator, iter, ir->components) {
1686     	 ir_constant *field_value = (ir_constant *)iter.get();
1687     	 int size = type_size(field_value->type);
1688     
1689     	 assert(size > 0);
1690     
1691     	 field_value->accept(this);
1692     	 src_reg = this->result;
1693     
1694     	 for (i = 0; i < (unsigned int)size; i++) {
1695     	    ir_to_mesa_emit_op1(ir, OPCODE_MOV, temp, src_reg);
1696     
1697     	    src_reg.index++;
1698     	    temp.index++;
1699     	 }
1700           }
1701           this->result = temp_base;
1702           return;
1703        }
1704     
1705        if (ir->type->is_array()) {
1706           ir_to_mesa_src_reg temp_base = get_temp(ir->type);
1707           ir_to_mesa_dst_reg temp = ir_to_mesa_dst_reg_from_src(temp_base);
1708           int size = type_size(ir->type->fields.array);
1709     
1710           assert(size > 0);
1711     
1712           for (i = 0; i < ir->type->length; i++) {
1713     	 ir->array_elements[i]->accept(this);
1714     	 src_reg = this->result;
1715     	 for (int j = 0; j < size; j++) {
1716     	    ir_to_mesa_emit_op1(ir, OPCODE_MOV, temp, src_reg);
1717     
1718     	    src_reg.index++;
1719     	    temp.index++;
1720     	 }
1721           }
1722           this->result = temp_base;
1723           return;
1724        }
1725     
1726        if (ir->type->is_matrix()) {
1727           ir_to_mesa_src_reg mat = get_temp(ir->type);
1728           ir_to_mesa_dst_reg mat_column = ir_to_mesa_dst_reg_from_src(mat);
1729     
1730           for (i = 0; i < ir->type->matrix_columns; i++) {
1731     	 assert(ir->type->base_type == GLSL_TYPE_FLOAT);
1732     	 values = &ir->value.f[i * ir->type->vector_elements];
1733     
1734     	 src_reg = ir_to_mesa_src_reg(PROGRAM_CONSTANT, -1, NULL);
1735     	 src_reg.index = _mesa_add_unnamed_constant(this->prog->Parameters,
1736     						values,
1737     						ir->type->vector_elements,
1738     						&src_reg.swizzle);
1739     	 ir_to_mesa_emit_op1(ir, OPCODE_MOV, mat_column, src_reg);
1740     
1741     	 mat_column.index++;
1742           }
1743     
1744           this->result = mat;
1745           return;
1746        }
1747     
1748        src_reg.file = PROGRAM_CONSTANT;
1749        switch (ir->type->base_type) {
1750        case GLSL_TYPE_FLOAT:
1751           values = &ir->value.f[0];
1752           break;
1753        case GLSL_TYPE_UINT:
1754           for (i = 0; i < ir->type->vector_elements; i++) {
1755     	 values[i] = ir->value.u[i];
1756           }
1757           break;
1758        case GLSL_TYPE_INT:
1759           for (i = 0; i < ir->type->vector_elements; i++) {
1760     	 values[i] = ir->value.i[i];
1761           }
1762           break;
1763        case GLSL_TYPE_BOOL:
1764           for (i = 0; i < ir->type->vector_elements; i++) {
1765     	 values[i] = ir->value.b[i];
1766           }
1767           break;
1768        default:
1769           assert(!"Non-float/uint/int/bool constant");
1770        }
1771     
1772        this->result = ir_to_mesa_src_reg(PROGRAM_CONSTANT, -1, ir->type);
1773        this->result.index = _mesa_add_unnamed_constant(this->prog->Parameters,
1774     						   values,
1775     						   ir->type->vector_elements,
1776     						   &this->result.swizzle);
1777     }

1821     void
1822     ir_to_mesa_visitor::visit(ir_call *ir)
1823     {
1824        ir_to_mesa_instruction *call_inst;
1825        ir_function_signature *sig = ir->get_callee();
1826        function_entry *entry = get_function_signature(sig);
1827        int i;
1828     
1829        /* Process in parameters. */
1830        exec_list_iterator sig_iter = sig->parameters.iterator();
1831        foreach_iter(exec_list_iterator, iter, *ir) {
1832           ir_rvalue *param_rval = (ir_rvalue *)iter.get();
1833           ir_variable *param = (ir_variable *)sig_iter.get();
1834     
1835           if (param->mode == ir_var_in ||
1836     	  param->mode == ir_var_inout) {
1837     	 variable_storage *storage = find_variable_storage(param);
1838     	 assert(storage);
1839     
1840     	 param_rval->accept(this);
1841     	 ir_to_mesa_src_reg r = this->result;
1842     
1843     	 ir_to_mesa_dst_reg l;
1844     	 l.file = storage->file;
1845     	 l.index = storage->index;
1846     	 l.reladdr = NULL;
1847     	 l.writemask = WRITEMASK_XYZW;
1848     	 l.cond_mask = COND_TR;
1849     
1850     	 for (i = 0; i < type_size(param->type); i++) {
1851     	    ir_to_mesa_emit_op1(ir, OPCODE_MOV, l, r);
1852     	    l.index++;
1853     	    r.index++;
1854     	 }
1855           }
1856     
1857           sig_iter.next();
1858        }
1859        assert(!sig_iter.has_next());
1860     
1861        /* Emit call instruction */
1862        call_inst = ir_to_mesa_emit_op1(ir, OPCODE_CAL,
1863     				   ir_to_mesa_undef_dst, ir_to_mesa_undef);
1864        call_inst->function = entry;
1865     
1866        /* Process out parameters. */
1867        sig_iter = sig->parameters.iterator();
1868        foreach_iter(exec_list_iterator, iter, *ir) {
1869           ir_rvalue *param_rval = (ir_rvalue *)iter.get();
1870           ir_variable *param = (ir_variable *)sig_iter.get();
1871     
1872           if (param->mode == ir_var_out ||
1873     	  param->mode == ir_var_inout) {
1874     	 variable_storage *storage = find_variable_storage(param);
1875     	 assert(storage);
1876     
1877     	 ir_to_mesa_src_reg r;
1878     	 r.file = storage->file;
1879     	 r.index = storage->index;
1880     	 r.reladdr = NULL;
1881     	 r.swizzle = SWIZZLE_NOOP;
1882     	 r.negate = 0;
1883     
1884     	 param_rval->accept(this);
1885     	 ir_to_mesa_dst_reg l = ir_to_mesa_dst_reg_from_src(this->result);
1886     
1887     	 for (i = 0; i < type_size(param->type); i++) {
1888     	    ir_to_mesa_emit_op1(ir, OPCODE_MOV, l, r);
1889     	    l.index++;
1890     	    r.index++;
1891     	 }
1892           }
1893     
1894           sig_iter.next();
1895        }
1896        assert(!sig_iter.has_next());
1897     
1898        /* Process return value. */
1899        this->result = entry->return_reg;
1900     }

1985     void
1986     ir_to_mesa_visitor::visit(ir_texture *ir)
1987     {
1988        ir_to_mesa_src_reg result_src, coord, lod_info, projector;
1989        ir_to_mesa_dst_reg result_dst, coord_dst;
1990        ir_to_mesa_instruction *inst = NULL;
1991        prog_opcode opcode = OPCODE_NOP;
1992     
1993        ir->coordinate->accept(this);
1994     
1995        /* Put our coords in a temp.  We'll need to modify them for shadow,
1996         * projection, or LOD, so the only case we'd use it as is is if
1997         * we're doing plain old texturing.  Mesa IR optimization should
1998         * handle cleaning up our mess in that case.
1999         */
2000        coord = get_temp(glsl_type::vec4_type);
2001        coord_dst = ir_to_mesa_dst_reg_from_src(coord);
2002        ir_to_mesa_emit_op1(ir, OPCODE_MOV, coord_dst,
2003     		       this->result);
2004     
2005        if (ir->projector) {
2006           ir->projector->accept(this);
2007           projector = this->result;
2008        }
2009     
2010        /* Storage for our result.  Ideally for an assignment we'd be using
2011         * the actual storage for the result here, instead.
2012         */
2013        result_src = get_temp(glsl_type::vec4_type);
2014        result_dst = ir_to_mesa_dst_reg_from_src(result_src);
2015     
2016        switch (ir->op) {
2017        case ir_tex:
2018           opcode = OPCODE_TEX;
2019           break;
2020        case ir_txb:
2021           opcode = OPCODE_TXB;
2022           ir->lod_info.bias->accept(this);
2023           lod_info = this->result;
2024           break;
2025        case ir_txl:
2026           opcode = OPCODE_TXL;
2027           ir->lod_info.lod->accept(this);
2028           lod_info = this->result;
2029           break;
2030        case ir_txd:
2031        case ir_txf:
2032           assert(!"GLSL 1.30 features unsupported");
2033           break;
2034        }
2035     
2036        if (ir->projector) {
2037           if (opcode == OPCODE_TEX) {
2038     	 /* Slot the projector in as the last component of the coord. */
2039     	 coord_dst.writemask = WRITEMASK_W;
2040     	 ir_to_mesa_emit_op1(ir, OPCODE_MOV, coord_dst, projector);
2041     	 coord_dst.writemask = WRITEMASK_XYZW;
2042     	 opcode = OPCODE_TXP;
2043           } else {
2044     	 ir_to_mesa_src_reg coord_w = coord;
2045     	 coord_w.swizzle = SWIZZLE_WWWW;
2046     
2047     	 /* For the other TEX opcodes there's no projective version
2048     	  * since the last slot is taken up by lod info.  Do the
2049     	  * projective divide now.
2050     	  */
2051     	 coord_dst.writemask = WRITEMASK_W;
2052     	 ir_to_mesa_emit_op1(ir, OPCODE_RCP, coord_dst, projector);
2053     
2054     	 coord_dst.writemask = WRITEMASK_XYZ;
2055     	 ir_to_mesa_emit_op2(ir, OPCODE_MUL, coord_dst, coord, coord_w);
2056     
2057     	 coord_dst.writemask = WRITEMASK_XYZW;
2058     	 coord.swizzle = SWIZZLE_XYZW;
2059           }
2060        }
2061     
2062        if (ir->shadow_comparitor) {
2063           /* Slot the shadow value in as the second to last component of the
2064            * coord.
2065            */
2066           ir->shadow_comparitor->accept(this);
2067           coord_dst.writemask = WRITEMASK_Z;
2068           ir_to_mesa_emit_op1(ir, OPCODE_MOV, coord_dst, this->result);
2069           coord_dst.writemask = WRITEMASK_XYZW;
2070        }
2071     
2072        if (opcode == OPCODE_TXL || opcode == OPCODE_TXB) {
2073           /* Mesa IR stores lod or lod bias in the last channel of the coords. */
2074           coord_dst.writemask = WRITEMASK_W;
2075           ir_to_mesa_emit_op1(ir, OPCODE_MOV, coord_dst, lod_info);
2076           coord_dst.writemask = WRITEMASK_XYZW;
2077        }
2078     
2079        inst = ir_to_mesa_emit_op1(ir, opcode, result_dst, coord);
2080     
2081        if (ir->shadow_comparitor)
2082           inst->tex_shadow = GL_TRUE;
2083     
2084        inst->sampler = get_sampler_uniform_value(ir->sampler);
2085     
2086        const glsl_type *sampler_type = ir->sampler->type;
2087     
2088        switch (sampler_type->sampler_dimensionality) {
2089        case GLSL_SAMPLER_DIM_1D:
2090           inst->tex_target = (sampler_type->sampler_array)
2091     	 ? TEXTURE_1D_ARRAY_INDEX : TEXTURE_1D_INDEX;
2092           break;
2093        case GLSL_SAMPLER_DIM_2D:
2094           inst->tex_target = (sampler_type->sampler_array)
2095     	 ? TEXTURE_2D_ARRAY_INDEX : TEXTURE_2D_INDEX;
2096           break;
2097        case GLSL_SAMPLER_DIM_3D:
2098           inst->tex_target = TEXTURE_3D_INDEX;
2099           break;
2100        case GLSL_SAMPLER_DIM_CUBE:
2101           inst->tex_target = TEXTURE_CUBE_INDEX;
2102           break;
2103        case GLSL_SAMPLER_DIM_RECT:
2104           inst->tex_target = TEXTURE_RECT_INDEX;
2105           break;
2106        case GLSL_SAMPLER_DIM_BUF:
2107           assert(!"FINISHME: Implement ARB_texture_buffer_object");
2108           break;
2109        default:
2110           assert(!"Should not get here.");
2111        }
2112     
2113        this->result = result_src;
2114     }

2232     set_branchtargets(ir_to_mesa_visitor *v,
2233     		  struct prog_instruction *mesa_instructions,

2240     
2241        for (i = 0; i < num_instructions; i++) {
2242           switch (mesa_instructions[i].Opcode) {

2259        loop_stack = talloc_zero_array(v->mem_ctx, int, loop_count);
2260     
2261        for (i = 0; i < num_instructions; i++) {
2262           switch (mesa_instructions[i].Opcode) {
2263           case OPCODE_IF:
2264     	 if_stack[if_stack_pos] = i;
2265     	 if_stack_pos++;
2266     	 break;
2267           case OPCODE_ELSE:
2268     	 mesa_instructions[if_stack[if_stack_pos - 1]].BranchTarget = i;
2269     	 if_stack[if_stack_pos - 1] = i;
2270     	 break;
2271           case OPCODE_ENDIF:
2272     	 mesa_instructions[if_stack[if_stack_pos - 1]].BranchTarget = i;
2273     	 if_stack_pos--;
2274     	 break;
2275           case OPCODE_BGNLOOP:
2276     	 loop_stack[loop_stack_pos] = i;
2277     	 loop_stack_pos++;
2278     	 break;
2279           case OPCODE_ENDLOOP:
2280     	 loop_stack_pos--;
2281     	 /* Rewrite any breaks/conts at this nesting level (haven't
2282     	  * already had a BranchTarget assigned) to point to the end
2283     	  * of the loop.
2284     	  */
2285     	 for (j = loop_stack[loop_stack_pos]; j < i; j++) {
2286     	    if (mesa_instructions[j].Opcode == OPCODE_BRK ||
2287     		mesa_instructions[j].Opcode == OPCODE_CONT) {
2288     	       if (mesa_instructions[j].BranchTarget == -1) {
2289     		  mesa_instructions[j].BranchTarget = i;
2290     	       }
2291     	    }
2292     	 }
2293     	 /* The loop ends point at each other. */
2294     	 mesa_instructions[i].BranchTarget = loop_stack[loop_stack_pos];
2295     	 mesa_instructions[loop_stack[loop_stack_pos]].BranchTarget = i;
2296     	 break;
2297           case OPCODE_CAL:
2298     	 foreach_iter(exec_list_iterator, iter, v->function_signatures) {
2299     	    function_entry *entry = (function_entry *)iter.get();
2300     
2301     	    if (entry->sig_id == mesa_instructions[i].BranchTarget) {
2302     	       mesa_instructions[i].BranchTarget = entry->inst;
2303     	       break;
2304     	    }
2305     	 }
2306     	 break;
2307           default:
2308     	 break;
2309           }
2310        }
2311     }

2370        struct gl_uniform *u;
2371        int pos;
2372     };

2374     /* The shader_program->Uniforms list is almost sorted in increasing
2375      * uniform->{Frag,Vert}Pos locations, but not quite when there are
2376      * uniforms shared between targets.  We need to add parameters in
2377      * increasing order for the targets.
2378      */
2379     static int
2380     sort_uniforms(const void *a, const void *b)
2381     {
2382        struct uniform_sort *u1 = (struct uniform_sort *)a;
2383        struct uniform_sort *u2 = (struct uniform_sort *)b;
2384     
2385        return u1->pos - u2->pos;
2386     }

2388     /* Add the uniforms to the parameters.  The linker chose locations
2389      * in our parameters lists (which weren't created yet), which the
2390      * uniforms code will use to poke values into our parameters list
2391      * when uniforms are updated.
2392      */
2393     static void
2394     add_uniforms_to_parameters_list(struct gl_shader_program *shader_program,
2395     				struct gl_shader *shader,
2396     				struct gl_program *prog)
2397     {
2398        unsigned int i;
2399        unsigned int next_sampler = 0, num_uniforms = 0;
2400        struct uniform_sort *sorted_uniforms;
2401     
2402        sorted_uniforms = talloc_array(NULL, struct uniform_sort,
2403     				  shader_program->Uniforms->NumUniforms);
2404     
2405        for (i = 0; i < shader_program->Uniforms->NumUniforms; i++) {
2406           struct gl_uniform *uniform = shader_program->Uniforms->Uniforms + i;
2407           int parameter_index = -1;
2408     
2409           switch (shader->Type) {
2410           case GL_VERTEX_SHADER:
2411     	 parameter_index = uniform->VertPos;
2412     	 break;
2413           case GL_FRAGMENT_SHADER:
2414     	 parameter_index = uniform->FragPos;
2415     	 break;
2416           case GL_GEOMETRY_SHADER:
2417     	 parameter_index = uniform->GeomPos;
2418     	 break;
2419           }
2420     
2421           /* Only add uniforms used in our target. */
2422           if (parameter_index != -1) {
2423     	 sorted_uniforms[num_uniforms].pos = parameter_index;
2424     	 sorted_uniforms[num_uniforms].u = uniform;
2425     	 num_uniforms++;
2426           }
2427        }
2428     
2429        qsort(sorted_uniforms, num_uniforms, sizeof(struct uniform_sort),
2430     	 sort_uniforms);
2431     
2432        for (i = 0; i < num_uniforms; i++) {
2433           struct gl_uniform *uniform = sorted_uniforms[i].u;
2434           int parameter_index = sorted_uniforms[i].pos;
2435           const glsl_type *type = uniform->Type;
2436           unsigned int size;
2437     
2438           if (type->is_vector() ||
2439     	  type->is_scalar()) {
2440     	 size = type->vector_elements;
2441           } else {
2442     	 size = type_size(type) * 4;
2443           }
2444     
2445           gl_register_file file;
2446           if (type->is_sampler() ||
2447     	  (type->is_array() && type->fields.array->is_sampler())) {
2448     	 file = PROGRAM_SAMPLER;
2449           } else {
2450     	 file = PROGRAM_UNIFORM;
2451           }
2452     
2453           GLint index = _mesa_lookup_parameter_index(prog->Parameters, -1,
2454     						 uniform->Name);
2455     
2456           if (index < 0) {
2457     	 index = _mesa_add_parameter(prog->Parameters, file,
2458     				     uniform->Name, size, type->gl_type,
2459     				     NULL, NULL, 0x0);
2460     
2461     	 /* Sampler uniform values are stored in prog->SamplerUnits,
2462     	  * and the entry in that array is selected by this index we
2463     	  * store in ParameterValues[].
2464     	  */
2465     	 if (file == PROGRAM_SAMPLER) {
2466     	    for (unsigned int j = 0; j < size / 4; j++)
2467     	       prog->Parameters->ParameterValues[index + j][0] = next_sampler++;
2468     	 }
2469     
2470     	 /* The location chosen in the Parameters list here (returned
2471     	  * from _mesa_add_uniform) has to match what the linker chose.
2472     	  */
2473     	 if (index != parameter_index) {
2474     	    fail_link(shader_program, "Allocation of uniform `%s' to target "
2475     		      "failed (%d vs %d)\n",
2476     		      uniform->Name, index, parameter_index);
2477     	 }
2478           }
2479        }
2480     
2481        talloc_free(sorted_uniforms);
2482     }

2484     static void
2485     set_uniform_initializer(GLcontext *ctx, void *mem_ctx,
2486     			struct gl_shader_program *shader_program,
2487     			const char *name, const glsl_type *type,
2488     			ir_constant *val)
2489     {
2490        if (type->is_record()) {
2491           ir_constant *field_constant;
2492     
2493           field_constant = (ir_constant *)val->components.get_head();
2494     
2495           for (unsigned int i = 0; i < type->length; i++) {
2496     	 const glsl_type *field_type = type->fields.structure[i].type;
2497     	 const char *field_name = talloc_asprintf(mem_ctx, "%s.%s", name,
2498     					    type->fields.structure[i].name);
2499     	 set_uniform_initializer(ctx, mem_ctx, shader_program, field_name,
2500     				 field_type, field_constant);
2501     	 field_constant = (ir_constant *)field_constant->next;
2502           }
2503           return;
2504        }
2505     
2506        int loc = _mesa_get_uniform_location(ctx, shader_program, name);
2507     
2508        if (loc == -1) {
2509           fail_link(shader_program,
2510     		"Couldn't find uniform for initializer %s\n", name);
2511           return;
2512        }
2513     
2514        for (unsigned int i = 0; i < (type->is_array() ? type->length : 1); i++) {
2515           ir_constant *element;
2516           const glsl_type *element_type;
2517           if (type->is_array()) {
2518     	 element = val->array_elements[i];
2519     	 element_type = type->fields.array;
2520           } else {
2521     	 element = val;
2522     	 element_type = type;
2523           }
2524     
2525           void *values;
2526     
2527           if (element_type->base_type == GLSL_TYPE_BOOL) {
2528     	 int *conv = talloc_array(mem_ctx, int, element_type->components());
2529     	 for (unsigned int j = 0; j < element_type->components(); j++) {
2530     	    conv[j] = element->value.b[j];
2531     	 }
2532     	 values = (void *)conv;
2533     	 element_type = glsl_type::get_instance(GLSL_TYPE_INT,
2534     						element_type->vector_elements,
2535     						1);
2536           } else {
2537     	 values = &element->value;
2538           }
2539     
2540           if (element_type->is_matrix()) {
2541     	 _mesa_uniform_matrix(ctx, shader_program,
2542     			      element_type->matrix_columns,
2543     			      element_type->vector_elements,
2544     			      loc, 1, GL_FALSE, (GLfloat *)values);
2545     	 loc += element_type->matrix_columns;
2546           } else {
2547     	 _mesa_uniform(ctx, shader_program, loc, element_type->matrix_columns,
2548     		       values, element_type->gl_type);
2549     	 loc += type_size(element_type);
2550           }
2551        }
2552     }

2580     struct gl_program *
2581     get_mesa_program(GLcontext *ctx, struct gl_shader_program *shader_program,
2582     		 struct gl_shader *shader)
2583     {
2584        ir_to_mesa_visitor v;
2585        struct prog_instruction *mesa_instructions, *mesa_inst;
2586        ir_instruction **mesa_instruction_annotation;
2587        int i;
2588        struct gl_program *prog;
2589        GLenum target;
2590        const char *target_string;
2591        GLboolean progress;
2592        struct gl_shader_compiler_options *options =
2593              &ctx->ShaderCompilerOptions[_mesa_shader_type_to_index(shader->Type)];
2594     
2595        switch (shader->Type) {
2596        case GL_VERTEX_SHADER:
2597           target = GL_VERTEX_PROGRAM_ARB;
2598           target_string = "vertex";
2599           break;
2600        case GL_FRAGMENT_SHADER:
2601           target = GL_FRAGMENT_PROGRAM_ARB;
2602           target_string = "fragment";
2603           break;
2604        default:
2605           assert(!"should not be reached");
2606           return NULL;
2607        }
2608     
2609        validate_ir_tree(shader->ir);
2610     
2611        prog = ctx->Driver.NewProgram(ctx, target, shader_program->Name);
2612        if (!prog)
2613           return NULL;
2614        prog->Parameters = _mesa_new_parameter_list();
2615        prog->Varying = _mesa_new_parameter_list();
2616        prog->Attributes = _mesa_new_parameter_list();
2617        v.ctx = ctx;
2618        v.prog = prog;
2619        v.shader_program = shader_program;
2620        v.options = options;
2621     
2622        add_uniforms_to_parameters_list(shader_program, shader, prog);
2623     
2624        /* Emit Mesa IR for main(). */
2625        visit_exec_list(shader->ir, &v);
2626        v.ir_to_mesa_emit_op0(NULL, OPCODE_END);
2627     
2628        /* Now emit bodies for any functions that were used. */
2629        do {
2630           progress = GL_FALSE;
2631     
2632           foreach_iter(exec_list_iterator, iter, v.function_signatures) {
2633     	 function_entry *entry = (function_entry *)iter.get();
2634     
2635     	 if (!entry->bgn_inst) {
2636     	    v.current_function = entry;
2637     
2638     	    entry->bgn_inst = v.ir_to_mesa_emit_op0(NULL, OPCODE_BGNSUB);
2639     	    entry->bgn_inst->function = entry;
2640     
2641     	    visit_exec_list(&entry->sig->body, &v);
2642     
2643     	    ir_to_mesa_instruction *last;
2644     	    last = (ir_to_mesa_instruction *)v.instructions.get_tail();
2645     	    if (last->op != OPCODE_RET)
2646     	       v.ir_to_mesa_emit_op0(NULL, OPCODE_RET);
2647     
2648     	    ir_to_mesa_instruction *end;
2649     	    end = v.ir_to_mesa_emit_op0(NULL, OPCODE_ENDSUB);
2650     	    end->function = entry;
2651     
2652     	    progress = GL_TRUE;
2653     	 }
2654           }
2655        } while (progress);
2656     
2657        prog->NumTemporaries = v.next_temp;
2658     
2659        int num_instructions = 0;
2660        foreach_iter(exec_list_iterator, iter, v.instructions) {
2661           num_instructions++;
2662        }
2663     
2664        mesa_instructions =
2665           (struct prog_instruction *)calloc(num_instructions,
2666     					sizeof(*mesa_instructions));
2667        mesa_instruction_annotation = talloc_array(v.mem_ctx, ir_instruction *,
2668     					      num_instructions);
2669     
2670        mesa_inst = mesa_instructions;
2671        i = 0;
2672        foreach_iter(exec_list_iterator, iter, v.instructions) {
2673           ir_to_mesa_instruction *inst = (ir_to_mesa_instruction *)iter.get();
2674     
2675           mesa_inst->Opcode = inst->op;
2676           mesa_inst->CondUpdate = inst->cond_update;
2677           mesa_inst->DstReg.File = inst->dst_reg.file;
2678           mesa_inst->DstReg.Index = inst->dst_reg.index;
2679           mesa_inst->DstReg.CondMask = inst->dst_reg.cond_mask;
2680           mesa_inst->DstReg.WriteMask = inst->dst_reg.writemask;
2681           mesa_inst->DstReg.RelAddr = inst->dst_reg.reladdr != NULL;
2682           mesa_inst->SrcReg[0] = mesa_src_reg_from_ir_src_reg(inst->src_reg[0]);
2683           mesa_inst->SrcReg[1] = mesa_src_reg_from_ir_src_reg(inst->src_reg[1]);
2684           mesa_inst->SrcReg[2] = mesa_src_reg_from_ir_src_reg(inst->src_reg[2]);
2685           mesa_inst->TexSrcUnit = inst->sampler;
2686           mesa_inst->TexSrcTarget = inst->tex_target;
2687           mesa_inst->TexShadow = inst->tex_shadow;
2688           mesa_instruction_annotation[i] = inst->ir;
2689     
2690           /* Set IndirectRegisterFiles. */
2691           if (mesa_inst->DstReg.RelAddr)
2692              prog->IndirectRegisterFiles |= 1 << mesa_inst->DstReg.File;
2693     
2694           for (unsigned src = 0; src < 3; src++)
2695              if (mesa_inst->SrcReg[src].RelAddr)
2696                 prog->IndirectRegisterFiles |= 1 << mesa_inst->SrcReg[src].File;
2697     
2698           if (options->EmitNoIfs && mesa_inst->Opcode == OPCODE_IF) {
2699     	 fail_link(shader_program, "Couldn't flatten if statement\n");
2700           }
2701     
2702           switch (mesa_inst->Opcode) {
2703           case OPCODE_BGNSUB:
2704     	 inst->function->inst = i;
2705     	 mesa_inst->Comment = strdup(inst->function->sig->function_name());
2706     	 break;
2707           case OPCODE_ENDSUB:
2708     	 mesa_inst->Comment = strdup(inst->function->sig->function_name());
2709     	 break;
2710           case OPCODE_CAL:
2711     	 mesa_inst->BranchTarget = inst->function->sig_id; /* rewritten later */
2712     	 break;
2713           case OPCODE_ARL:
2714     	 prog->NumAddressRegs = 1;
2715     	 break;
2716           default:
2717     	 break;
2718           }
2719     
2720           mesa_inst++;
2721           i++;
2722        }
2723     
2724        set_branchtargets(&v, mesa_instructions, num_instructions);
2725     
2726        if (ctx->Shader.Flags & GLSL_DUMP) {
2727           printf("\n");
2728           printf("GLSL IR for linked %s program %d:\n", target_string,
2729     	     shader_program->Name);
2730           _mesa_print_ir(shader->ir, NULL);
2731           printf("\n");
2732           printf("\n");
2733           printf("Mesa IR for linked %s program %d:\n", target_string,
2734     	     shader_program->Name);
2735           print_program(mesa_instructions, mesa_instruction_annotation,
2736     		    num_instructions);
2737        }
2738     
2739        prog->Instructions = mesa_instructions;
2740        prog->NumInstructions = num_instructions;
2741     
2742        do_set_program_inouts(shader->ir, prog);
2743        count_resources(prog);
2744     
2745        _mesa_reference_program(ctx, &shader->Program, prog);
2746     
2747        if ((ctx->Shader.Flags & GLSL_NO_OPT) == 0) {
2748           _mesa_optimize_program(ctx, prog);
2749        }
2750     
2751        return prog;
2752     }

2764     GLboolean
2765     _mesa_ir_link_shader(GLcontext *ctx, struct gl_shader_program *prog)
2766     {
2767        assert(prog->LinkStatus);
2768     
2769        for (unsigned i = 0; i < prog->_NumLinkedShaders; i++) {
2770           bool progress;
2771           exec_list *ir = prog->_LinkedShaders[i]->ir;
2772           struct gl_shader_compiler_options *options =
2773                 &ctx->ShaderCompilerOptions[_mesa_shader_type_to_index(prog->_LinkedShaders[i]->Type)];
2774     
2775           do {
2776     	 progress = false;
2777     
2778     	 /* Lowering */
2779     	 do_mat_op_to_vec(ir);
2780     	 do_mod_to_fract(ir);
2781     	 do_div_to_mul_rcp(ir);
2782     	 do_explog_to_explog2(ir);
2783     
2784     	 progress = do_lower_jumps(ir, true, true, options->EmitNoMainReturn, options->EmitNoCont, options->EmitNoLoops) || progress;
2785     
2786     	 progress = do_common_optimization(ir, true, options->MaxUnrollIterations) || progress;
2787     
2788     	 if (options->EmitNoIfs)
2789     	    progress = do_if_to_cond_assign(ir) || progress;
2790     
2791     	 if (options->EmitNoNoise)
2792     	    progress = lower_noise(ir) || progress;
2793     
2794     	 /* If there are forms of indirect addressing that the driver
2795     	  * cannot handle, perform the lowering pass.
2796     	  */
2797     	 if (options->EmitNoIndirectInput || options->EmitNoIndirectOutput
2798     	     || options->EmitNoIndirectTemp || options->EmitNoIndirectUniform)
2799     	   progress =
2800     	     lower_variable_index_to_cond_assign(ir,
2801     						 options->EmitNoIndirectInput,
2802     						 options->EmitNoIndirectOutput,
2803     						 options->EmitNoIndirectTemp,
2804     						 options->EmitNoIndirectUniform)
2805     	     || progress;
2806     
2807     	 progress = do_vec_index_to_cond_assign(ir) || progress;
2808           } while (progress);
2809     
2810           validate_ir_tree(ir);
2811        }
2812     
2813        for (unsigned i = 0; i < prog->_NumLinkedShaders; i++) {
2814           struct gl_program *linked_prog;
2815           bool ok = true;
2816     
2817           linked_prog = get_mesa_program(ctx, prog, prog->_LinkedShaders[i]);
2818     
2819           switch (prog->_LinkedShaders[i]->Type) {
2820           case GL_VERTEX_SHADER:
2821     	 _mesa_reference_vertprog(ctx, &prog->VertexProgram,
2822     				  (struct gl_vertex_program *)linked_prog);
2823     	 ok = ctx->Driver.ProgramStringNotify(ctx, GL_VERTEX_PROGRAM_ARB,
2824     					      linked_prog);
2825     	 break;
2826           case GL_FRAGMENT_SHADER:
2827     	 _mesa_reference_fragprog(ctx, &prog->FragmentProgram,
2828     				  (struct gl_fragment_program *)linked_prog);
2829     	 ok = ctx->Driver.ProgramStringNotify(ctx, GL_FRAGMENT_PROGRAM_ARB,
2830     					      linked_prog);
2831     	 break;
2832           }
2833           if (!ok) {
2834     	 return GL_FALSE;
2835           }
2836           _mesa_reference_program(ctx, &linked_prog, NULL);
2837        }
2838     
2839        return GL_TRUE;
2840     }

2846           new(shader) _mesa_glsl_parse_state(ctx, shader->Type, shader);
2847     
2848        const char *source = shader->Source;
2849        /* Check if the user called glCompileShader without first calling
2850         * glShaderSource.  This should fail to compile, but not raise a GL_ERROR.
2851         */
2852        if (source == NULL) {
2853           shader->CompileStatus = GL_FALSE;
2854           return;
2855        }
2856     
2857        state->error = preprocess(state, &source, &state->info_log,
2858     			     &ctx->Extensions, ctx->API);
2859     
2860        if (ctx->Shader.Flags & GLSL_DUMP) {
2861           printf("GLSL source for shader %d:\n", shader->Name);
2862           printf("%s\n", shader->Source);
2863        }
2864     
2865        if (!state->error) {
2866          _mesa_glsl_lexer_ctor(state, source);
2867          _mesa_glsl_parse(state);
2868          _mesa_glsl_lexer_dtor(state);
2869        }
2870     
2871        talloc_free(shader->ir);
2872        shader->ir = new(shader) exec_list;
2873        if (!state->error && !state->translation_unit.is_empty())
2874           _mesa_ast_to_hir(shader->ir, state);
2875     
2876        if (!state->error && !shader->ir->is_empty()) {
2877           validate_ir_tree(shader->ir);
2878     
2879           /* Do some optimization at compile time to reduce shader IR size
2880            * and reduce later work if the same shader is linked multiple times
2881            */
2882           while (do_common_optimization(shader->ir, false, 32))
2883     	 ;
2884     
2885           validate_ir_tree(shader->ir);
2886        }
2887     
2888        shader->symbols = state->symbols;
2889     
2890        shader->CompileStatus = !state->error;
2891        shader->InfoLog = state->info_log;
2892        shader->Version = state->language_version;
2893        memcpy(shader->builtins_to_link, state->builtins_to_link,
2894     	  sizeof(shader->builtins_to_link[0]) * state->num_builtins_to_link);
2895        shader->num_builtins_to_link = state->num_builtins_to_link;
2896     
2897        if (ctx->Shader.Flags & GLSL_LOG) {
2898           _mesa_write_shader_to_file(shader);
2899        }
2900     
2901        if (ctx->Shader.Flags & GLSL_DUMP) {
2902           if (shader->CompileStatus) {
2903     	 printf("GLSL IR for shader %d:\n", shader->Name);
2904     	 _mesa_print_ir(shader->ir, NULL);
2905     	 printf("\n\n");
2906           } else {
2907     	 printf("GLSL shader %d failed to compile.\n", shader->Name);
2908           }
2909           if (shader->InfoLog && shader->InfoLog[0] != 0) {
2910     	 printf("GLSL shader %d info log:\n", shader->Name);
2911     	 printf("%s\n", shader->InfoLog);
2912           }
2913        }
2914     
2915        /* Retain any live IR, but trash the rest. */
2916        reparent_ir(shader->ir, shader->ir);
2917     
2918        talloc_free(state);
2919     
2920        if (shader->CompileStatus) {
2921           if (!ctx->Driver.CompileShader(ctx, shader))
2922     	 shader->CompileStatus = GL_FALSE;
2923        }
2924     }

2926     void
2927     _mesa_glsl_link_shader(GLcontext *ctx, struct gl_shader_program *prog)
2928     {
2929        unsigned int i;
2930     
2931        _mesa_clear_shader_program_data(ctx, prog);
2932     
2933        prog->LinkStatus = GL_TRUE;
2934     
2935        for (i = 0; i < prog->NumShaders; i++) {
2936           if (!prog->Shaders[i]->CompileStatus) {
2937     	 fail_link(prog, "linking with uncompiled shader");
2938     	 prog->LinkStatus = GL_FALSE;
2939           }
2940        }
2941     
2942        prog->Varying = _mesa_new_parameter_list();
2943        _mesa_reference_vertprog(ctx, &prog->VertexProgram, NULL);
2944        _mesa_reference_fragprog(ctx, &prog->FragmentProgram, NULL);
2945     
2946        if (prog->LinkStatus) {
2947           link_shaders(ctx, prog);
2948        }
2949     
2950        if (prog->LinkStatus) {
2951           if (!ctx->Driver.LinkShader(ctx, prog)) {
2952     	 prog->LinkStatus = GL_FALSE;
2953           }
2954        }
2955     
2956        set_uniform_initializers(ctx, prog);
2957     
2958        if (ctx->Shader.Flags & GLSL_DUMP) {
2959           if (!prog->LinkStatus) {
2960     	 printf("GLSL shader program %d failed to link\n", prog->Name);
2961           }
2962     
2963           if (prog->InfoLog && prog->InfoLog[0] != 0) {
2964     	 printf("GLSL shader program %d info log:\n", prog->Name);
2965     	 printf("%s\n", prog->InfoLog);
2966           }
2967        }
2968     }
