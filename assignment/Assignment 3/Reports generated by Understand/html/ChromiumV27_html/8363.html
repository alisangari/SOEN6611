24     /**
25      * \file ir_constant_expression.cpp
26      * Evaluate and process constant valued expressions
27      *
28      * In GLSL, constant valued expressions are used in several places.  These
29      * must be processed and evaluated very early in the compilation process.
30      *
31      *    * Sizes of arrays
32      *    * Initializers for uniforms
33      *    * Initializers for \c const variables
34      */
35

37     #include "main/core.h" /* for MAX2, MIN2, CLAMP */
38     #include "ir.h"
39     #include "ir_visitor.h"
40     #include "glsl_types.h"
41     #include "safe_strcmp.h"

43     static float
44     dot(ir_constant *op0, ir_constant *op1)
45     {
46        assert(op0->type->is_float() && op1->type->is_float());
47     
48        float result = 0;
49        for (unsigned c = 0; c < op0->type->components(); c++)
50           result += op0->value.f[c] * op1->value.f[c];
51     
52        return result;
53     }
54     
55     ir_constant *
56     ir_expression::constant_expression_value()
57     {
58        ir_constant *op[2] = { NULL, NULL };
59        ir_constant_data data;
60     
61        memset(&data, 0, sizeof(data));
62     
63        for (unsigned operand = 0; operand < this->get_num_operands(); operand++) {
64           op[operand] = this->operands[operand]->constant_expression_value();
65           if (!op[operand])
66     	 return NULL;
67        }
68     
69        if (op[1] != NULL)
70           assert(op[0]->type->base_type == op[1]->type->base_type);
71     
72        bool op0_scalar = op[0]->type->is_scalar();
73        bool op1_scalar = op[1] != NULL && op[1]->type->is_scalar();
74     
75        /* When iterating over a vector or matrix's components, we want to increase
76         * the loop counter.  However, for scalars, we want to stay at 0.
77         */
78        unsigned c0_inc = op0_scalar ? 0 : 1;
79        unsigned c1_inc = op1_scalar ? 0 : 1;
80        unsigned components;
81        if (op1_scalar || !op[1]) {
82           components = op[0]->type->components();
83        } else {
84           components = op[1]->type->components();
85        }
86     
87        void *ctx = talloc_parent(this);
88     
89        /* Handle array operations here, rather than below. */
90        if (op[0]->type->is_array()) {
91           assert(op[1] != NULL && op[1]->type->is_array());
92           switch (this->operation) {
93           case ir_binop_all_equal:
94     	 return new(ctx) ir_constant(op[0]->has_value(op[1]));
95           case ir_binop_any_nequal:
96     	 return new(ctx) ir_constant(!op[0]->has_value(op[1]));
97           default:
98     	 break;
99           }
100           return NULL;
101        }
102     
103        switch (this->operation) {
104        case ir_unop_logic_not:
105           assert(op[0]->type->base_type == GLSL_TYPE_BOOL);
106           for (unsigned c = 0; c < op[0]->type->components(); c++)
107     	 data.b[c] = !op[0]->value.b[c];
108           break;
109     
110        case ir_unop_f2i:
111           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
112           for (unsigned c = 0; c < op[0]->type->components(); c++) {
113     	 data.i[c] = op[0]->value.f[c];
114           }
115           break;
116        case ir_unop_i2f:
117           assert(op[0]->type->base_type == GLSL_TYPE_INT);
118           for (unsigned c = 0; c < op[0]->type->components(); c++) {
119     	 data.f[c] = op[0]->value.i[c];
120           }
121           break;
122        case ir_unop_u2f:
123           assert(op[0]->type->base_type == GLSL_TYPE_UINT);
124           for (unsigned c = 0; c < op[0]->type->components(); c++) {
125     	 data.f[c] = op[0]->value.u[c];
126           }
127           break;
128        case ir_unop_b2f:
129           assert(op[0]->type->base_type == GLSL_TYPE_BOOL);
130           for (unsigned c = 0; c < op[0]->type->components(); c++) {
131     	 data.f[c] = op[0]->value.b[c] ? 1.0 : 0.0;
132           }
133           break;
134        case ir_unop_f2b:
135           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
136           for (unsigned c = 0; c < op[0]->type->components(); c++) {
137     	 data.b[c] = bool(op[0]->value.f[c]);
138           }
139           break;
140        case ir_unop_b2i:
141           assert(op[0]->type->base_type == GLSL_TYPE_BOOL);
142           for (unsigned c = 0; c < op[0]->type->components(); c++) {
143     	 data.u[c] = op[0]->value.b[c] ? 1 : 0;
144           }
145           break;
146        case ir_unop_i2b:
147           assert(op[0]->type->is_integer());
148           for (unsigned c = 0; c < op[0]->type->components(); c++) {
149     	 data.b[c] = bool(op[0]->value.u[c]);
150           }
151           break;
152     
153        case ir_unop_any:
154           assert(op[0]->type->is_boolean());
155           data.b[0] = false;
156           for (unsigned c = 0; c < op[0]->type->components(); c++) {
157     	 if (op[0]->value.b[c])
158     	    data.b[0] = true;
159           }
160           break;
161     
162        case ir_unop_trunc:
163           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
164           for (unsigned c = 0; c < op[0]->type->components(); c++) {
165     	 data.f[c] = truncf(op[0]->value.f[c]);
166           }
167           break;
168     
169        case ir_unop_ceil:
170           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
171           for (unsigned c = 0; c < op[0]->type->components(); c++) {
172     	 data.f[c] = ceilf(op[0]->value.f[c]);
173           }
174           break;
175     
176        case ir_unop_floor:
177           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
178           for (unsigned c = 0; c < op[0]->type->components(); c++) {
179     	 data.f[c] = floorf(op[0]->value.f[c]);
180           }
181           break;
182     
183        case ir_unop_fract:
184           for (unsigned c = 0; c < op[0]->type->components(); c++) {
185     	 switch (this->type->base_type) {
186     	 case GLSL_TYPE_UINT:
187     	    data.u[c] = 0;
188     	    break;
189     	 case GLSL_TYPE_INT:
190     	    data.i[c] = 0;
191     	    break;
192     	 case GLSL_TYPE_FLOAT:
193     	    data.f[c] = op[0]->value.f[c] - floor(op[0]->value.f[c]);
194     	    break;
195     	 default:
196     	    assert(0);
197     	 }
198           }
199           break;
200     
201        case ir_unop_sin:
202           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
203           for (unsigned c = 0; c < op[0]->type->components(); c++) {
204     	 data.f[c] = sinf(op[0]->value.f[c]);
205           }
206           break;
207     
208        case ir_unop_cos:
209           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
210           for (unsigned c = 0; c < op[0]->type->components(); c++) {
211     	 data.f[c] = cosf(op[0]->value.f[c]);
212           }
213           break;
214     
215        case ir_unop_neg:
216           for (unsigned c = 0; c < op[0]->type->components(); c++) {
217     	 switch (this->type->base_type) {
218     	 case GLSL_TYPE_UINT:
219     	    data.u[c] = -op[0]->value.u[c];
220     	    break;
221     	 case GLSL_TYPE_INT:
222     	    data.i[c] = -op[0]->value.i[c];
223     	    break;
224     	 case GLSL_TYPE_FLOAT:
225     	    data.f[c] = -op[0]->value.f[c];
226     	    break;
227     	 default:
228     	    assert(0);
229     	 }
230           }
231           break;
232     
233        case ir_unop_abs:
234           for (unsigned c = 0; c < op[0]->type->components(); c++) {
235     	 switch (this->type->base_type) {
236     	 case GLSL_TYPE_UINT:
237     	    data.u[c] = op[0]->value.u[c];
238     	    break;
239     	 case GLSL_TYPE_INT:
240     	    data.i[c] = op[0]->value.i[c];
241     	    if (data.i[c] < 0)
242     	       data.i[c] = -data.i[c];
243     	    break;
244     	 case GLSL_TYPE_FLOAT:
245     	    data.f[c] = fabs(op[0]->value.f[c]);
246     	    break;
247     	 default:
248     	    assert(0);
249     	 }
250           }
251           break;
252     
253        case ir_unop_sign:
254           for (unsigned c = 0; c < op[0]->type->components(); c++) {
255     	 switch (this->type->base_type) {
256     	 case GLSL_TYPE_UINT:
257     	    data.u[c] = op[0]->value.i[c] > 0;
258     	    break;
259     	 case GLSL_TYPE_INT:
260     	    data.i[c] = (op[0]->value.i[c] > 0) - (op[0]->value.i[c] < 0);
261     	    break;
262     	 case GLSL_TYPE_FLOAT:
263     	    data.f[c] = float((op[0]->value.f[c] > 0)-(op[0]->value.f[c] < 0));
264     	    break;
265     	 default:
266     	    assert(0);
267     	 }
268           }
269           break;
270     
271        case ir_unop_rcp:
272           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
273           for (unsigned c = 0; c < op[0]->type->components(); c++) {
274     	 switch (this->type->base_type) {
275     	 case GLSL_TYPE_UINT:
276     	    if (op[0]->value.u[c] != 0.0)
277     	       data.u[c] = 1 / op[0]->value.u[c];
278     	    break;
279     	 case GLSL_TYPE_INT:
280     	    if (op[0]->value.i[c] != 0.0)
281     	       data.i[c] = 1 / op[0]->value.i[c];
282     	    break;
283     	 case GLSL_TYPE_FLOAT:
284     	    if (op[0]->value.f[c] != 0.0)
285     	       data.f[c] = 1.0 / op[0]->value.f[c];
286     	    break;
287     	 default:
288     	    assert(0);
289     	 }
290           }
291           break;
292     
293        case ir_unop_rsq:
294           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
295           for (unsigned c = 0; c < op[0]->type->components(); c++) {
296     	 data.f[c] = 1.0 / sqrtf(op[0]->value.f[c]);
297           }
298           break;
299     
300        case ir_unop_sqrt:
301           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
302           for (unsigned c = 0; c < op[0]->type->components(); c++) {
303     	 data.f[c] = sqrtf(op[0]->value.f[c]);
304           }
305           break;
306     
307        case ir_unop_exp:
308           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
309           for (unsigned c = 0; c < op[0]->type->components(); c++) {
310     	 data.f[c] = expf(op[0]->value.f[c]);
311           }
312           break;
313     
314        case ir_unop_exp2:
315           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
316           for (unsigned c = 0; c < op[0]->type->components(); c++) {
317     	 data.f[c] = exp2f(op[0]->value.f[c]);
318           }
319           break;
320     
321        case ir_unop_log:
322           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
323           for (unsigned c = 0; c < op[0]->type->components(); c++) {
324     	 data.f[c] = logf(op[0]->value.f[c]);
325           }
326           break;
327     
328        case ir_unop_log2:
329           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
330           for (unsigned c = 0; c < op[0]->type->components(); c++) {
331     	 data.f[c] = log2f(op[0]->value.f[c]);
332           }
333           break;
334     
335        case ir_unop_dFdx:
336        case ir_unop_dFdy:
337           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
338           for (unsigned c = 0; c < op[0]->type->components(); c++) {
339     	 data.f[c] = 0.0;
340           }
341           break;
342     
343        case ir_binop_pow:
344           assert(op[0]->type->base_type == GLSL_TYPE_FLOAT);
345           for (unsigned c = 0; c < op[0]->type->components(); c++) {
346     	 data.f[c] = powf(op[0]->value.f[c], op[1]->value.f[c]);
347           }
348           break;
349     
350        case ir_binop_dot:
351           data.f[0] = dot(op[0], op[1]);
352           break;
353     
354        case ir_binop_min:
355           assert(op[0]->type == op[1]->type || op0_scalar || op1_scalar);
356           for (unsigned c = 0, c0 = 0, c1 = 0;
357     	   c < components;
358     	   c0 += c0_inc, c1 += c1_inc, c++) {
359     
360     	 switch (op[0]->type->base_type) {
361     	 case GLSL_TYPE_UINT:
362     	    data.u[c] = MIN2(op[0]->value.u[c0], op[1]->value.u[c1]);
363     	    break;
364     	 case GLSL_TYPE_INT:
365     	    data.i[c] = MIN2(op[0]->value.i[c0], op[1]->value.i[c1]);
366     	    break;
367     	 case GLSL_TYPE_FLOAT:
368     	    data.f[c] = MIN2(op[0]->value.f[c0], op[1]->value.f[c1]);
369     	    break;
370     	 default:
371     	    assert(0);
372     	 }
373           }
374     
375           break;
376        case ir_binop_max:
377           assert(op[0]->type == op[1]->type || op0_scalar || op1_scalar);
378           for (unsigned c = 0, c0 = 0, c1 = 0;
379     	   c < components;
380     	   c0 += c0_inc, c1 += c1_inc, c++) {
381     
382     	 switch (op[0]->type->base_type) {
383     	 case GLSL_TYPE_UINT:
384     	    data.u[c] = MAX2(op[0]->value.u[c0], op[1]->value.u[c1]);
385     	    break;
386     	 case GLSL_TYPE_INT:
387     	    data.i[c] = MAX2(op[0]->value.i[c0], op[1]->value.i[c1]);
388     	    break;
389     	 case GLSL_TYPE_FLOAT:
390     	    data.f[c] = MAX2(op[0]->value.f[c0], op[1]->value.f[c1]);
391     	    break;
392     	 default:
393     	    assert(0);
394     	 }
395           }
396           break;
397     
398        case ir_binop_cross:
399           assert(op[0]->type == glsl_type::vec3_type);
400           assert(op[1]->type == glsl_type::vec3_type);
401           data.f[0] = (op[0]->value.f[1] * op[1]->value.f[2] -
402     		   op[1]->value.f[1] * op[0]->value.f[2]);
403           data.f[1] = (op[0]->value.f[2] * op[1]->value.f[0] -
404     		   op[1]->value.f[2] * op[0]->value.f[0]);
405           data.f[2] = (op[0]->value.f[0] * op[1]->value.f[1] -
406     		   op[1]->value.f[0] * op[0]->value.f[1]);
407           break;
408     
409        case ir_binop_add:
410           assert(op[0]->type == op[1]->type || op0_scalar || op1_scalar);
411           for (unsigned c = 0, c0 = 0, c1 = 0;
412     	   c < components;
413     	   c0 += c0_inc, c1 += c1_inc, c++) {
414     
415     	 switch (op[0]->type->base_type) {
416     	 case GLSL_TYPE_UINT:
417     	    data.u[c] = op[0]->value.u[c0] + op[1]->value.u[c1];
418     	    break;
419     	 case GLSL_TYPE_INT:
420     	    data.i[c] = op[0]->value.i[c0] + op[1]->value.i[c1];
421     	    break;
422     	 case GLSL_TYPE_FLOAT:
423     	    data.f[c] = op[0]->value.f[c0] + op[1]->value.f[c1];
424     	    break;
425     	 default:
426     	    assert(0);
427     	 }
428           }
429     
430           break;
431        case ir_binop_sub:
432           assert(op[0]->type == op[1]->type || op0_scalar || op1_scalar);
433           for (unsigned c = 0, c0 = 0, c1 = 0;
434     	   c < components;
435     	   c0 += c0_inc, c1 += c1_inc, c++) {
436     
437     	 switch (op[0]->type->base_type) {
438     	 case GLSL_TYPE_UINT:
439     	    data.u[c] = op[0]->value.u[c0] - op[1]->value.u[c1];
440     	    break;
441     	 case GLSL_TYPE_INT:
442     	    data.i[c] = op[0]->value.i[c0] - op[1]->value.i[c1];
443     	    break;
444     	 case GLSL_TYPE_FLOAT:
445     	    data.f[c] = op[0]->value.f[c0] - op[1]->value.f[c1];
446     	    break;
447     	 default:
448     	    assert(0);
449     	 }
450           }
451     
452           break;
453        case ir_binop_mul:
454           /* Check for equal types, or unequal types involving scalars */
455           if ((op[0]->type == op[1]->type && !op[0]->type->is_matrix())
456     	  || op0_scalar || op1_scalar) {
457     	 for (unsigned c = 0, c0 = 0, c1 = 0;
458     	      c < components;
459     	      c0 += c0_inc, c1 += c1_inc, c++) {
460     
461     	    switch (op[0]->type->base_type) {
462     	    case GLSL_TYPE_UINT:
463     	       data.u[c] = op[0]->value.u[c0] * op[1]->value.u[c1];
464     	       break;
465     	    case GLSL_TYPE_INT:
466     	       data.i[c] = op[0]->value.i[c0] * op[1]->value.i[c1];
467     	       break;
468     	    case GLSL_TYPE_FLOAT:
469     	       data.f[c] = op[0]->value.f[c0] * op[1]->value.f[c1];
470     	       break;
471     	    default:
472     	       assert(0);
473     	    }
474     	 }
475           } else {
476     	 assert(op[0]->type->is_matrix() || op[1]->type->is_matrix());
477     
478     	 /* Multiply an N-by-M matrix with an M-by-P matrix.  Since either
479     	  * matrix can be a GLSL vector, either N or P can be 1.
480     	  *
481     	  * For vec*mat, the vector is treated as a row vector.  This
482     	  * means the vector is a 1-row x M-column matrix.
483     	  *
484     	  * For mat*vec, the vector is treated as a column vector.  Since
485     	  * matrix_columns is 1 for vectors, this just works.
486     	  */
487     	 const unsigned n = op[0]->type->is_vector()
488     	    ? 1 : op[0]->type->vector_elements;
489     	 const unsigned m = op[1]->type->vector_elements;
490     	 const unsigned p = op[1]->type->matrix_columns;
491     	 for (unsigned j = 0; j < p; j++) {
492     	    for (unsigned i = 0; i < n; i++) {
493     	       for (unsigned k = 0; k < m; k++) {
494     		  data.f[i+n*j] += op[0]->value.f[i+n*k]*op[1]->value.f[k+m*j];
495     	       }
496     	    }
497     	 }
498           }
499     
500           break;
501        case ir_binop_div:
502           assert(op[0]->type == op[1]->type || op0_scalar || op1_scalar);
503           for (unsigned c = 0, c0 = 0, c1 = 0;
504     	   c < components;
505     	   c0 += c0_inc, c1 += c1_inc, c++) {
506     
507     	 switch (op[0]->type->base_type) {
508     	 case GLSL_TYPE_UINT:
509     	    data.u[c] = op[0]->value.u[c0] / op[1]->value.u[c1];
510     	    break;
511     	 case GLSL_TYPE_INT:
512     	    data.i[c] = op[0]->value.i[c0] / op[1]->value.i[c1];
513     	    break;
514     	 case GLSL_TYPE_FLOAT:
515     	    data.f[c] = op[0]->value.f[c0] / op[1]->value.f[c1];
516     	    break;
517     	 default:
518     	    assert(0);
519     	 }
520           }
521     
522           break;
523        case ir_binop_mod:
524           assert(op[0]->type == op[1]->type || op0_scalar || op1_scalar);
525           for (unsigned c = 0, c0 = 0, c1 = 0;
526     	   c < components;
527     	   c0 += c0_inc, c1 += c1_inc, c++) {
528     
529     	 switch (op[0]->type->base_type) {
530     	 case GLSL_TYPE_UINT:
531     	    data.u[c] = op[0]->value.u[c0] % op[1]->value.u[c1];
532     	    break;
533     	 case GLSL_TYPE_INT:
534     	    data.i[c] = op[0]->value.i[c0] % op[1]->value.i[c1];
535     	    break;
536     	 case GLSL_TYPE_FLOAT:
537     	    /* We don't use fmod because it rounds toward zero; GLSL specifies
538     	     * the use of floor.
539     	     */
540     	    data.f[c] = op[0]->value.f[c0] - op[1]->value.f[c1]
541     	       * floorf(op[0]->value.f[c0] / op[1]->value.f[c1]);
542     	    break;
543     	 default:
544     	    assert(0);
545     	 }
546           }
547     
548           break;
549     
550        case ir_binop_logic_and:
551           assert(op[0]->type->base_type == GLSL_TYPE_BOOL);
552           for (unsigned c = 0; c < op[0]->type->components(); c++)
553     	 data.b[c] = op[0]->value.b[c] && op[1]->value.b[c];
554           break;
555        case ir_binop_logic_xor:
556           assert(op[0]->type->base_type == GLSL_TYPE_BOOL);
557           for (unsigned c = 0; c < op[0]->type->components(); c++)
558     	 data.b[c] = op[0]->value.b[c] ^ op[1]->value.b[c];
559           break;
560        case ir_binop_logic_or:
561           assert(op[0]->type->base_type == GLSL_TYPE_BOOL);
562           for (unsigned c = 0; c < op[0]->type->components(); c++)
563     	 data.b[c] = op[0]->value.b[c] || op[1]->value.b[c];
564           break;
565     
566        case ir_binop_less:
567           switch (op[0]->type->base_type) {
568           case GLSL_TYPE_UINT:
569     	 data.b[0] = op[0]->value.u[0] < op[1]->value.u[0];
570     	 break;
571           case GLSL_TYPE_INT:
572     	 data.b[0] = op[0]->value.i[0] < op[1]->value.i[0];
573     	 break;
574           case GLSL_TYPE_FLOAT:
575     	 data.b[0] = op[0]->value.f[0] < op[1]->value.f[0];
576     	 break;
577           default:
578     	 assert(0);
579           }
580           break;
581        case ir_binop_greater:
582           switch (op[0]->type->base_type) {
583           case GLSL_TYPE_UINT:
584     	 data.b[0] = op[0]->value.u[0] > op[1]->value.u[0];
585     	 break;
586           case GLSL_TYPE_INT:
587     	 data.b[0] = op[0]->value.i[0] > op[1]->value.i[0];
588     	 break;
589           case GLSL_TYPE_FLOAT:
590     	 data.b[0] = op[0]->value.f[0] > op[1]->value.f[0];
591     	 break;
592           default:
593     	 assert(0);
594           }
595           break;
596        case ir_binop_lequal:
597           switch (op[0]->type->base_type) {
598           case GLSL_TYPE_UINT:
599     	 data.b[0] = op[0]->value.u[0] <= op[1]->value.u[0];
600     	 break;
601           case GLSL_TYPE_INT:
602     	 data.b[0] = op[0]->value.i[0] <= op[1]->value.i[0];
603     	 break;
604           case GLSL_TYPE_FLOAT:
605     	 data.b[0] = op[0]->value.f[0] <= op[1]->value.f[0];
606     	 break;
607           default:
608     	 assert(0);
609           }
610           break;
611        case ir_binop_gequal:
612           switch (op[0]->type->base_type) {
613           case GLSL_TYPE_UINT:
614     	 data.b[0] = op[0]->value.u[0] >= op[1]->value.u[0];
615     	 break;
616           case GLSL_TYPE_INT:
617     	 data.b[0] = op[0]->value.i[0] >= op[1]->value.i[0];
618     	 break;
619           case GLSL_TYPE_FLOAT:
620     	 data.b[0] = op[0]->value.f[0] >= op[1]->value.f[0];
621     	 break;
622           default:
623     	 assert(0);
624           }
625           break;
626        case ir_binop_equal:
627           assert(op[0]->type == op[1]->type);
628           for (unsigned c = 0; c < components; c++) {
629     	 switch (op[0]->type->base_type) {
630     	 case GLSL_TYPE_UINT:
631     	    data.b[c] = op[0]->value.u[c] == op[1]->value.u[c];
632     	    break;
633     	 case GLSL_TYPE_INT:
634     	    data.b[c] = op[0]->value.i[c] == op[1]->value.i[c];
635     	    break;
636     	 case GLSL_TYPE_FLOAT:
637     	    data.b[c] = op[0]->value.f[c] == op[1]->value.f[c];
638     	    break;
639     	 default:
640     	    assert(0);
641     	 }
642           }
643           break;
644        case ir_binop_nequal:
645           assert(op[0]->type != op[1]->type);
646           for (unsigned c = 0; c < components; c++) {
647     	 switch (op[0]->type->base_type) {
648     	 case GLSL_TYPE_UINT:
649     	    data.b[c] = op[0]->value.u[c] != op[1]->value.u[c];
650     	    break;
651     	 case GLSL_TYPE_INT:
652     	    data.b[c] = op[0]->value.i[c] != op[1]->value.i[c];
653     	    break;
654     	 case GLSL_TYPE_FLOAT:
655     	    data.b[c] = op[0]->value.f[c] != op[1]->value.f[c];
656     	    break;
657     	 default:
658     	    assert(0);
659     	 }
660           }
661           break;
662        case ir_binop_all_equal:
663           data.b[0] = op[0]->has_value(op[1]);
664           break;
665        case ir_binop_any_nequal:
666           data.b[0] = !op[0]->has_value(op[1]);
667           break;
668     
669        default:
670           /* FINISHME: Should handle all expression types. */
671           return NULL;
672        }
673     
674        return new(ctx) ir_constant(this->type, &data);
675     }

687     ir_swizzle::constant_expression_value()
688     {
689        ir_constant *v = this->val->constant_expression_value();
690     
691        if (v != NULL) {
692           ir_constant_data data = { { 0 } };
693     
694           const unsigned swiz_idx[4] = {
695     	 this->mask.x, this->mask.y, this->mask.z, this->mask.w
696           };
697     
698           for (unsigned i = 0; i < this->mask.num_components; i++) {
699     	 switch (v->type->base_type) {
700     	 case GLSL_TYPE_UINT:
701     	 case GLSL_TYPE_INT:   data.u[i] = v->value.u[swiz_idx[i]]; break;
702     	 case GLSL_TYPE_FLOAT: data.f[i] = v->value.f[swiz_idx[i]]; break;
703     	 case GLSL_TYPE_BOOL:  data.b[i] = v->value.b[swiz_idx[i]]; break;
704     	 default:              assert(!"Should not get here."); break;
705     	 }
706           }
707     
708           void *ctx = talloc_parent(this);
709           return new(ctx) ir_constant(this->type, &data);
710        }
711        return NULL;
712     }

735     ir_constant *
736     ir_dereference_array::constant_expression_value()
737     {
738        ir_constant *array = this->array->constant_expression_value();
739        ir_constant *idx = this->array_index->constant_expression_value();
740     
741        if ((array != NULL) && (idx != NULL)) {
742           void *ctx = talloc_parent(this);
743           if (array->type->is_matrix()) {
744     	 /* Array access of a matrix results in a vector.
745     	  */
746     	 const unsigned column = idx->value.u[0];
747     
748     	 const glsl_type *const column_type = array->type->column_type();
749     
750     	 /* Offset in the constant matrix to the first element of the column
751     	  * to be extracted.
752     	  */
753     	 const unsigned mat_idx = column * column_type->vector_elements;
754     
755     	 ir_constant_data data;
756     
757     	 switch (column_type->base_type) {
758     	 case GLSL_TYPE_UINT:
759     	 case GLSL_TYPE_INT:
760     	    for (unsigned i = 0; i < column_type->vector_elements; i++)
761     	       data.u[i] = array->value.u[mat_idx + i];
762     
763     	    break;
764     
765     	 case GLSL_TYPE_FLOAT:
766     	    for (unsigned i = 0; i < column_type->vector_elements; i++)
767     	       data.f[i] = array->value.f[mat_idx + i];
768     
769     	    break;
770     
771     	 default:
772     	    assert(!"Should not get here.");
773     	    break;
774     	 }
775     
776     	 return new(ctx) ir_constant(column_type, &data);
777           } else if (array->type->is_vector()) {
778     	 const unsigned component = idx->value.u[0];
779     
780     	 return new(ctx) ir_constant(array, component);
781           } else {
782     	 const unsigned index = idx->value.u[0];
783     	 return array->get_array_element(index)->clone(ctx, NULL);
784           }
785        }
786        return NULL;
787     }

813     
814     ir_constant *
815     ir_call::constant_expression_value()
816     {
817        if (this->type == glsl_type::error_type)
818           return NULL;
819     
820        /* From the GLSL 1.20 spec, page 23:
821         * "Function calls to user-defined functions (non-built-in functions)
822         *  cannot be used to form constant expressions."
823         */
824        if (!this->callee->is_builtin)
825           return NULL;
826     
827        unsigned num_parameters = 0;
828     
829        /* Check if all parameters are constant */
830        ir_constant *op[3];
831        foreach_list(n, &this->actual_parameters) {
832           ir_constant *constant = ((ir_rvalue *) n)->constant_expression_value();
833           if (constant == NULL)
834     	 return NULL;
835     
836           op[num_parameters] = constant;
837     
838           assert(num_parameters < 3);
839           num_parameters++;
840        }
841     
842        /* Individual cases below can either:
843         * - Assign "expr" a new ir_expression to evaluate (for basic opcodes)
844         * - Fill "data" with appopriate constant data
845         * - Return an ir_constant directly.
846         */
847        void *mem_ctx = talloc_parent(this);
848        ir_expression *expr = NULL;
849     
850        ir_constant_data data;
851        memset(&data, 0, sizeof(data));
852     
853        const char *callee = this->callee_name();
854        if (safe_strcmp(callee, "abs") == 0) {
855           expr = new(mem_ctx) ir_expression(ir_unop_abs, type, op[0], NULL);
856        } else if (safe_strcmp(callee, "all") == 0) {
857           assert(op[0]->type->is_boolean());
858           for (unsigned c = 0; c < op[0]->type->components(); c++) {
859     	 if (!op[0]->value.b[c])
860     	    return new(mem_ctx) ir_constant(false);
861           }
862           return new(mem_ctx) ir_constant(true);
863        } else if (safe_strcmp(callee, "any") == 0) {
864           assert(op[0]->type->is_boolean());
865           for (unsigned c = 0; c < op[0]->type->components(); c++) {
866     	 if (op[0]->value.b[c])
867     	    return new(mem_ctx) ir_constant(true);
868           }
869           return new(mem_ctx) ir_constant(false);
870        } else if (safe_strcmp(callee, "acos") == 0) {
871           assert(op[0]->type->is_float());
872           for (unsigned c = 0; c < op[0]->type->components(); c++)
873     	 data.f[c] = acosf(op[0]->value.f[c]);
874        } else if (safe_strcmp(callee, "asin") == 0) {
875           assert(op[0]->type->is_float());
876           for (unsigned c = 0; c < op[0]->type->components(); c++)
877     	 data.f[c] = asinf(op[0]->value.f[c]);
878        } else if (safe_strcmp(callee, "atan") == 0) {
879           assert(op[0]->type->is_float());
880           if (num_parameters == 2) {
881     	 assert(op[1]->type->is_float());
882     	 for (unsigned c = 0; c < op[0]->type->components(); c++)
883     	    data.f[c] = atan2f(op[0]->value.f[c], op[1]->value.f[c]);
884           } else {
885     	 for (unsigned c = 0; c < op[0]->type->components(); c++)
886     	    data.f[c] = atanf(op[0]->value.f[c]);
887           }
888        } else if (safe_strcmp(callee, "dFdx") == 0 || safe_strcmp(callee, "dFdy") == 0) {
889           return ir_constant::zero(mem_ctx, this->type);
890        } else if (safe_strcmp(callee, "ceil") == 0) {
891           expr = new(mem_ctx) ir_expression(ir_unop_ceil, type, op[0], NULL);
892        } else if (safe_strcmp(callee, "clamp") == 0) {
893           assert(num_parameters == 3);
894           unsigned c1_inc = op[1]->type->is_scalar() ? 0 : 1;
895           unsigned c2_inc = op[2]->type->is_scalar() ? 0 : 1;
896           for (unsigned c = 0, c1 = 0, c2 = 0;
897     	   c < op[0]->type->components();
898     	   c1 += c1_inc, c2 += c2_inc, c++) {
899     
900     	 switch (op[0]->type->base_type) {
901     	 case GLSL_TYPE_UINT:
902     	    data.u[c] = CLAMP(op[0]->value.u[c], op[1]->value.u[c1],
903     			      op[2]->value.u[c2]);
904     	    break;
905     	 case GLSL_TYPE_INT:
906     	    data.i[c] = CLAMP(op[0]->value.i[c], op[1]->value.i[c1],
907     			      op[2]->value.i[c2]);
908     	    break;
909     	 case GLSL_TYPE_FLOAT:
910     	    data.f[c] = CLAMP(op[0]->value.f[c], op[1]->value.f[c1],
911     			      op[2]->value.f[c2]);
912     	    break;
913     	 default:
914     	    assert(!"Should not get here.");
915     	 }
916           }
917        } else if (safe_strcmp(callee, "cos") == 0) {
918           expr = new(mem_ctx) ir_expression(ir_unop_cos, type, op[0], NULL);
919        } else if (safe_strcmp(callee, "cosh") == 0) {
920           assert(op[0]->type->is_float());
921           for (unsigned c = 0; c < op[0]->type->components(); c++)
922     	 data.f[c] = coshf(op[0]->value.f[c]);
923        } else if (safe_strcmp(callee, "cross") == 0) {
924           expr = new(mem_ctx) ir_expression(ir_binop_cross, type, op[0], op[1]);
925        } else if (safe_strcmp(callee, "degrees") == 0) {
926           assert(op[0]->type->is_float());
927           for (unsigned c = 0; c < op[0]->type->components(); c++)
928     	 data.f[c] = 180.0/M_PI * op[0]->value.f[c];
929        } else if (safe_strcmp(callee, "distance") == 0) {
930           assert(op[0]->type->is_float() && op[1]->type->is_float());
931           float length_squared = 0.0;
932           for (unsigned c = 0; c < op[0]->type->components(); c++) {
933     	 float t = op[0]->value.f[c] - op[1]->value.f[c];
934     	 length_squared += t * t;
935           }
936           return new(mem_ctx) ir_constant(sqrtf(length_squared));
937        } else if (safe_strcmp(callee, "dot") == 0) {
938           return new(mem_ctx) ir_constant(dot(op[0], op[1]));
939        } else if (safe_strcmp(callee, "equal") == 0) {
940           assert(op[0]->type->is_vector() && op[1] && op[1]->type->is_vector());
941           for (unsigned c = 0; c < op[0]->type->components(); c++) {
942     	 switch (op[0]->type->base_type) {
943     	 case GLSL_TYPE_UINT:
944     	    data.b[c] = op[0]->value.u[c] == op[1]->value.u[c];
945     	    break;
946     	 case GLSL_TYPE_INT:
947     	    data.b[c] = op[0]->value.i[c] == op[1]->value.i[c];
948     	    break;
949     	 case GLSL_TYPE_FLOAT:
950     	    data.b[c] = op[0]->value.f[c] == op[1]->value.f[c];
951     	    break;
952     	 case GLSL_TYPE_BOOL:
953     	    data.b[c] = op[0]->value.b[c] == op[1]->value.b[c];
954     	    break;
955     	 default:
956     	    assert(!"Should not get here.");
957     	 }
958           }
959        } else if (safe_strcmp(callee, "exp") == 0) {
960           expr = new(mem_ctx) ir_expression(ir_unop_exp, type, op[0], NULL);
961        } else if (safe_strcmp(callee, "exp2") == 0) {
962           expr = new(mem_ctx) ir_expression(ir_unop_exp2, type, op[0], NULL);
963        } else if (safe_strcmp(callee, "faceforward") == 0) {
964           if (dot(op[2], op[1]) < 0)
965     	 return op[0];
966           for (unsigned c = 0; c < op[0]->type->components(); c++)
967     	 data.f[c] = -op[0]->value.f[c];
968        } else if (safe_strcmp(callee, "floor") == 0) {
969           expr = new(mem_ctx) ir_expression(ir_unop_floor, type, op[0], NULL);
970        } else if (safe_strcmp(callee, "fract") == 0) {
971           expr = new(mem_ctx) ir_expression(ir_unop_fract, type, op[0], NULL);
972        } else if (safe_strcmp(callee, "fwidth") == 0) {
973           return ir_constant::zero(mem_ctx, this->type);
974        } else if (safe_strcmp(callee, "greaterThan") == 0) {
975           assert(op[0]->type->is_vector() && op[1] && op[1]->type->is_vector());
976           for (unsigned c = 0; c < op[0]->type->components(); c++) {
977     	 switch (op[0]->type->base_type) {
978     	 case GLSL_TYPE_UINT:
979     	    data.b[c] = op[0]->value.u[c] > op[1]->value.u[c];
980     	    break;
981     	 case GLSL_TYPE_INT:
982     	    data.b[c] = op[0]->value.i[c] > op[1]->value.i[c];
983     	    break;
984     	 case GLSL_TYPE_FLOAT:
985     	    data.b[c] = op[0]->value.f[c] > op[1]->value.f[c];
986     	    break;
987     	 default:
988     	    assert(!"Should not get here.");
989     	 }
990           }
991        } else if (safe_strcmp(callee, "greaterThanEqual") == 0) {
992           assert(op[0]->type->is_vector() && op[1] && op[1]->type->is_vector());
993           for (unsigned c = 0; c < op[0]->type->components(); c++) {
994     	 switch (op[0]->type->base_type) {
995     	 case GLSL_TYPE_UINT:
996     	    data.b[c] = op[0]->value.u[c] >= op[1]->value.u[c];
997     	    break;
998     	 case GLSL_TYPE_INT:
999     	    data.b[c] = op[0]->value.i[c] >= op[1]->value.i[c];
1000     	    break;
1001     	 case GLSL_TYPE_FLOAT:
1002     	    data.b[c] = op[0]->value.f[c] >= op[1]->value.f[c];
1003     	    break;
1004     	 default:
1005     	    assert(!"Should not get here.");
1006     	 }
1007           }
1008        } else if (safe_strcmp(callee, "inversesqrt") == 0) {
1009           expr = new(mem_ctx) ir_expression(ir_unop_rsq, type, op[0], NULL);
1010        } else if (safe_strcmp(callee, "length") == 0) {
1011           return new(mem_ctx) ir_constant(sqrtf(dot(op[0], op[0])));
1012        } else if (safe_strcmp(callee, "lessThan") == 0) {
1013           assert(op[0]->type->is_vector() && op[1] && op[1]->type->is_vector());
1014           for (unsigned c = 0; c < op[0]->type->components(); c++) {
1015     	 switch (op[0]->type->base_type) {
1016     	 case GLSL_TYPE_UINT:
1017     	    data.b[c] = op[0]->value.u[c] < op[1]->value.u[c];
1018     	    break;
1019     	 case GLSL_TYPE_INT:
1020     	    data.b[c] = op[0]->value.i[c] < op[1]->value.i[c];
1021     	    break;
1022     	 case GLSL_TYPE_FLOAT:
1023     	    data.b[c] = op[0]->value.f[c] < op[1]->value.f[c];
1024     	    break;
1025     	 default:
1026     	    assert(!"Should not get here.");
1027     	 }
1028           }
1029        } else if (safe_strcmp(callee, "lessThanEqual") == 0) {
1030           assert(op[0]->type->is_vector() && op[1] && op[1]->type->is_vector());
1031           for (unsigned c = 0; c < op[0]->type->components(); c++) {
1032     	 switch (op[0]->type->base_type) {
1033     	 case GLSL_TYPE_UINT:
1034     	    data.b[c] = op[0]->value.u[c] <= op[1]->value.u[c];
1035     	    break;
1036     	 case GLSL_TYPE_INT:
1037     	    data.b[c] = op[0]->value.i[c] <= op[1]->value.i[c];
1038     	    break;
1039     	 case GLSL_TYPE_FLOAT:
1040     	    data.b[c] = op[0]->value.f[c] <= op[1]->value.f[c];
1041     	    break;
1042     	 default:
1043     	    assert(!"Should not get here.");
1044     	 }
1045           }
1046        } else if (safe_strcmp(callee, "log") == 0) {
1047           expr = new(mem_ctx) ir_expression(ir_unop_log, type, op[0], NULL);
1048        } else if (safe_strcmp(callee, "log2") == 0) {
1049           expr = new(mem_ctx) ir_expression(ir_unop_log2, type, op[0], NULL);
1050        } else if (safe_strcmp(callee, "matrixCompMult") == 0) {
1051           assert(op[0]->type->is_float() && op[1]->type->is_float());
1052           for (unsigned c = 0; c < op[0]->type->components(); c++)
1053     	 data.f[c] = op[0]->value.f[c] * op[1]->value.f[c];
1054        } else if (safe_strcmp(callee, "max") == 0) {
1055           expr = new(mem_ctx) ir_expression(ir_binop_max, type, op[0], op[1]);
1056        } else if (safe_strcmp(callee, "min") == 0) {
1057           expr = new(mem_ctx) ir_expression(ir_binop_min, type, op[0], op[1]);
1058        } else if (safe_strcmp(callee, "mix") == 0) {
1059           assert(op[0]->type->is_float() && op[1]->type->is_float());
1060           if (op[2]->type->is_float()) {
1061     	 unsigned c2_inc = op[2]->type->is_scalar() ? 0 : 1;
1062     	 unsigned components = op[0]->type->components();
1063     	 for (unsigned c = 0, c2 = 0; c < components; c2 += c2_inc, c++) {
1064     	    data.f[c] = op[0]->value.f[c] * (1 - op[2]->value.f[c2]) +
1065     			op[1]->value.f[c] * op[2]->value.f[c2];
1066     	 }
1067           } else {
1068     	 assert(op[2]->type->is_boolean());
1069     	 for (unsigned c = 0; c < op[0]->type->components(); c++)
1070     	    data.f[c] = op[op[2]->value.b[c] ? 1 : 0]->value.f[c];
1071           }
1072        } else if (safe_strcmp(callee, "mod") == 0) {
1073           expr = new(mem_ctx) ir_expression(ir_binop_mod, type, op[0], op[1]);
1074        } else if (safe_strcmp(callee, "normalize") == 0) {
1075           assert(op[0]->type->is_float());
1076           float length = sqrtf(dot(op[0], op[0]));
1077     
1078           if (length == 0)
1079     	 return ir_constant::zero(mem_ctx, this->type);
1080     
1081           for (unsigned c = 0; c < op[0]->type->components(); c++)
1082     	 data.f[c] = op[0]->value.f[c] / length;
1083        } else if (safe_strcmp(callee, "not") == 0) {
1084           expr = new(mem_ctx) ir_expression(ir_unop_logic_not, type, op[0], NULL);
1085        } else if (safe_strcmp(callee, "notEqual") == 0) {
1086           assert(op[0]->type->is_vector() && op[1] && op[1]->type->is_vector());
1087           for (unsigned c = 0; c < op[0]->type->components(); c++) {
1088     	 switch (op[0]->type->base_type) {
1089     	 case GLSL_TYPE_UINT:
1090     	    data.b[c] = op[0]->value.u[c] != op[1]->value.u[c];
1091     	    break;
1092     	 case GLSL_TYPE_INT:
1093     	    data.b[c] = op[0]->value.i[c] != op[1]->value.i[c];
1094     	    break;
1095     	 case GLSL_TYPE_FLOAT:
1096     	    data.b[c] = op[0]->value.f[c] != op[1]->value.f[c];
1097     	    break;
1098     	 case GLSL_TYPE_BOOL:
1099     	    data.b[c] = op[0]->value.b[c] != op[1]->value.b[c];
1100     	    break;
1101     	 default:
1102     	    assert(!"Should not get here.");
1103     	 }
1104           }
1105        } else if (safe_strcmp(callee, "outerProduct") == 0) {
1106           assert(op[0]->type->is_vector() && op[1]->type->is_vector());
1107           const unsigned m = op[0]->type->vector_elements;
1108           const unsigned n = op[1]->type->vector_elements;
1109           for (unsigned j = 0; j < n; j++) {
1110     	 for (unsigned i = 0; i < m; i++) {
1111     	    data.f[i+m*j] = op[0]->value.f[i] * op[1]->value.f[j];
1112     	 }
1113           }
1114        } else if (safe_strcmp(callee, "pow") == 0) {
1115           expr = new(mem_ctx) ir_expression(ir_binop_pow, type, op[0], op[1]);
1116        } else if (safe_strcmp(callee, "radians") == 0) {
1117           assert(op[0]->type->is_float());
1118           for (unsigned c = 0; c < op[0]->type->components(); c++)
1119     	 data.f[c] = M_PI/180.0 * op[0]->value.f[c];
1120        } else if (safe_strcmp(callee, "reflect") == 0) {
1121           assert(op[0]->type->is_float());
1122           float dot_NI = dot(op[1], op[0]);
1123           for (unsigned c = 0; c < op[0]->type->components(); c++)
1124     	 data.f[c] = op[0]->value.f[c] - 2 * dot_NI * op[1]->value.f[c];
1125        } else if (safe_strcmp(callee, "refract") == 0) {
1126           const float eta = op[2]->value.f[0];
1127           const float dot_NI = dot(op[1], op[0]);
1128           const float k = 1.0 - eta * eta * (1.0 - dot_NI * dot_NI);
1129           if (k < 0.0) {
1130     	 return ir_constant::zero(mem_ctx, this->type);
1131           } else {
1132     	 for (unsigned c = 0; c < type->components(); c++) {
1133     	    data.f[c] = eta * op[0]->value.f[c] - (eta * dot_NI + sqrtf(k))
1134     			    * op[1]->value.f[c];
1135     	 }
1136           }
1137        } else if (safe_strcmp(callee, "sign") == 0) {
1138           expr = new(mem_ctx) ir_expression(ir_unop_sign, type, op[0], NULL);
1139        } else if (safe_strcmp(callee, "sin") == 0) {
1140           expr = new(mem_ctx) ir_expression(ir_unop_sin, type, op[0], NULL);
1141        } else if (safe_strcmp(callee, "sinh") == 0) {
1142           assert(op[0]->type->is_float());
1143           for (unsigned c = 0; c < op[0]->type->components(); c++)
1144     	 data.f[c] = sinhf(op[0]->value.f[c]);
1145        } else if (safe_strcmp(callee, "smoothstep") == 0) {
1146           assert(num_parameters == 3);
1147           assert(op[1]->type == op[0]->type);
1148           unsigned edge_inc = op[0]->type->is_scalar() ? 0 : 1;
1149           for (unsigned c = 0, e = 0; c < type->components(); e += edge_inc, c++) {
1150     	 const float edge0 = op[0]->value.f[e];
1151     	 const float edge1 = op[1]->value.f[e];
1152     	 if (edge0 == edge1) {
1153     	    data.f[c] = 0.0; /* Avoid a crash - results are undefined anyway */
1154     	 } else {
1155     	    const float numerator = op[2]->value.f[c] - edge0;
1156     	    const float denominator = edge1 - edge0;
1157     	    const float t = CLAMP(numerator/denominator, 0, 1);
1158     	    data.f[c] = t * t * (3 - 2 * t);
1159     	 }
1160           }
1161        } else if (safe_strcmp(callee, "sqrt") == 0) {
1162           expr = new(mem_ctx) ir_expression(ir_unop_sqrt, type, op[0], NULL);
1163        } else if (safe_strcmp(callee, "step") == 0) {
1164           assert(op[0]->type->is_float() && op[1]->type->is_float());
1165           /* op[0] (edge) may be either a scalar or a vector */
1166           const unsigned c0_inc = op[0]->type->is_scalar() ? 0 : 1;
1167           for (unsigned c = 0, c0 = 0; c < type->components(); c0 += c0_inc, c++)
1168     	 data.f[c] = (op[1]->value.f[c] < op[0]->value.f[c0]) ? 0.0 : 1.0;
1169        } else if (safe_strcmp(callee, "tan") == 0) {
1170           assert(op[0]->type->is_float());
1171           for (unsigned c = 0; c < op[0]->type->components(); c++)
1172     	 data.f[c] = tanf(op[0]->value.f[c]);
1173        } else if (safe_strcmp(callee, "tanh") == 0) {
1174           assert(op[0]->type->is_float());
1175           for (unsigned c = 0; c < op[0]->type->components(); c++)
1176     	 data.f[c] = tanhf(op[0]->value.f[c]);
1177        } else if (safe_strcmp(callee, "transpose") == 0) {
1178           assert(op[0]->type->is_matrix());
1179           const unsigned n = op[0]->type->vector_elements;
1180           const unsigned m = op[0]->type->matrix_columns;
1181           for (unsigned j = 0; j < m; j++) {
1182     	 for (unsigned i = 0; i < n; i++) {
1183     	    data.f[m*i+j] += op[0]->value.f[i+n*j];
1184     	 }
1185           }
1186        } else {
1187           /* Unsupported builtin - some are not allowed in constant expressions. */
1188           return NULL;
1189        }
1190     
1191        if (expr != NULL)
1192           return expr->constant_expression_value();
1193     
1194        return new(mem_ctx) ir_constant(this->type, &data);
1195     }
