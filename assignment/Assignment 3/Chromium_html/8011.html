30     #include "PluginTest.h"
31     
32     using namespace std;
33     
34     // A test where the plug-ins scriptable object either has or doesn't have an invokeDefault function.
35     class PluginScriptableNPObjectInvokeDefault : public PluginTest {
36     public:
37         PluginScriptableNPObjectInvokeDefault(NPP npp, const string& identifier)
38             : PluginTest(npp, identifier)
39         {
40         }
41     
42     private:
43         struct NPObjectWithoutInvokeDefault : Object<NPObjectWithoutInvokeDefault> { };
44     
45         struct NPObjectWithInvokeDefault : Object<NPObjectWithInvokeDefault> {
46         public:
47             bool invokeDefault(const NPVariant*, uint32_t, NPVariant* result)
48             {
49                 INT32_TO_NPVARIANT(1, *result);
50                 return true;
51             }
52         };
53     
54         virtual NPError NPP_GetValue(NPPVariable variable, void *value) OVERRIDE
55         {
56             if (variable != NPPVpluginScriptableNPObject)
57                 return NPERR_GENERIC_ERROR;
58     
59             NPObject* object;
60             if (identifier() == "plugin-scriptable-npobject-invoke-default")
61                 object = NPObjectWithInvokeDefault::create(this);
62             else
63                 object = NPObjectWithoutInvokeDefault::create(this);
64     
65             *(NPObject**)value = object;
66     
67             return NPERR_NO_ERROR;
68         }
69     };
70     
71     static PluginTest::Register<PluginScriptableNPObjectInvokeDefault> pluginScriptableNPObjectInvokeDefault("plugin-scriptable-npobject-invoke-default");
72     static PluginTest::Register<PluginScriptableNPObjectInvokeDefault> pluginScriptableNPObjectNoInvokeDefault("plugin-scriptable-npobject-no-invoke-default");
