30     #include "PluginTest.h"
31     
32     #include "PluginObject.h"
33     
34     using namespace std;
35     
36     // Passing null for our NPP_GetValue function pointer should not crash.
37     
38     class NullNPPGetValuePointer : public PluginTest {
39     public:
40         NullNPPGetValuePointer(NPP, const string& identifier);
41     
42     private:
43         virtual NPError NPP_Destroy(NPSavedData**) OVERRIDE;
44         virtual NPError NPP_GetValue(NPPVariable, void* value) OVERRIDE;
45     
46         NPP_GetValueProcPtr m_originalNPPGetValuePointer;
47     };
48     
49     static PluginTest::Register<NullNPPGetValuePointer> registrar("null-npp-getvalue-pointer");
50     
51     NullNPPGetValuePointer::NullNPPGetValuePointer(NPP npp, const string& identifier)
52         : PluginTest(npp, identifier)
53         , m_originalNPPGetValuePointer(pluginFunctions->getvalue)
54     {
55         // Be sneaky and null out the getvalue pointer the browser is holding. This simulates a plugin
56         // that doesn't implement NPP_GetValue (like Shockwave Director 10.3 on Windows). Note that if
57         // WebKit copies the NPPluginFuncs struct this technique will have no effect and WebKit will
58         // call into our NPP_GetValue implementation.
59         pluginFunctions->getvalue = 0;
60     }
61     
62     NPError NullNPPGetValuePointer::NPP_Destroy(NPSavedData**)
63     {
64         // Set the NPP_GetValue pointer back the way it was before we mucked with it so we don't mess
65         // up future uses of the plugin module.
66         pluginFunctions->getvalue = m_originalNPPGetValuePointer;
67         return NPERR_NO_ERROR;
68     }
69     
70     NPError NullNPPGetValuePointer::NPP_GetValue(NPPVariable, void*)
71     {
72         pluginLog(m_npp, "NPP_GetValue was called but should not have been. Maybe WebKit copied the NPPluginFuncs struct, which would invalidate this test.");
73         return NPERR_GENERIC_ERROR;
74     }
